              <p>I have implemented quickly the quick sort algorithm in JavaScript using prototype based style.</p> 
              <p>The code is rather short (<a href="http://jsfiddle.net/b8mfY/" rel="nofollow">http://jsfiddle.net/b8mfY/</a>):</p> 
              <pre><code>&lt;script&gt;

"use strict";

var Sort = function( inputArray ) {
    this.arrayTest = undefined;
    this.init( inputArray );
};
Sort.prototype.init = function( inputArray ) {
    if ( inputArray instanceof Array &amp;&amp; inputArray.length &gt; 0 ) {
        this.arrayTest = inputArray;
        this.DumpArray( this.arrayTest );
        this.QuickSort( this.arrayTest, 0, this.arrayTest.length - 1 );
        this.DumpArray( this.arrayTest );
    }
    else console.log( 'Input argument isn\'t an array object or its length equals zero.' );
};
Sort.prototype.QuickSort = function( data, first, last ) {
    var i = first;
    var j = last;
    var x = data[ Math.ceil( ( first + last ) / 2 ) ];

    do {
        while ( data[ i ] &lt; x ) i++;
        while ( data[ j ] &gt; x ) j--;

        if ( i &lt;= j ) {
            if ( i &lt; j ) {
                if ( data[ i ] !== data[ j ] ) {
                    var tempItem = data[ i ];
                    data[ i ] = data[ j ];
                    data[ j ] = tempItem;
                }
            }
            i++;
            j--;
        }

    } while ( i &lt;= j );

    if ( i &lt; last ) this.QuickSort( data, i, last );
    if ( first &lt; j ) this.QuickSort( data, first, j );
}
Sort.prototype.DumpArray = function() {
    var output = 'The content of array: [ ';

    if ( this.arrayTest instanceof Array &amp;&amp; this.arrayTest.length &gt; 0 ) {
        this.arrayTest.forEach( function( item ) {
            output = output.concat( [ item, ', ' ].join( '' ) );
        });
    }
    else console.log( 'Can\'t dump the input array. It\'s not an array object or its length equals zero.' );

    output = output.substring( 0, output.length - 2 );
    output = output.concat( ' ].' );
    console.log( output );
};

window.onload = new Sort([
    10,
    1,
    27,
    4090,
    100000,
    2222,
    4934
]);

&lt;/script&gt;

             <div class="post-text" itemprop="text"> 
              <p>Looks okay to me.</p> 
              <ul> 
               <li>I would probably go for <code>if ( i != j ) {</code> which could be faster than <code>if ( i &lt; j ) {</code></li> 
               <li>You have a ton of short variable names, I guess that's ok for a sorting routine, though I would still rename <code>x</code> to <code>m</code>(iddle).</li> 
               <li>I would have declared <code>tempItem</code> right under <code>i</code> and <code>j</code></li> 
              </ul> 
              <p>For the part you are not so interested in;</p> 
              <ul> 
               <li>Capital S in <code>Sort</code> is reserved for constructors</li> 
               <li>I think <code>reduce</code> is more appropriate than <code>forEach</code> in <code>DumpArray</code></li> 
               <li>You should not put code on the same line after an <code>else</code> statement</li> 
               <li><p>Now that I think about it the dump routine could be reduced to</p> <pre><code>Sort.prototype.DumpArray = function() {

var prefix = 'The content of array: [ ';
//Duck typing, just test for join
if( this.arrayTest.join )
    console.log( prefix + this.arrayTest.join(', ') + ' ]'  )
else
    console.log( "Input: " , this.arrayTest );
};
</code></pre></li>
