                }

            if ((pc / caa.length) &gt; 0.5) {
                return true;
            }
        }
        return false;
    }

    // create object listing the SOUNDEX values for each letter
    // -1 indicates that the letter is not coded, but is used for coding
    //  0 indicates that the letter is omitted for modern census archives
    //                              but acts like -1 for older census archives
    //  1 is for BFPV
    //  2 is for CGJKQSXZ
    //  3 is for DT
    //  4 is for L
    //  5 is for MN my home state
    //  6 is for R
    function makesoundex() {
        this.a = -1
        this.b = 1
        this.c = 2
        this.d = 3
        this.e = -1
        this.f = 1
        this.g = 2
        this.h = 0
        this.i = -1
        this.j = 2
        this.k = 2
        this.l = 4
        this.m = 5
        this.n = 5
        this.o = -1
        this.p = 1
        this.q = 2
        this.r = 6
        this.s = 2
        this.t = 3
        this.u = -1
        this.v = 1
        this.w = 0
        this.x = 2
        this.y = -1
        this.z = 2
    }

    var sndx = new makesoundex()

    // check to see that the input is valid
    function isSurname(name) {
        if (name == "" || name == null) {
            return false
        } else {
            for (var i = 0; i &lt; name.length; i++) {
                var letter = name.charAt(i)
                if (!(letter &gt;= 'a' &amp;&amp; letter &lt;= 'z' || letter &gt;= 'A' &amp;&amp; letter &lt;= 'Z')) {
                    return false
                }
            }
        }
        return true
    }

    // Collapse out directly adjacent sounds
    // 1. Assume that surname.length&gt;=1
    // 2. Assume that surname contains only lowercase letters
    function collapse(surname) {
        if (surname.length == 1) {
            return surname
        }
        var right = collapse(surname.substring(1, surname.length))
        if (sndx[surname.charAt(0)] == sndx[right.charAt(0)]) {
            return surname.charAt(0) + right.substring(1, right.length)
        }
        return surname.charAt(0) + right
    }

    // Collapse out directly adjacent sounds using the new National Archives method
    // 1. Assume that surname.length&gt;=1
    // 2. Assume that surname contains only lowercase letters
    // 3. H and W are completely ignored
    function omit(surname) {
        if (surname.length == 1) {
            return surname
        }
        var right = omit(surname.substring(1, surname.length))
        if (!sndx[right.charAt(0)]) {
            return surname.charAt(0) + right.substring(1, right.length)
        }
        return surname.charAt(0) + right
    }

    // Output the coded sequence
    function output_sequence(seq) {
        var output = seq.charAt(0).toUpperCase() // Retain first letter
        output += "-" // Separate letter with a dash
        var stage2 = seq.substring(1, seq.length)
        var count = 0
        for (var i = 0; i &lt; stage2.length &amp;&amp; count &lt; 3; i++) {
            if (sndx[stage2.charAt(i)] &gt; 0) {
                output += sndx[stage2.charAt(i)]
                count++
            }
        }
        for (; count &lt; 3; count++) {
            output += "0"
        }
        return output
    }

    // Compute the SOUNDEX code for the surname
    function soundex(value) {
        if (!isSurname(value)) {
            return null
        }
        var stage1 = collapse(value.toLowerCase())
        //form.result.value=output_sequence(stage1);

        var stage1 = omit(value.toLowerCase())
        var stage2 = collapse(stage1)
        return output_sequence(stage2);

    }

    function clean(u) {
        var u = u.replace(/\,/g, "");
        u = u.toLowerCase().split(" ");
        var cw = ["ARRAY OF WORDS TO BE EXCLUDED FROM COMPARISON"];
        var n = [];
        for (var y = 0; y &lt; u.length; y++) {
            var test = false;
            for (var z = 0; z &lt; cw.length; z++) {
                if (u[y] != "" &amp;&amp; u[y] != cw[z]) {
                    test = true;
                    break;
                }
            }
            if (test) {
    //Don't use &amp; or $ in comparison
                var val = u[y].replace("$", "").replace("&amp;", "");
                n.push(val);
            }
        }
        return n;
    }
</code></pre>

              <p>Here's an answer based on Levenshtein distance <a href="https://en.wikipedia.org/wiki/Levenshtein_distance" rel="noreferrer">https://en.wikipedia.org/wiki/Levenshtein_distance</a></p> 
              <pre><code>function similarity(s1, s2) {
  var longer = s1;
  var shorter = s2;
  if (s1.length &lt; s2.length) {
    longer = s2;
    shorter = s1;
  }
  var longerLength = longer.length;
  if (longerLength == 0) {
    return 1.0;
  }
  return (longerLength - editDistance(longer, shorter)) / parseFloat(longerLength);
}
</code></pre> 
              <p>For calculating edit distance</p> 
              <pre><code>function editDistance(s1, s2) {
  s1 = s1.toLowerCase();
  s2 = s2.toLowerCase();

  var costs = new Array();
  for (var i = 0; i &lt;= s1.length; i++) {
    var lastValue = i;
    for (var j = 0; j &lt;= s2.length; j++) {
      if (i == 0)
        costs[j] = j;
      else {
        if (j &gt; 0) {
          var newValue = costs[j - 1];
          if (s1.charAt(i - 1) != s2.charAt(j - 1))
            newValue = Math.min(Math.min(newValue, lastValue),
              costs[j]) + 1;
          costs[j - 1] = lastValue;
          lastValue = newValue;
        }
      }
    }
    if (i &gt; 0)
      costs[s2.length] = lastValue;
  }
  return costs[s2.length];
}
</code></pre>

             <div class="post-text" itemprop="text"> 
              <p>Here is a very simple function that does a comparison and returns a percentage based on equivalency. While it has not been tested for all possible scenarios, it may help you get started. </p> 
              <pre><code>function similar(a,b) {
    var lengthA = a.length;
    var lengthB = b.length;
    var equivalency = 0;
    var minLength = (a.length &gt; b.length) ? b.length : a.length;    
    var maxLength = (a.length &lt; b.length) ? b.length : a.length;    
    for(var i = 0; i &lt; minLength; i++) {
        if(a[i] == b[i]) {
            equivalency++;
        }
    }


    var weight = equivalency / maxLength;
    return (weight * 100) + "%";
}
alert(similar("test","tes"));   // 75%
alert(similar("test","test"));  // 100%
alert(similar("test","testt")); // 80%
alert(similar("test","tess"));  // 75%
</code></pre>

              <p>How about function <code>similar_text</code> from <a href="http://phpjs.org/functions/similar_text:902" rel="nofollow">PHP.js library</a>?</p> 
              <p>It is based on a PHP function with <a href="http://www.php.net/manual/en/function.similar-text.php" rel="nofollow">the same name</a>.</p> 
              <pre><code>function similar_text (first, second) {
    // Calculates the similarity between two strings  
    // discuss at: http://phpjs.org/functions/similar_text

    if (first === null || second === null || typeof first === 'undefined' || typeof second === 'undefined') {
        return 0;
    }

    first += '';
    second += '';

    var pos1 = 0,
        pos2 = 0,
        max = 0,
        firstLength = first.length,
        secondLength = second.length,
        p, q, l, sum;

    max = 0;

    for (p = 0; p &lt; firstLength; p++) {
        for (q = 0; q &lt; secondLength; q++) {
            for (l = 0;
            (p + l &lt; firstLength) &amp;&amp; (q + l &lt; secondLength) &amp;&amp; (first.charAt(p + l) === second.charAt(q + l)); l++);
            if (l &gt; max) {
                max = l;
                pos1 = p;
                pos2 = q;
            }
        }
    }

    sum = max;

    if (sum) {
        if (pos1 &amp;&amp; pos2) {
            sum += this.similar_text(first.substr(0, pos2), second.substr(0, pos2));
        }

        if ((pos1 + max &lt; firstLength) &amp;&amp; (pos2 + max &lt; secondLength)) {
            sum += this.similar_text(first.substr(pos1 + max, firstLength - pos1 - max), second.substr(pos2 + max, secondLength - pos2 - max));
        }
    }

    return sum;
}
</code></pre>

              <p>Just one I quickly wrote that might be good enough for your purposes:</p> 
              <pre><code>function Compare(strA,strB){
    for(var result = 0, i = strA.length; i--;){
        if(typeof strB[i] == 'undefined' || strA[i] == strB[i]);
        else if(strA[i].toLowerCase() == strB[i].toLowerCase())
            result++;
        else
            result += 4;
    }
    return 1 - (result + 4*Math.abs(strA.length - strB.length))/(2*(strA.length+strB.length));
}
</code></pre> 
              <p>This weighs characters that are the same but different case 1 quarter as heavily as characters that are completely different or missing. It returns a number between 0 and 1, 1 meaning the strings are identical. 0 meaning they have no similarities. Examples:</p> 
              <pre><code>Compare("Apple", "Apple")    // 1
Compare("Apples", "Apple")   // 0.8181818181818181
Compare("Apples", "apple")   // 0.7727272727272727
Compare("a", "A")            // 0.75
Compare("Apples", "appppp")  // 0.45833333333333337
Compare("a", "b")            // 0
</code></pre>
