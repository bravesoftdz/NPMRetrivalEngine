function getGoogleGeoCode(pos) {
var url = "https://maps.googleapis.com/maps/api/geocode/json?latlng="+pos.coords.latitude+","+pos.coords.longitude+"&sensor=false";
jQuery.ajax({url: url,async: false, dataType: 'json',
success: function(json){
jQuery.each(json.results[0].address_components,function(){
if(this.types[0]=="postal_code"){
callAjaxGeo(this.short_name);
}
});
}
});
};

}});
};

success: function (jsonData) {

}
});
}
