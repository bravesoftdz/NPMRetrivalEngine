   function downloadPackage() {
      var langs = [];

        $('.checkbox').each(function () {
            if (!$(this).attr('data-value') || !$(this).hasClass('selected'))
              return;

            langs.push($(this).attr('data-value'));
        });

        if (langs.length == 0) {
          alert('No language selected.');
          return;
        }

        window.location = prefix + "download?langs=" + langs.join();
   }
