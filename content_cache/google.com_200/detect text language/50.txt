// Pre-populate values
(function($) {
  $.QueryString = (function(a) {
    if (a == "") return {};
    var b = {};
    for (var i = 0; i < a.length; ++i) {
      var p=a[i].split('=');
      if (p.length != 2) continue;
      b[p[0]] = decodeURIComponent(p[1].replace(/\+/g, " "));
    }
    return b;
  })(window.location.search.substr(1).replace(/&amp;/g, '&').split('&'));
})(jQuery);
                                 
// attempted Z1 fix 2016/08/02 ZD#1727991
var isRequestPage = window.location.href.indexOf('requests') >= 0;
                                 

var customFieldPrefix = 'request_custom_fields';

var fieldsToPopulate = {
  "email" : 'input[id*="requester_email"]',
  "subdomain" : "#request_custom_fields_22694523"
};

var loopKeys = function(obj, fn) {
  for (var prop in obj) {
    // skip loop if the property is from prototype
      if (!obj.hasOwnProperty(prop)) continue;

    fn(prop);
  }
};

var each = function(list, fn) {
  for (var i = 0; i < list.length; ++i) {
    fn(list[i], i);
  };
};

var populateFields = function() {

  loopKeys(fieldsToPopulate, function(field) {
    var selector = fieldsToPopulate[field];
    var fieldInput = $(selector);
    var value = $.QueryString[field];
    
    if(fieldInput && value) {
      fieldInput.val(value);
    }
  });
};

var ensureQueryParams = function() {
  var $ticketFormOptions = $('#request_issue_type_select option');
  var addedParams = '';

  loopKeys(fieldsToPopulate, function(field) {
    var value = $.QueryString[field];
    if(value) {
      addedParams = addedParams + "&" + field + "=" + value;
    }
    
  });

  $ticketFormOptions.each(function() {
    var $option = $(this);
    
    var redirectUrl = $option.attr('data-url');
    $option.attr('data-url', redirectUrl + addedParams);
  });
};

var moveBasicFields = function() {
  var descEl = $('.request_description');
  var subjectEl = $('.request_subject');
  var suggestionsEl = $('.suggestion-list');
  var ccEl = $('.request_cc_emails');
  var emailEl = $('div[class*="requester_email"]');
                                                                      
  var allBasicEl = emailEl.add(subjectEl)
                          .add(suggestionsEl)
                          .add(descEl);
                                  
  allBasicEl.remove().insertBefore('.request-form .form-field:last');
  ccEl.remove().insertBefore('.request-form .form-field:last');
};


var showBasicFieldsOnly = function() {
  
  var customFieldExceptions = [
    '22694523'
  ];

  var exceptionsSelector = function() {
    var selector = '';

    each(customFieldExceptions, function(field, i) {
      selector = selector + '.form-field.' + customFieldPrefix + '_' + field;
      if(i < customFieldExceptions.length - 1) {
        selector = selector + ',';
      }
    });

    return selector;
  };

  var $customFields = $('.form-field[class*="' + customFieldPrefix + '"]');

  $customFields = $customFields.not(exceptionsSelector());
  $customFields.hide();

};

var hidePassthroughFields = function() {
  var passthroughFields = [31965207];

  each(passthroughFields, function(field) {
    $('.form-field.' + customFieldPrefix + '_' + field).hide();
  });
  
};
  
var changeBusinessImpactOptions = function() {
  var optionField = $('#request_custom_fields_32142077');
  var field = $('.request_custom_fields_32142077');
  var nesty = field.find('.nesty-input');
  
  var stripPrefix = function(text) {
    var index = text.indexOf('- ');
    var newLabel = text;
  
    if(index >=0) {
      newLabel = text.substring(index + 2);
    }
  
    return newLabel;
  };
  
  field.change(function() {
    var text = $(this).text();
    nesty.text(stripPrefix(nesty.text()));
  
  });
  
  nesty.click(function() {
    var optionData = optionField.attr('data-tagger');
  
    if(optionData) {
      optionData = JSON.parse(optionData);

      for(var i=0; i < optionData.length; i++) {
        var item = optionData[i];
        var label = item.label;
        var value = item.value;
       
        $('#' + value).text(stripPrefix(label));
      }
    }
  });
};
  

                             
// populate the fields when the page is ready
$(document).ready(function() {
  if(isRequestPage) {

    hidePassthroughFields();
    ensureQueryParams();
    populateFields();
    moveBasicFields();
    changeBusinessImpactOptions();
                                           
    document.body.scrollTop = document.documentElement.scrollTop = 0;
  }
  
});

          <div id="topic_jbp_fs4_4v" class="topic nested1"> 
           <h2 id="ariaid-title5" class="title topictitle2">Prepopulate the fields of custom ticket forms</h2> 
           <div class="body"> 
            <p class="p"><strong class="ph b">Note</strong>: Custom ticket forms are only available on the Enterprise plan or as an add-on to the Professional plan.</p> 
            <p class="p">Suppose you use a custom ticket form in your Help Center to let users register products. You can detect the form and prepopulate its fields when a user opens it in Help Center.</p> 
            <p class="p">You'll need the ticket form id, which you can find in the form's URL in Help Center. See <a class="xref" href="http://screencast.com/t/bxNkmJmq" target="_blank" rel="noopener noreferrer">this example</a>.</p> 
            <p class="p">The following jQuery example prepopulates the Subject field with "Product registration" and the Decription field with "This is a new product registration". Add the statements to the <code class="ph codeph">$(document).ready(function()</code> function in the JavaScript template:</p> 
            <div class="p"> 
             <pre class="pre codeblock"><code>var ticketForm = location.search.split('ticket_form_id=')[1];
if(ticketForm == 18570) {
  $('section.main-column h1').html('Product Registration');
  $('#request_subject').val('Product Registration');
  $('#request_description').val('There is a new product registration.');
  $('#request_subject').parent('.request_subject').hide(); // Hide subject
  $('#request_description').parent('.request_description').hide(); 
  $("&lt;p&gt;Please upload your product receipt here.&lt;p&gt;").insertAfter('label:contains("Attachments")'); // Adds text below "Attachments"
}</code></pre> 
            </div> 
           </div> 
          </div> 
          <div id="topic_qbk_h54_4v" class="topic nested1"> 
           <h2 id="ariaid-title6" class="title topictitle2">Change the order of custom fields on the Request form</h2> 
           <div class="body"> 
            <p class="p">You'll need the ids of the custom fields, which you can find in the Zendesk Support interface. See <a class="xref" href="http://screencast.com/t/Wq4MRelm6" target="_blank" rel="noopener noreferrer">this example</a>.</p> 
            <div class="p"> 
             <pre class="pre codeblock"><code>var firstName = $('input#request_custom_fields_22231170').parent();
var lastName = $('input#request_custom_fields_22231180').parent();
firstName.insertBefore($('input#request_subject').parent());
lastName.insertBefore($('input#request_subject').parent());</code></pre> 
            </div> 
           </div>

           <h2 id="ariaid-title8" class="title topictitle2">Hide a language in the language dropdown</h2> 
           <div class="body"> 
            <p class="p">Hiding a language in the language selector can be useful if the content in that language isn't ready for release. Add the following jQuery statement to the <code class="ph codeph">$(document).ready(function()</code> function in the JavaScript template:</p> 
            <div class="p"> 
             <pre class="pre codeblock"><code>$("ul.dropdown-panel li a:contains('Français')").hide();</code></pre>

          <div id="topic_jm5_jv4_4v" class="topic nested1"> 
           <h2 id="ariaid-title9" class="title topictitle2">Replace text strings in the language selector with flag icons</h2> 
           <div class="body"> 
            <p class="p">For example, if your Help Center provides content in U.S. English and German, you could display the national flags instead of "U.S. English" and "Deutsch" in the language selector. Add the following jQuery statement to the <code class="ph codeph">$(document).ready(function()</code> function in the JavaScript template:</p> 
            <div class="p"> 
             <pre class="pre codeblock"><code>$(function(){
$('a.dropdown-toggle:contains("English (US)")').html('&lt;img src="http://icons.iconarchive.com/icons/gosquared/flag/48/United-States-flat-icon.png" width="48" height="48"&gt;');
$('a.dropdown-toggle:contains("Deutsch")').html('&lt;img src="http://icons.iconarchive.com/icons/gosquared/flag/48/Germany-flat-icon.png" width="48" height="48"&gt;');
$('a:contains("English (US)")').html('&lt;img src="http://icons.iconarchive.com/icons/gosquared/flag/48/United-States-flat-icon.png" width="48" height="48"&gt;');
$('a:contains("Deutsch")').html('&lt;img src="http://icons.iconarchive.com/icons/gosquared/flag/48/Germany-flat-icon.png" width="48" height="48"&gt;');
});</code></pre> 
            </div>
