               </blockquote> There is no good way to carry java objects between steps in a service or a BPD. The problem is that the main engine that is running the services and BPD is using Rhino, which is server side java script. While you can map java data containers to JS variables, JS does not support objects. As such you can't put a java object into your tw.* variable. You can put the data from your java object into your tw.* variable, which is what I think you saw happening with the Hash Map, but I don't think you can call the methods on it after it has been passed between the BPD and service layer (which is a pass by value operation).
               <br> 
               <br> Before the 7.0 release you could, if you wanted make a global JS variable by simply omitting the "var" on your declaration. You could then use that variable anywhere you wanted. The problem was that the rules for "Global" were a bit problematic. There was 1 per JVM, but if you were running a cluster, you had multiple JVMs. Also, they were not thread safe, so if 2 services were accessing the global in the same JVM at the same time you could encounter unpredictable results. var dwUnescapeQuoteBlocks = function() {
	dojo.query('div[id^="quote_"]').forEach(function(node, index, arr){
		dwUnescapeQuoteBlock(node)
	});
}

var dwUnescapeQuoteBlock = function(node) {
    var nodeHtml = node.innerHTML;
	nodeHtml = nodeHtml.replace(/&lt;pre/ig,"<pre").replace(/dir="ltr"&gt;/ig,'dir="ltr">').replace(/&lt;\/pre&gt;/ig,"</pre>");
	node.innerHTML = nodeHtml;
}

var highlightDwCodeBlocks = function() {
  if (!window.ace) {
	setTimeout(function() {highlightDwCodeBlocks(); },500);
	return;
  }
  dwUnescapeQuoteBlocks();
  var blocks = dojo.query("pre.dw");
  for (var i =0;i<blocks.length;i++) { 
	var codeBlock = blocks[i];
	codeBlock.className = codeBlock.className.replace(/dw/ig,"dwedited");
	codeBlock.style.width = codeBlock.offsetWidth+"px";
	codeBlock.style.height = (codeBlock.offsetHeight+20)+"px";
	//codeBlock.style.top = codeBlock.offsetTop+"px";
	//codeBlock.style.display="block";
	//codeBlock.innerHTML = codeBlock.innerHTML.replace(/<br>/ig,"<br/>\n !!! \n<br/>\n");
  }
  for (var i =0;i<blocks.length;i++) {
	var codeBlock = blocks[i];
	codeBlock.style.position = "relative";  
	
	var editor = window.ace.edit(codeBlock);
	editor.setReadOnly(true);
	if (codeBlock.className.indexOf("js")>=0)
		editor.getSession().setMode("ace/mode/javascript");
	else if (codeBlock.className.indexOf("html")>=0)
		editor.getSession().setMode("ace/mode/html");	
	else if (codeBlock.className.indexOf("java")>=0)
		editor.getSession().setMode("ace/mode/java");
	else if (codeBlock.className.indexOf("css")>=0)
		editor.getSession().setMode("ace/mode/css");		
  }  
}
