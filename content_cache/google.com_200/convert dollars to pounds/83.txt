                I was intrigued by this one.&nbsp;&nbsp;My function works well, but only for 0 - 9999.&nbsp;&nbsp;I didn't add support for anything over 4 digits, mostly to save time.
                <br>
                <blockquote>
                 <h4>Quote:</h4>
                 <div class="body">
                  <br>5000 (ie. $5,000.00)
                  <br>TO
                  <br>Five Thousand and 00/100
                  <br>
                 </div>
                </blockquote>Since you capitalized 
                <span style="font-weight:bold;">Five</span> and 
                <span style="font-weight:bold;">Thousand</span>, I added an option to capitalize each word in the return value.&nbsp;&nbsp;I did not add support for decimals, but if you really need it, I'm sure I can add it in tonight.&nbsp;&nbsp;Thanks for the challenge Nigel - this one was fun :)

                  <br>onload = init;
                  <br>
                  <br>function init() {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;var ProperNames = true;&nbsp;&nbsp;&nbsp;&nbsp;// If this is true, we'll tidy it up.
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;var numToConvert = 1234;&nbsp;&nbsp;&nbsp;&nbsp;// The number we want to convert.

                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (ret == ":(") {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;alert("Please enter a number from 0 to 9999.");

                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;tRet = ret.substring(0,1).toUpperCase();
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;ret = tRet + ret.substring(1);
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Now go through all others
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;for (c=0; c&lt;ret.length; c++) {

                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;lRet = ret.substring(0,c+1);
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;tRet = ret.substring(c+1,c+2).toUpperCase();
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;rRet = ret.substring(c+2);
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;ret = lRet + tRet + rRet;
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;alert(ret);

                  <br>
                  <br>function numToName(numAmt) {
                  <br>strAmt = numAmt.toString();
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;if (strAmt.length &gt; 4) {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;retAmt = ":(";
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;return retAmt;
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;}
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;if ((numAmt &gt; 20) &amp;&amp; (numAmt &lt; 100) &amp;&amp; (strRight(strAmt,1) != "0")) {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;retAmt = str[strLeft(strAmt,1)+"0"];
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;retAmt += " " + str[strRight(strAmt,1)];
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;} else if (numAmt &gt; 99) {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (strAmt.length == 4) {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;retAmt = str[strLeft(strAmt,1)] + " thousand";
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (strMid(strAmt,1,1) != "0") {
                  <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;retAmt += " " + str[strMid(strAmt,1,1)];
