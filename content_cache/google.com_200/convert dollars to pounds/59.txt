function daysInFebruary (year){<br>
&nbsp;&nbsp;&nbsp;// February has 29 days in any year evenly divisible by four,<br>

&nbsp;&nbsp;&nbsp;return (((year % 4 == 0) &amp;&amp; ( (!(year % 100 == 0)) || (year % 400 == 0))) ? 29 : 28 );<br>

function DaysArray(month,year) {<br>

&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;this[i] = 31<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (i==4 || i==6 || i==9 || i==11) {this[i] = 30}<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (i==2) {this[i] = daysInFebruary(year)}<br>
&nbsp;&nbsp;&nbsp;} <br>
&nbsp;&nbsp;&nbsp;return this<br>
