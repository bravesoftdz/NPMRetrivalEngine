    (function(){

        function templateReplace(str, find, replace) {
            return str.split(find).join(replace);
        }

        var STATUS_TRANSLATIONS = {"Cancelled": "Cancelled", "Delayed": "Delayed", "Diverted": "Diverted - Contact airline", "Expected": "Expected", "InAir": "In air", "InGate": "At gate", "Landed": "Landed", "NoTakeoffInfo": "No information - Contact airline", "OutGate": "Taxiing", "PastFlight": "No information available", "Scheduled": "Scheduled", "Unknown": "Unknown"};

        var STATUS_TYPES = {
            DEPARTURES: 'departures',
            ARRIVALS: 'arrivals'
        };

        var TRANSLATED_BOARDS = {};
        TRANSLATED_BOARDS["arrivals"] = "Arrivals";
        TRANSLATED_BOARDS["departures"] = "Departures";

        var FlightStatusTableRow = function(data, type, airportNames, localisation) {

            var details = (type == STATUS_TYPES.DEPARTURES) ? data.departure.details : data.arrival.details;
            var airportName = ((type == STATUS_TYPES.DEPARTURES) ? data.departure.airportName : data.arrival.airportName) || '';
            var date = localisation.parseDate(data.date + ' ' + data.time, 'yyyy-MM-dd H:m');

            this.time = {
                display: localisation.formatTime(date),
                sort: date.getTime()
            };

            this.flightNumber = data.flightNumber;
            this.airline = data.airline.name;
            this.airlineFs = data.airline.fs;
            this.terminal =  details.terminal || '';
            this.terminalMobile = details.terminal ? 'Terminal {terminal}'.replace('{terminal}', details.terminal) : '';
            this.country = (type == STATUS_TYPES.DEPARTURES) ? data.arrival.country : data.departure.country;

            this.place = function() {
                var iataCode = (type == STATUS_TYPES.DEPARTURES) ? data.arrival.airport : data.departure.airport;
                var localisedAirportNames = airportNames[iataCode];
                var translatedPlaceName = localisedAirportNames && localisedAirportNames[Skyscanner.culture.name];
                return translatedPlaceName || airportName;
            };

            this.status = function() {
                var defaultStatusKey = 'Unknown'
                return STATUS_TRANSLATIONS[data.status] || STATUS_TRANSLATIONS[defaultStatusKey] || '';
            };

            this.logo = function() {
                var logo = '<img class="airline-icon" src="//logos.skyscnr.com/images/airlines/favicon/{airline-fs}.png" alt="{airline}" onerror="this.className+=\' load-error\'"/>'
                logo = templateReplace(logo, '{airline}', this.airline);
                logo = templateReplace(logo, '{airline-fs}', this.airlineFs);
                return logo;
            };

            this.flightDetails = {
                display: [
                    this.time.display,
                    this.place(),
                    this.terminalMobile
                ].filter(function(detail) {
                    return !!detail;
                }).join('<br/>'),
                sort: this.time.sort
            };

            this.statusMobile = function() {
                var contents = '<span class="airline">{logo} {flightNumber}</span>{status}';
                contents = templateReplace(contents, '{logo}', this.logo());
                contents = templateReplace(contents, '{flightNumber}', this.flightNumber);
                contents = templateReplace(contents, '{status}', this.status());
                return contents;
            };

            this.airlineFlight = function() {
                // Sometimes we don't have airline information
                var flightDetails = [];
                if (this.airline) { flightDetails.push(this.airline);}
                flightDetails.push(this.flightNumber)
                flightDetails = flightDetails.join(' - ')

                var contents = '{logo} {flightDetails}';
                contents = templateReplace(contents, '{logo}', this.logo());
                contents = templateReplace(contents, '{flightDetails}', flightDetails);
                return contents;
            };
        };

        var FlightStatusWidget = function(){
            this.STATS_ENDPOINT = "https://s3-eu-west-1.amazonaws.com/flight-stats.eu-west-1.prod.aws.skyscnr.com/stats/";
            this.iata = "DXB";
            this.activeBoard = "arrivals";
            this.localisation = new Skyscanner.localisation.Localisation();
            this.features = {
                anchorLinks: false
            };
            this.setupFilters();
            this.setupBoards();
        };

        FlightStatusWidget.prototype.setupBoards = function()
        {
            this.boards = {};
            this.boardHeaders = {};
            this.tableInstances = {};

            $('#flight-status-widget-loading').hide();
            $('#filters-box').show();
            

            for(var type in STATUS_TYPES)
            {
                var boardType = STATUS_TYPES[type];
                this.boards[boardType] = $("#flight-status-board-" + boardType);
                this.boardHeaders[boardType] = $("#flight-status-board-header-" + boardType);
                (function(self, type){
                    self.boardHeaders[boardType].on('click', self.tabClicked.bind(self, boardType));
                })(this, type);
            }

            var activeBoard = "arrivals";
            this.switchToBoard(activeBoard);
        };

        FlightStatusWidget.prototype.tabClicked = function(type, event)
        {
            event.preventDefault();
            this.switchToBoard(type);
        };

        FlightStatusWidget.prototype.switchToBoard = function(type)
        {
            if (this.features.anchorLinks) { this.updateLocationHash(type); }
            this.terminalFilterSelect.val("");
            this.flightsSearchBox.val("");
            this.activeBoardId = type;
            this.activeBoard = this.boards[this.activeBoardId];

            // Only update if board is empty
            if(this.activeBoard.children('tbody').children().length == 0) {
                this.update();
            } else {
                this.updateFilterOptions();
            }

            if(type == STATUS_TYPES.ARRIVALS) {
                $('#flights-search-box').attr("placeholder","Airline, flight, origin");
            } else {
                $('#flights-search-box').attr("placeholder","Airline, flight, destination" );
            }

            for(var boardId in this.boards)
            {
                if(boardId == type)
                {
                    this.boardHeaders[boardId].addClass('active');
                    this.boards[boardId].show();
                }
                else
                {
                    this.boardHeaders[boardId].removeClass('active');
                    this.boards[boardId].hide();
                }
            }

            // Only update if board is empty
            if(this.activeBoard.children('tbody').children().length == 0) {
                this.update();
            } else {
                this.refreshFilters();
            }
        };

        FlightStatusWidget.prototype.getSelectedBoard = function()
        {
            var translatedBoardFromHash = window.location.hash.slice(1);
            return this.features.anchorLinks && _.invert(TRANSLATED_BOARDS)[translatedBoardFromHash] || this.activeBoard;
        };

        FlightStatusWidget.prototype.updateLocationHash = function(boardType)
        {
            var anchor = this.boardHeaders[boardType][0];
            var translatedHashFromBoard = TRANSLATED_BOARDS[boardType];
            // We add and remove the 'name' attribute to prevent page scroll
            // when changing the location hash.
            anchor.removeAttribute('name');
            window.location.hash = '#' + translatedHashFromBoard;
            anchor.setAttribute('name', translatedHashFromBoard);
        };

        FlightStatusWidget.prototype.refreshFilters = function()
        {
            var table = this.tableInstances[this.activeBoardId];
            table.draw();
        };

        FlightStatusWidget.prototype.getTableData = function(data, type, airportNames)
        {
            return data.flights.map(function(rowData) {
                return new FlightStatusTableRow(rowData, type, airportNames, this.localisation);
            }, this);
        };

        FlightStatusWidget.prototype.renderTable = function(data, type)
        {
            if (this.tableInstances[type]) {
                this.updateTable(this.tableInstances[type], data);
            } else {
                this.initTable(type, data);
            }

            this.updateFilterOptions();
        };

        FlightStatusWidget.prototype.updateTable = function(table, data)
        {
            table.clear();
            table.rows.add(data);
            table.draw();
        };

         FlightStatusWidget.prototype.getActiveTable = function()
         {
             return this.tableInstances[this.activeBoardId];
         };

        FlightStatusWidget.prototype.updateFilterOptions = function()
        {
            var select = this.terminalFilterSelect;
            var selectedFilterOption = select.val();
            var activeTable = this.getActiveTable();

            var defaultOptions = [
                '<option value="">All terminals</option>'

          <h2>Dubai Departures Information</h2> 
          <p>Generally speaking, you should plan to arrive at Dubai International Airport at least 90 minutes before your departure. However, it’s widely recommended to arrive three hours ahead of time. As of 2016, Dubai is considered the world's busiest airport in terms of international travel. If you don’t check in online, most airlines will allow you to check in three hours prior to your Dubai departure. If you arrive early, you will find several ways to spend your extra time. If you want to grab a bite to eat, you will find almost any kind of cuisine you desire including Chinese, Japanese, Indian, American, French, and local fare. You can spend your time at enjoyable sit-down restaurants or familiar fast food favorites. If you want to relax before your flight, you can visit one of the airport's spas or consider purchasing a day pass to rest in one of the several lounges at the Dubai airport. You may also visit the Zen Garden and health clubs or catch a few hours of sleep in a "SnoozeCube." If you love to shop when you travel, Dubai International Airport has no shortage of retail stores. Find the perfect souvenir at the duty-free shops that contain fine fragrances, alcohol, tobacco, and various confections.</p> 
          <h2>Dubai Arrivals Information</h2> 
          <p>Most of Dubai airport’s arrivals head to Terminal 1 and 3. Regardless of where in the airport your flight arrives, you’ll need to follow the signs to the baggage collection area and then through customs. If you prefer assistance navigating Dubai's busy airport, consider using Marhaba, the airport's meet-and-greet service. Marhaba provides meet-and-greet packages for solo travelers and families on Dubai flight arrivals. Packages are available for all budgets. An agent will meet you after you enter the terminal or right before the immigration area. They will quickly escort you through immigration, customs, and the rest of the airport and arrange whatever form of transportation you are taking to the city. You should expect to spend a minimum of 30 minutes from the time you arrive until you are ready to leave the airport. Marhaba meet-and-greet packages are also available for Dubai departures. If you have had a really long flight and are too tired to deal with traveling through the airport right away, you may also consider using a SnoozeCube when you arrive.</p> 
          <h2>Pick up and drop off at Dubai airport</h2> 
          <p>Those who are picking up or dropping off passengers at Dubai International Airport can do so in front of the terminal from which they’ll be arriving or departing. If your host prefers to meet you inside of the airport after you pass through customs, they can park in one of the short-term parking areas by each terminal. Valet parking is available at Terminals 1 and 3.</p> 
          <h2>How to get to and from Dubai airport</h2> 
          <p>You have several options for traveling to and from Dubai International Airport. Public transportation and taxis serve Dubai airport arrivals and departures. The Dubai Metro offers two lines that leave every 10 minutes to take you to the city during business hours on Saturday through Thursday. Dubai Metro service begins Friday after 2:00 p.m. You can also connect with buses or taxis if necessary. Buy a "Nol" card before you leave the airport to pay for public transportation.</p> 
          <p>You may prefer taking a bus or taxi directly from the airport. Buses depart from and arrive across from each terminal. If you have obtained an international drivers' license prior to your departure, consider renting a car when you arrive in Dubai. Hertz, Dollar, Europcar, Fast, Payless, and Sixt car rental agencies serve Dubai International Airport.</p>
