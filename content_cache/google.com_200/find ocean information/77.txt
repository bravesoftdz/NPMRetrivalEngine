 Footer.$cartToggle = $('.cart-top');
 Footer.isDocked = true;
 Footer.disabled = false;
 Footer.scrollBeforeUndock = 0;
 Footer.init = function () {
         Footer.reset();

         Footer.$cartToggle.click(function () {
                 console.log('here');
                 Footer.$footerFloat.toggleClass('full-cart');
         });
 };

     Footer.undock = function () {
         Footer.$footerFloat.removeClass('footer-float-fixed');
         Footer.$footerFloat.addClass('full-cart');
         Footer.isDocked = false;
 };


 Footer.getScrollBeforeUndock = function () {
         var scrollAmount = Footer.$footer.offset().top - $(window).height();

         // if docked, subtrack FooterFloat height
         if (Footer.isDocked) scrollAmount += $('.footer-float').height();

         return scrollAmount;
 };

 Footer.dock = function () {
         Footer.$footerFloat.addClass('footer-float-fixed');
         Footer.$footerFloat.removeClass('full-cart');
         Footer.isDocked = true;
 };

 Footer.reset = function () {
         if (Footer.$footer.length) {
                 var newScrollBeforeUndock = Footer.getScrollBeforeUndock();

                 // update and trigger docking check if it differs
                 if (Footer.scrollBeforeUndock != newScrollBeforeUndock) {
                         Footer.scrollBeforeUndock = newScrollBeforeUndock;
                         var scrollTop = document.body.scrollTop || document.documentElement.scrollTop;
                         Footer.checkScroll(scrollTop);
                 }
         }
 };

 Footer.checkScroll = function (scrollTop) {
         // do nothing if disabled (for design-guide)
         Footer.disabled = Footer.$body.hasClass('disable-docking-nav');
         if (Footer.disabled) return;

         // track thresholds for docking and shrinking
         if (scrollTop > Footer.scrollBeforeUndock && Footer.isDocked) Footer.undock();

         if (scrollTop <= Footer.scrollBeforeUndock && !Footer.isDocked) Footer.dock();
 };

 Footer.init();


});
