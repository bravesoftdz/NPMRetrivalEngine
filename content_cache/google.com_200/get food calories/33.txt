    }

    function bundlesReady(fn) {
        syAsyncInit.hasRun ? run(fn) : syAsyncInit.push(fn);
    }

    function flushBundlesReady() {
        _.map(syAsyncInit, run);
        syAsyncInit = [];
        syAsyncInit.hasRun = true;
    }

    function apiReady(params) {
        bundlesReady(function() {
            SymphonyAPI(params);
        });
    }

    function setLazyLoadReady() {
        lazyLoadReady = true;
        var toAddModulesNumber = externalNgModulesToAdd.length;
        if (toAddModulesNumber) {
            for (var i = 0; i < toAddModulesNumber; ++i) {
                externalNgModulesToAdd[i] && externalNgModulesToAdd[i]();
            }
            window.triggerLazyLoad();
        }
    }

    function isLazyLoadReady() { return lazyLoadReady; }

    function addNgModule(cb) {
        if (lazyLoadReady) {
            cb();
            window.triggerLazyLoad();
        } else {
            externalNgModulesToAdd.push(cb);
        }
    }

    return {
        bundlesReady: bundlesReady,
        flushBundlesReady: flushBundlesReady,
        apiReady: apiReady,
        addNgModule: addNgModule,
        setLazyLoadReady: setLazyLoadReady,
        isLazyLoadReady: isLazyLoadReady
    };
