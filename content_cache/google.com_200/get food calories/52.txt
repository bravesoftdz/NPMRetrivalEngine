			function toggleCursor(id){
				document.getElementById(id).style.cursor = "pointer";
			}

			function openPage(url){
				window.open("https://evhi.wisconsin.gov/evhi?appusertype=E", "_self", false);

			function displayBroadcast(errorMessage){
				for(i = 0; i < errorMessage.length; i++){
					var currentTime = new Date().getTime();
					var startTime = new Date(parseInt(errorMessage[i].notificationStartTime));
					var endTime = new Date(parseInt(errorMessage[i].notificationEndTime));
					var startDate = new Date(parseInt(errorMessage[i].notificationStartDate));
					var endDate = new Date(parseInt(errorMessage[i].notificationEndDate));
				
					startDate.setHours(startTime.getHours());
					startDate.setMinutes(startTime.getMinutes());
					endDate.setHours(endTime.getHours());
					endDate.setMinutes(endTime.getMinutes());
				
					if(currentTime > startDate.getTime() && currentTime < endDate.getTime()){
						document.all.broadcastTableClient.style.display='block'; 
						broadcastMessage = errorMessage[i].notificationMsg;
						
						var preErrHTML = document.all.broadcastSpanClient.innerHTML;
						if(preErrHTML != null && preErrHTML != ""){        
							prevErrMsg = preErrHTML + "<BR />";
						}else{    
							prevErrMsg = preErrHTML;   
						}   
					
						document.all.broadcastSpanClient.innerHTML=prevErrMsg+broadcastMessage+"<BR>";
					}
				}
			}

			function adjustImage() {
				var loginAccountButtonElement = document.getElementById("BUTTON_LOGIN_TO_ACCOUNT");
				var createAccountButtonElement = document.getElementById("BUTTON_CREATE_AN_ACCOUNT");

				loginAccountButtonElement.style.marginLeft = "0px";
				createAccountButtonElement.style.marginLeft = "0px";
				
				var width1 = parseFloat(document.defaultView.getComputedStyle(document.documentElement, null).width);
				var width2 = parseFloat(document.documentElement.clientWidth);
				var ratio = parseFloat(width1 / width2);
				
				if (ratio < 1) {
					createAccountButtonElement.style.marginLeft = "-1px";
				}
				else if (ratio > 1) {
					loginAccountButtonElement.style.marginLeft = "-1px";
				}
			}
			
			function displayWarning(warningMessage){
				var currentTime = new Date().getTime();
				var startTime = new Date(parseInt(warningMessage[0].notificationStartTime));
				var endTime = new Date(parseInt(warningMessage[0].notificationEndTime));
				var startDate = new Date(parseInt(warningMessage[0].notificationStartDate));
				var endDate = new Date(parseInt(warningMessage[0].notificationEndDate));
		
				startDate.setHours(startTime.getHours());
				startDate.setMinutes(startTime.getMinutes());
				endDate.setHours(endTime.getHours());
				endDate.setMinutes(endTime.getMinutes());
				
				if(currentTime < endDate.getTime()){
					var warningMsgNotification = [];
					
					for(i = 0; i < warningMessage.length; i++){
						if(warningMessage[i].notificationMsg){
							warningMsgNotification.push(warningMessage[i].notificationMsg);
							if(i == 0){
								warningMsgNotification.push("<BR><BR>");
							}
						}
					}
					
					if("<BR><BR>" == warningMsgNotification){
						document.getElementById("DowntimeContainer").style.display = 'none';
					} else {
						document.getElementById("DowntimeContainer").style.display = 'block';
						document.getElementById("DowntimeTop").innerHTML = warningMsgNotification.join(' ');
					}
				}
			}
