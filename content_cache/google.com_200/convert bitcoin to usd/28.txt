        PS.info.user.isClient = false;
        PS.info.user.isDemoClient = false;
        PS.info.user.isLoggedIn = false;
        PS.info.user.isProspect = true;

        //If there's the session open cookie, the user has logged in
        PS.info.user.isLoggedIn = PS.util.Page.sessionOpen();

        //Gets the ID cookie, to check if it's a client or not
        var idCookie = PS.util.Cookie.get('ID');
        if(!!idCookie){
            //If there's a A=1 or a CS inside the ID cookie, then the user can be a client or a demo client
            if((idCookie.indexOf('A=1') >= 0) || (idCookie.indexOf(':CS=') >= 0)){
                if(idCookie.indexOf('demo-account') < 0){
                    PS.info.user.isClient = true;

        }

    PS.startup.redirectUserToURL = function (userRedirectURL, transferQueryString) {
        var queryString = '';
        if(!!transferQueryString){
            queryString = window.location.search;
        }
    	var loc = window.location.href,
        	cqMode = (loc.indexOf('iggroup.local') > -1);

        //If the url exists and we are not inside the author, redirect
        if ((!!userRedirectURL) && !cqMode && (window.location.pathname !== userRedirectURL)) {
            if(window.PS.util.Browser.storeReferrer){
                window.PS.util.Browser.storeReferrer();
            }
            window.location = userRedirectURL + queryString;
        }
    };

    PS.startup.generateHashNumber = function() {
        //This cookie will contain the timestamp of the first access
        var id = PS.util.Cookie.get('user_ts');

        if(!id){
            id = Math.floor(Math.random() * 10000000000000);
            PS.util.Cookie.set('user_ts',id);
        }

        //This code is taken from Login.js
        var idMD5 = PS.util.getMD5(id + "");//make sure we pass a string. if you pass a number the returned value is always the same
        var hashnum = parseFloat(idMD5);
        hashnum /= 4294967296; //magic number from script attached to W-268850
        hashnum += 0.5; //another seemingly magic number
        return hashnum;
    };

    PS.startup.redirectUser = function(userRedirectsMap) {
        if(userRedirectsMap == undefined){
            return false;
        }

        var matchingList = [];

        //Updates and retrieves the user segment
        var currentSegment = PS.startup.getUserSegment();

        var percentageNormaliser = 0;

        //We run through the redirects, to find out which redirects are associated with the current user segment and
        //to find out the total number inside the percentage (to normalise it)
        for(var i = 0; i < userRedirectsMap.length; i++){
            var userSegmentRedirect = userRedirectsMap[i];
            if(userSegmentRedirect.segment === currentSegment){
                percentageNormaliser+=userSegmentRedirect.percentage;
                matchingList.push(userSegmentRedirect);
            }
        }

        //If there is at least one redirect with a percentage greater than 0, then we normalise the value and redirect
        //the user accordingly to his hash value
        if(percentageNormaliser > 0){
            var normalisedPercentage = 0;
            for(var j = 0; j < matchingList.length; j++){
                normalisedPercentage += matchingList[j].percentage/percentageNormaliser;
                if(PS.startup.generateHashNumber() <= normalisedPercentage){
                    PS.util.addUserSegment(matchingList[j].name);

                    PS.startup.redirectUserToURL(matchingList[j].redirectPath, true);
                    return true;
                }
            }
        }
        return false;

    };

    PS.util.addUserSegment = function(segment){
        if(!!segment){

            //Gets the list of segments
            var userSegmentList = PS.util.Cookie.get('user_sgm');

            var found = false;

            //If the list exists
            if(!!userSegmentList){
                var userSegmentsSet = userSegmentList.split(',');

                //We search if the segment is already stored
                for(var k = 0; k< userSegmentsSet.length;k ++){
                    if(userSegmentsSet[k] === segment){
                        found = true;
                    }
                }
            }

            //If the segment isn't inside the cookie
            if(!found){
                //If the list is empty the segment is written, otherwise is added
                if(!userSegmentList){
                    userSegmentList = segment;
                }
                else {
                    userSegmentList += ','+segment;
                }
            }
            //The cookie is stored again
            PS.util.Cookie.set('user_sgm',userSegmentList);
        }
    };

    //Check if the user is a client
    PS.startup.isClient = function () {
        PS.startup.updateUserSegment();
        return PS.info.user.isClient;
    };
