            });
        });
    });

        $('#tabContent').children('div').hide(); //hide all tab divs
        var urlLocation = document.location.toString();
        if (urlLocation.match('#')) {
            var urlAnchor = '#' + urlLocation.split('#')[1];
            var tabHeaderDiv = urlAnchor + "Header";
            $(tabHeaderDiv).addClass('activeTab');
            $(urlAnchor).fadeIn();
        } else {
            $('#tabContent div:first').show(); //show the first div
            $('#tabHeader ul li:first').addClass('activeTab'); //set the first tab to activeTab
        }

        //this function is for handling the tab header clicks
        $('#tabHeader ul li a').click(function() {
            $('#tabHeader ul li').removeClass('activeTab');
            $(this).parent().addClass('activeTab');
            var currentTab = $(this).attr('href');

            //fade in/fade out
            $('#tabContent').children('div:visible').fadeOut(function() {
                $(currentTab).fadeIn()
            });
            return false;
        });

            $('#tabHeader ul li').removeClass('activeTab');

            $(currentTabHeader).addClass('activeTab');

            //fade in/fade out
            $('#tabContent').children('div:visible').fadeOut(function() {
                $(currentTab).fadeIn()
            });
            return false;
        });

























});
