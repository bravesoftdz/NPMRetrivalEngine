// Function to validate Credit Card Numbers

    function validateCreditCard($cardType, $cardNo){
        $returnVal = true;
        if($cardType == "visa"){
            if (!preg_match("/^(?:4[0-9]{12}(?:[0-9]{3})?)$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }
        else if($cardType == "mastercard"){
            if (!preg_match("/^(?:5[1-5][0-9]{14})$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }
        else if($cardType == "americanexpress"){
            if (!preg_match("/^(?:3[47][0-9]{13})$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }
        else if($cardType == "dinersclub"){
            if (!preg_match("/^(?:3(?:0[0-5]|[68][0-9])[0-9]{11})$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }
        else if($cardType == "discover"){
            if (!preg_match("/^(?:6(?:011|5[0-9][0-9])[0-9]{12})$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }
        else if($cardType == "jcb"){
            if (!preg_match("/^(?:(?:2131|1800|35\d{3})\d{11})$/", $cardNo)) {
                $returnVal =  false;
            } else {
                $returnVal = true;
            }
        }

        return $returnVal;
    }
