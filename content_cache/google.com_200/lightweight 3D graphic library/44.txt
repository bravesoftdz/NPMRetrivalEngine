    var sourceLinkRoot = 'https://github.com/liabru/matter-tools/blob/master';

    MatterTools.Demo.setExample = Matter.Common.chainPathAfter(
        MatterTools, 'Demo.setExample', 
        function(demo, example) {
            var world = demo.example.instance.engine.world;
            var render = demo.example.instance.render;
            var renderOptions = render.options;

            var bodies = Matter.Composite.allBodies(world);

            for (var i = 0; i < bodies.length; i += 1) {
                var body = bodies[i];

                if (body.isStatic && body.label === 'Rectangle Body') {
                    body.render.fillStyle = 'transparent';
                    body.render.lineWidth = 1;
                    body.render.strokeStyle = '#aaa';
                }
            }

            var constraints = Matter.Composite.allConstraints(world);

            for (var i = 0; i < constraints.length; i += 1) {
                var constraint = constraints[i];
                constraint.render.strokeStyle = '#ccc';
            }

            renderOptions.background = 'transparent';
            renderOptions.wireframes = false;
            renderOptions.showDebug = false;
            renderOptions.showBroadphase = false;
            renderOptions.showBounds = false;
            renderOptions.showVelocity = false;
            renderOptions.showCollisions = false;
            renderOptions.showAxes = false;
            renderOptions.showPositions = false;
            renderOptions.showAngleIndicator = false;
            renderOptions.showIds = false;
            renderOptions.showShadows = false;
            renderOptions.showVertexNumbers = false;
            renderOptions.showConvexHulls = false;
            renderOptions.showInternalEdges = false;
            renderOptions.showSeparations = false;
        }
    );

    var demo = MatterTools.Demo.create({
        inline: true,
        preventZoom: true,
        resetOnOrientation: false,
        startExample: true,

        toolbar: {
            title: 'matter-js',
            reset: false,
            source: false,
            inspector: false,
            tools: false,
            fullscreen: false,
            exampleSelect: false
        },
        
        examples: [
            {
                name: 'Newton\'s Cradle',

                init: Example.newtonsCradle,
                sourceLink: sourceLinkRoot + '/newtonsCradle.js'
            },
            {

                init: Example.constraints,
                sourceLink: sourceLinkRoot + '/constraints.js'
            },
            {
                name: 'Cloth',
                id: 'cloth',
                init: Example.cloth,
                sourceLink: sourceLinkRoot + '/cloth.js'
            },
            {
                name: 'Concave',
                id: 'concave',

            },
            {

    var exampleIndex = 0;
    var interval;

    var startInterval = function() {
        interval = setInterval(function() {
            exampleIndex = (exampleIndex + 1) % demo.examples.length;

            MatterTools.Demo.setExample(demo, demo.examples[exampleIndex]);
        }, 2000);
    };

    var stopInterval = function() {
        clearInterval(interval);
    };

    canvasContainer.addEventListener('mousedown', stopInterval);
