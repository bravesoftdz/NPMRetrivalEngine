function _search_new() { // search for new categ structure 2015 (dmihai)
	$('#loading_container').show();
	$('#search_results_container').show();
	$('#mainContent').hide();
	$('#search_results').html('');
	if ($.isNumeric($("#srchtopic").val()) || $("#srchtopic").val().length <= 3) {
		$('#loading_container').hide();
		$('#search_results_container').show();
		$('#mainContent').hide();
		$('#search_results').html("A valid search query should contain at least one word and more than 3 characters.");
	} else {
		//omniture (copy-paste from _search function)
		if (typeof(s) == 'object') {
			// s=s_gi(s_account);
			s.linkTrackVars = 'prop22';
			s.linkTrackEvents = 'None';
			s.prop22 = 'main:'+$("#srchtopic").val();
			s.tl(false,'o', 'Support Center Search');
		}
		//done omniture
		
		var crt_lang = DEFAULT_LANGUAGE;
		var search_query = $('#srchtopic').val();
		var search_categs = window.search_categs;
		if (typeof(search_categs) == 'undefined') {
			var search_categs = [];
		}
		
		$.ajax({
			type: "post",
			url: "//www.bitdefender.com/site/KnowledgeBase/supportSearch/",
			data: {
				categs: search_categs,
				lang: crt_lang,
				query: search_query,
				ajax: 1
			},
			dataType: 'json',
			success: function(search_results) {
				$('#loading_container').hide();
				
				if(search_results == null || search_results.results == null || search_results.total_matches == '0') {
					_search_no_results_new();
				} else {
					if(search_results.total_matches != '0') {
						var articles = [];
						$.each(search_results.results, function(key, val) {
							var art_type = 'Article';
							if (val.article_type && val.article_type == 1)
								art_type = 'Video Tutorial';
							else if (val.article_type && val.article_type == 2)
								art_type = 'Manual';
							
							if (DEFAULT_LANGUAGE == 'frint')
								val.url = val.url.replace('bitdefender.com/','bitdefender.com/fr/');
							
							str = '<li> <a href="' + val.url + '" target="_blank">' + val.article_title + '</a> <span>(' + art_type + ')</span></li>';
							articles.push(str);
						});
						
						$('#search_results').html('<ol>'+ articles.join('')+'</ol>' + "<p>Didn't find what you're searching for? Try another search or <a href='//www.bitdefender.com/support/consumer.html'>see all our questions</a>.</p>" );
					} else {
						_search_no_results_new();
					}
				}
			}
		});
	}
}

	var text2 = "<p>Here is a list of some popular articles:</p>";
	
	if (DEFAULT_LANGUAGE == 'en' || text1.length != 'No articles were found matching your criteria. Make sure that words are spelled correctly or try different keywords.'.length) {
		$('#search_results').html('<p>'+text1+'</p>');

		}

	}
}
