            <h2>react-native-send-sms</h2> 
            <hr> 
            <p><strong>SendSMS</strong></p> 
            <blockquote> 
             <h5>A react-native component to send SMS messages. It works only for android. This component used android native api to send sms and response a callback to react-native. This react-native component desn't link to the mobile default messenger view. It will programmatically send the message to a phone number which is provided as a parameter and will response a callback with a string indicating message was sent or failure or no service.</h5> 
            </blockquote> 
            <p><em>usage</em></p> 
            <pre><code class="language-js">import SendSMS from 'react-native-send-sms';

SendSMS.send("+959254687254", "Hey.., this is me!\nGood to see you. Have a nice day.", (msg)=&gt;{ alert(msg) });
</code></pre> 
            <p>Response msg string will be one of the following:</p> 
            <ul> 
             <li>"SMS sent" - for successful message</li> 
             <li>"Generic failure" - for general failure</li> 
             <li>"No service" - for no mobile operator service</li> 
             <li>"Radio off" - for no mobile signal</li> 
             <li>"Null PDU" - for no PDU</li> 
            </ul> 
            <h6><em>Note:</em></h6> 
            <h6>Minimum android version is ==4.1== and supported ==RN &gt;= v0.29==.</h6> 
            <hr> 
            <h4>Installation</h4> 
            <pre><code>npm install react-native-send-sms --save
</code></pre> 
            <hr> 
            <h5><strong>Android Setup</strong></h5> 
            <p>1.In your <code>android/settings.gradle</code> file, make the following additions:</p> 
            <pre><code>include ':react-native-send-sms'
project(':react-native-send-sms').projectDir = new File(rootProject.projectDir, '../node_modules/react-native-send-sms/android/app')
</code></pre> 
            <p>2.In your <code>android/app/build.gradle</code> file, add the ':react-native-send-sms' project as a compile-time dependency:</p> 
            <pre><code>...
dependencies {
    ...
    compile project(':react-native-send-sms')
}
</code></pre> 
            <p>3.Update the <code>MainApplication.java</code> file as follow:</p> 
            <pre><code class="language-java">import com.facebook.react.ReactApplication;
...
import com.someone.sendsms.SendSMSPackage; // &lt;--- add here!


public class MainApplication extends Application implements ReactApplication {

  private final ReactNativeHost mReactNativeHost = new ReactNativeHost(this) {
    @Override
    protected boolean getUseDeveloperSupport() {
      return BuildConfig.DEBUG;
    }

    @Override
    protected List&lt;ReactPackage&gt; getPackages() {
      return Arrays.&lt;ReactPackage&gt;asList(
          new MainReactPackage(),
          new SendSMSPackage()     // &lt;--- add here!
      );
    }
  };

  @Override
  public ReactNativeHost getReactNativeHost() {
      return mReactNativeHost;
  }
}
</code></pre> 
            <p>4.In your <code>AndroidManifest.xml</code> file, add a user permission for sending SMS.</p> 
            <pre><code>&lt;uses-permission android:name="android.permission.SEND_SMS" /&gt;
</code></pre> 
            <hr> 
            <p><strong>Example</strong></p> 
            <pre><code class="language-js">import React, { Component } from 'react';
import {
  AppRegistry,
  StyleSheet,
  Text,
  View,
  TouchableOpacity,
  ToastAndroid
} from 'react-native';
import SendSMS from 'react-native-send-sms';

export default class RNSMS extends Component {
  sendSMSFunction() {
    SendSMS.send("+95912345678", "Hey.., this is me!\nGood to see you. Have a nice day.",
      (msg)=&gt;{
        ToastAndroid.show(msg, ToastAndroid.SHORT);
      }
    );
  }
  render() {
    return (
      &lt;View style={styles.container}&gt;        
        &lt;TouchableOpacity style={styles.button} onPress={this.sendSMSFunction.bind(this)}&gt;
          &lt;Text&gt;Send SMS&lt;/Text&gt;
        &lt;/TouchableOpacity&gt;
      &lt;/View&gt;
    );
  }
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    backgroundColor: '#F5FCFF',
  },  
  button: {
    padding: 10,
    borderWidth: .5,
    borderColor: '#bbb',
    margin: 10,
    alignItems: 'center',
    justifyContent: 'center'
  }
});

AppRegistry.registerComponent('RNSMS', () =&gt; RNSMS);
</code></pre>
