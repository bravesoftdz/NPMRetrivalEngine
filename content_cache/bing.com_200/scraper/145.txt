    function setEqualHeight(columns) {
        var tallestcolumn = 0;
        columns.each(
        function() {
            currentHeight = $(this).height();
            if(currentHeight > tallestcolumn) {
                tallestcolumn  = currentHeight;
                }
            }
        );
     columns.height(tallestcolumn);
    }

    var delay = (function(){
            var timer = 0;
            return function(callback, ms){
                clearTimeout (timer);
                timer = setTimeout(callback, ms);
        };
    })();

        }, 500);
