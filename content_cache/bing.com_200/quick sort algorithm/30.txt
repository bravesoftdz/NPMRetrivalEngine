       <p><b>Basic Idea</b></p> 
       <ul type="DISC"> 
        <li>Pick one element in the array, which will be the <i>pivot</i>.</li> 
        <li>Make one pass through the array, called a <i>partition</i> step, re-arranging the entries so that: 
         <ul type="DISC"> 
          <li>entries smaller than the pivot are to the left of the pivot.</li>

#include &lt;iostream&gt;

void PrintArray(int *array, int n) {
  for (int i = 0; i &lt; n; ++i)
    std::cout &lt;&lt; array[i] &lt;&lt; " " &lt;&lt; std::flush;
  std::cout &lt;&lt; std::endl;
}

inline void Swap(int &amp;a, int &amp;b){
    int k = a;
    a = b;
    b = k;
}

//reload rand to produce random number in a fixed range
inline int rand(int p, int q){
    int size = q - p + 1;
//    srand(time(NULL));
    return (p + rand() % size);
}
int Partition(int arr[], int lo, int hi){
    //produce ramdom subscript
    int t = rand(lo, hi);
    Swap(arr[t], arr[hi]);

    int index = lo - 1;
    int key = arr[hi];
    for(int i = lo ; i &lt; hi; i++){
        if(arr[i] &lt;= key)
            Swap(arr[++index], arr[i]);
    }
    Swap(arr[++index], arr[hi]);
    return index;
}
void QuickSortHelper(int arr[], int lo, int hi){
    if(lo &lt; hi){
        int index = Partition(arr, lo, hi);
        QuickSortHelper(arr, lo, index-1);
        QuickSortHelper(arr, index+1, hi);
    }
}

void QuickSort(int arr[], int arr_size){
    QuickSortHelper(arr, 0, arr_size-1);
}


int main() {
  int array[] = {94, 42, 50, 95, 333, 65, 54, 456, 1, 1234};
  int n = sizeof(array)/sizeof(array[0]);

  std::cout &lt;&lt; "Before Quick Sort :" &lt;&lt; std::endl;
  PrintArray(array, n);

  QuickSort(array, n);

  std::cout &lt;&lt; "After Quick Sort :" &lt;&lt; std::endl;
  PrintArray(array, n);
  return (0);
}
