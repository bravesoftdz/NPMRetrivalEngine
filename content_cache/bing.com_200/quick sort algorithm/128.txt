          <h1 style="font-family:'HelveticaNeue-Light', 'Helvetica Neue Light', 'Helvetica Neue', Helvetica, Arial, 'Lucida Grande', sans-serif;font-weight:300;font-size:27px;letter-spacing:2px;text-shadow:2px 2px #282828 , 3px 3px #383838 , 4px 4px #484848 , 5px 5px #585858;color:white;text-align:center;letter-spacing:2px;border:2px solid;border-radius:25px;border-color:darkorchid;box-shadow:5px 5px 2px darkorchid;border-style:outset;-moz-background-size:30% 100%;background-size:30% 100%;background-image:url('http://searchengineland.com/figz/wp-content/seloads/2012/02/html-Title-tag-best-practices-SEO.jpg');-moz-border-radius:25px;background-repeat:no-repeat;width:55%;position:relative;top:50%;left:20%;"><i style="font-size:35px;line-height:2em;padding-left:130px;">quickSort.cpp</i></h1> 
          <pre class="brush: cpp; title: ; notranslate" title="">
#include &lt;iostream&gt;

void quickSort(int a[], int first, int last);
int pivot(int a[], int first, int last);
void swap(int&amp; a, int&amp; b);
void swapNoTemp(int&amp; a, int&amp; b);
void print(int array[], const int&amp; N);

using namespace std;

int main()
{
    int test[] = { 7, -13, 1, 3, 10, 5, 2, 4 };
    int N = sizeof(test)/sizeof(int);

    cout &lt;&lt; "Size of test array :"  &lt;&lt; N &lt;&lt; endl;

    cout &lt;&lt; "Before sorting : " &lt;&lt; endl;
    print(test, N);

    quickSort(test, 0, N-1);

    cout &lt;&lt; endl &lt;&lt; endl &lt;&lt; "After sorting : " &lt;&lt; endl;
    print(test, N);
    
    return 0;
}

/**
 * Quicksort.
 * @param a - The array to be sorted.
 * @param first - The start of the sequence to be sorted.
 * @param last - The end of the sequence to be sorted.
*/
void quickSort( int a[], int first, int last ) 
{
    int pivotElement;

    if(first &lt; last)
    {
        pivotElement = pivot(a, first, last);
        quickSort(a, first, pivotElement-1);
        quickSort(a, pivotElement+1, last);
    }
}

/**
 * Find and return the index of pivot element.
 * @param a - The array.
 * @param first - The start of the sequence.
 * @param last - The end of the sequence.
 * @return - the pivot element
*/
int pivot(int a[], int first, int last) 
{
    int  p = first;
    int pivotElement = a[first];

    for(int i = first+1 ; i &lt;= last ; i++)
    {
        /* If you want to sort the list in the other order, change "&lt;=" to "&gt;" */
        if(a[i] &lt;= pivotElement)
        {
            p++;
            swap(a[i], a[p]);
        }
    }

    swap(a[p], a[first]);

    return p;
}


/**
 * Swap the parameters.
 * @param a - The first parameter.
 * @param b - The second parameter.
*/
void swap(int&amp; a, int&amp; b)
{
    int temp = a;
    a = b;
    b = temp;
}

/**
 * Swap the parameters without a temp variable.
 * Warning! Prone to overflow/underflow.
 * @param a - The first parameter.
 * @param b - The second parameter.
*/
void swapNoTemp(int&amp; a, int&amp; b)
{
    a -= b;
    b += a;// b gets the original value of a
    a = (b - a);// a gets the original value of b
}

/**
 * Print an array.
 * @param a - The array.
 * @param N - The size of the array.
*/
void print(int a[], const int&amp; N)
{
    for(int i = 0 ; i &lt; N ; i++)
        cout &lt;&lt; "array[" &lt;&lt; i &lt;&lt; "] = " &lt;&lt; a[i] &lt;&lt; endl;
}
