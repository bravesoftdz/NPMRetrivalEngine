     <h1>jvm-binary</h1> 
     <p>A library for reading and writing Java class-files. To get started importing the <code>Language.JVM</code> file should be sufficient for most uses.</p> 
     <p>If you want to access Code elements of methods it is recommended to import <code>Language.JVM.Attribute.Code</code> qualified, like this:</p> 
     <pre class="haskell"><code>import           Language.JVM
import qualified Language.JVM.Attribute.Code as Code
</code></pre> 
     <h2>Todo's</h2> 
     <ul> 
      <li><p>Add more Attributes as to the <a href="http://docs.oracle.com/javase/specs/jvms/se7/html/jvms-4.html#jvms-4.7">docs</a>. The most notable and required are:</p> 
       <ul> 
        <li>LineNumberTable</li> 
       </ul></li> 
      <li><p>Add lenses for better access of deep fields</p></li> 
      <li><p>Add documentation for Code</p></li> 
     </ul> 
     <h2>Developing</h2> 
     <p>Use stack to build, test, and benchmark.</p>

  var votesUrl = '/package/jvm-binary/votes';

    if (star.selected === false) {

    fill_stars(3, "out");
    fill_stars(this.id, "in");
    star.selected = true;
    star.id = this.id;
    var formData = {
      score: this.id
    }
    $.post(votesUrl, formData).done(function(data) {
        if(data != "Package voted for successfully") {
            modal.open({ content: data});
	}
    });
  });
  $('.clear-rating').click(function(e) {
    e.preventDefault()
    fill_stars(3, "out");
    star.selected = false;
    $.ajax({
      url: votesUrl,
      type: 'DELETE',
      success: function(result) {
        if(result != "Package vote removed successfully") {
          modal.open({ content: result });
	}
      }
    });
  });
  $(function() {

  var fill_stars = function(num, direction) {
    if (direction === "in")
      for (i = 0; i <= parseInt(num); i++)
        $("#" + i).removeClass('uncool').addClass('cool');
    else
      for (i = 0; i <= parseInt(num); i++)
        $("#" + i).removeClass('cool').addClass('uncool');
