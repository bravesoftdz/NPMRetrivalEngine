        $("#form-cta").submit(function (e) {
            e.preventDefault();
            var nome = $('input[name=nome]').val();
            var email = $('input[name=email]').val();
            var ddd = $('input[name=ddd]').val();
            var telefone = $('input[name=telefone]').val();
            var empresa = $('input[name=empresa]').val();
            var mensagem = $('textarea[name=mensagem]').val();
            var nodeId = $('input[name=parentId]').val();
            var pagina = window.location.href;
            $.ajax({
                type: 'POST',
                url: '/umbraco/surface/Content/SendSoliciteContatoJa',
                beforeSend: function () {
                    $("#btnEnviarMsg").text("Sending...");
                },
                dataType: "json",
                data: { nome: nome, email: email, ddd: ddd, telefone: telefone, empresa: empresa, mensagem: mensagem, nodeId: nodeId, pagina: pagina },
                success: function (data) {
                    if (data.success) {

                        swal({ title: "Message successfully sent!", text: "Your message was successfully sent. Soon, we will reply it!", type: "success", confirmButtonColor: "#c52727" });

                        $('input[name=nome]').val("");
                        $('input[name=email]').val("");
                        $('input[name=telefone]').val("");
                        $('input[name=ddd]').val("");
                        $('input[name=empresa]').val("");
                        $('textarea[name=mensagem]').val("");
                    }  else {
                        swal({ title: "Sending failed", text: "There was an error in sending your message!", type: "error" });
                    }
                },
                error: function (data) {

                    swal({ title: "Sending failed", text: "There was an error in sending your message!", type: "error" });

                },
                complete: function () {
                    $("#btnEnviarMsg").text("Send");
                    $('.fancy-select .trigger').text('Select');
                    $("#ctaButton").trigger('click');
                }

            });
        });
    </script>
