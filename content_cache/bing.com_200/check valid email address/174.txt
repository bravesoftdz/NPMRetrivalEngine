       <h1 class="entry-title">How to validate email address in&nbsp;iOS/Objective-C</h1> 
       <div class="entry-content clear"> 
        <div class="pd-rating" id="pd_rating_holder_7383096_post_12"></div>
        <br>
        <h3><strong style="font-size:1.17em;line-height:1;">Background:~</strong></h3> 
        <p dir="ltr" style="text-align:left;">Unlike other languages, in iOS sometimes we need to verify email address entered by user, especially in registration &amp; login process.</p> 
        <div dir="ltr" style="text-align:left;"> 
         <p>Following function will check the entered email address is valid or not and return the value based on validation.</p> 
         <div dir="ltr" style="line-height:1;margin-bottom:0;margin-top:0;"> 
          <h3 style="text-align:justify;"><strong style="font-size:1.17em;line-height:1;">Implementation:~</strong></h3> 
          <p style="text-align:justify;"><span style="color:#000000;"><i><span style="text-decoration:underline;">Utility.h</span></i></span></p> 
         </div> 
         <pre style="color:#008099;">#import &lt;Foundation/Foundation.h&gt;

@interface Utility : NSObject

+ (BOOL)isValidEmailAddress:(NSString *)emailAddress ;

+ (BOOL) validateEmail:(NSString*) emailAddress ;

@end</pre> 
         <p><span style="color:#000000;"><i><span style="text-decoration:underline;">Utility.m</span></i></span></p> 
         <pre style="color:#008099;">#import "Utility.h"

@implementation Utility

// Using NSPredicate

+ (BOOL)isValidEmailAddress:(NSString *)emailAddress {

    //Create a regex string
    NSString *stricterFilterString = @"[A-Z0-9a-z._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,4}" ;

    //Create predicate with format matching your regex string
    NSPredicate *emailTest = [NSPredicatepredicateWithFormat:
                              @"SELF MATCHES %@", stricterFilterString];

    //return true if email address is valid
    return [emailTest evaluateWithObject:emailAddress];
}

// Using NSRegularExpression

+ (BOOL) validateEmail:(NSString*) emailAddress {

    NSString *regExPattern = @"^[A-Z0-9._%+-]+@[A-Z0-9.-]+\\.[A-Z]{2,4}$";
    NSRegularExpression *regEx = [[NSRegularExpressionalloc]
                                  initWithPattern:regExPattern
                                  options:NSRegularExpressionCaseInsensitive
                                  error:nil];
    NSUInteger regExMatches = [regEx numberOfMatchesInString:emailAddress
                                                     options:0
                                                       range:NSMakeRange(0, [emailAddress length])];
    return (regExMatches == 0) ? NO : YES ;

}</pre> 
        </div> 
        <h3><strong>How to Use:~</strong></h3> 
        <p>Validate email address by calling any of the above method. For example,<span style="line-height:1.5em;">&nbsp;</span></p> 
        <pre style="color:#008099;">[Utility isValidEmailAddress:@"milan@example.com"]; OR 
[Utility validateEmail:@"sam@example.com"];</pre>
