      <h1>How to validate email address by using regular expression in java</h1> 
      <img align="middle" height="18" width="650" src="http://java2db.com/wp-content/themes/strProPlus/images/newdevider.jpg"> 
      <!--<div class="dateonhome">
				by &nbsp<a href="http://java2db.com/author/boss" title="Posts by boss" rel="author">boss</a>&nbsp;  on August 15, 2013 &nbsp | &nbsp <a href="http://java2db.com/java-lang/how-to-validate-email-address-by-using-regular-expression-in-java#respond">Leave a Comment</a>		    </div>--> 
      <br>
      <br> 
      <div style="border-bottom:1px dotted #ffffff; margin-bottom:-9px; padding:0px 0px 0px 0px; clear:both;"></div> 
      <p><a href="http://docs.oracle.com/javase/7/docs/api/java/util/regex/Pattern.html"><span style="color: #008000;">Regular Expression</span></a> is very useful to apply validation for various input fields like <span style="color: #ff9900;">email address</span>, <span style="color: #3366ff;">password</span> . To validate email address using regular expression , use the below conditions.</p> 
      <p>&nbsp;</p> 
      <h2 style="font-size: 12px;">Conditions to validate email address :</h2> 
      <table width="100%"> 
       <tbody> 
        <tr> 
         <td><span style="color: #ff6600;">^<br> </span></td> 
         <td><span style="color: #ff6600;">Beginning of the <a href="http://docs.oracle.com/javase/7/docs/api/java/util/regex/Pattern.html">regex</a>.</span></td> 
        </tr> 
        <tr> 
         <td><span style="color: #008000;">[a-zA-Z]</span></td> 
         <td><span style="color: #008000;">Small or capital alphabet</span>.(Ex :&nbsp; <span style="color: #ff0000;"><strong><a href="mailto:abcde@xyz.com"><span style="color: #ff0000;">a</span></a></strong></span>bcde@xyz.com or <span style="color: #ff0000;"><strong>A</strong></span>bcd@xyz.com).</td> 
        </tr> 
        <tr> 
         <td><span style="color: #800080;">(?!.*\\s)</span></td> 
         <td>To check blank spaces.</td> 
        </tr> 
        <tr> 
         <td><span style="color: #ff00ff;">.{4,15}</span></td> 
         <td><span style="color: #ff00ff;">Length should be 4 min and 15 max </span>(ie <strong><span style="color: #ff0000;"><span style="color: #ff0000;">abcd</span>e</span></strong>@xyz.com).</td> 
        </tr> 
        <tr> 
         <td><span style="color: #00ccff;">[a-z]{4,10}</span></td> 
         <td><span style="color: #00ccff;">Small alphabets with 4 min and 10 max length</span> (ie <span style="color: #333333;"><span style="color: #333333;">abcd</span></span><a>@<span style="color: #ff0000;"><strong>xyz</strong></span>.com</a>).</td> 
        </tr> 
        <tr> 
         <td><span style="color: #808080;">[.]</span></td> 
         <td><span style="color: #ff0000;"><strong>&nbsp;.</strong></span> character (ie abcd@xyz<strong><span style="color: #ff0000;">.</span></strong>com)</td> 
        </tr> 
        <tr> 
         <td><span style="color: #ff0000;"><strong>[a-z]{2,5}</strong></span></td> 
         <td><span style="color: #ff0000;">Small alphabets with&nbsp;2 min and&nbsp;5 max length</span> (ie abcd<a>@xyz.<strong><span style="color: #ff0000;">com</span></strong></a>).</td> 
        </tr> 
       </tbody> 
      </table> 
      <h2 style="font-size: 12px;"></h2> 
      <h2 style="font-size: 12px;"></h2> 
      <h2 style="font-size: 12px;"><span style="color: #666699;">Sample regular expression to validate email address :</span></h2> 
      <p><strong><span style="color: #ff6600;">^</span><span style="color: #008000;">[a-zA-Z]</span><span style="color: #800080;">(?!.*\\s)</span><span style="color: #ff00ff;">.{4,15}</span>@(<span style="color: #00ccff;">[a-z]{4,10}</span>)<span style="color: #666699;">[.]</span><span style="color: #ff0000;">[a-z]{2,5}</span></strong></p> 
      <p>&nbsp;</p> 
      <h2 style="font-size: 12px;"></h2> 
      <h2 style="font-size: 12px;"><span style="color: #33cccc;">Example code to validate email address:</span></h2>

import java.util.regex.Matcher;
import java.util.regex.Pattern;
class  emailAddress
{
	public static void main(String[] args) 
	{
	 String emailsToTest[] = new String[5];
	          //valid according to our regex
		   emailsToTest[0] = "abcde@xyz.com";
		   //starts with number is invalid
	         emailsToTest[1] = "3abcd@xyz.com";
		   //length of abcd is &lt; 4 (string position starts from 0)
		   emailsToTest[2] = "abcd@xyz.com";
		   //all numbers is invalid
		   emailsToTest[3] = "232423@xyz.co";
		    //without @ is invalid
		   emailsToTest[4] = "abcdexyz.co";

    String regex="^[a-zA-Z](?!.*\\s).{4,15}@[a-z]{3,10}[.][a-z]{2,5}";

	Pattern pattern = Pattern.compile(regex);
    for(String emailAddres: emailsToTest)
	{
	Matcher match = pattern.matcher(emailAddres);
    boolean result = match.matches();
	System.out.println(emailAddres+ " is "+result+" according to our regex");	
	}
	}
}</textarea>
