        <h1 " id="page-title"> Using Java Based Samples to Convert PDF to TIFF </h1> 
        <div class="region region-content"> 
         <div id="block-system-main" class="block block-system"> 
          <div class="content"> 
           <div id="node-334" class="node node-tech-tip node-full clearfix" about="/resources/techtips/conversion/using-java-based-samples-convert-pdf-tiff" typeof="sioc:Item foaf:Document"> 
            <span property="dc:title" content="Using Java Based Samples to Convert PDF to TIFF" class="rdf-meta element-hidden"></span>
            <span property="sioc:num_replies" content="0" datatype="xsd:integer" class="rdf-meta element-hidden"></span> 
            <div class="content clearfix"> 
             <div class="field field-name-body field-type-text-with-summary field-label-hidden">
              <div class="field-items">
               <div class="field-item even" property="content:encoded">
                <p>PDF to TIFF file conversion is often accomplished by using a printer driver. If you are an individual user only converting a few files, this method is sufficient. However, if you have multiple users dealing with a large number of documents and images across your organization, using a tool to convert your images is a more efficient method. Snowbound Software provides tools that convert PDF to TIFF files directly and rapidly without having to use a printer driver.<br> Snowbound Software’s RasterMaster Java Imaging SDK provides samples to allow you to automatically convert from the PDF to TIFF file format. The simpleconvertionDS and converter samples appear below and can be found in the Samples folder included with RasterMaster for the Java TM Platform:</p> 
                <p>Simpleconvertionds Sample</p> 
                <p>package simpleconvertionds;</p> 
                <p>/**</p> 
                <p>* Title: Simple Conversion Sample</p> 
                <p>* Description: Takes the path to an image you want to convert and the</p> 
                <p>* Format Name or Integer Value of the type of file format you want to output.</p> 
                <p>* It then saves the image to the current directory as "out.sbd".</p> 
                <p>* Copyright: Copyright (c) 2005</p> 
                <p>* Company: Snowbound Software, Inc.</p> 
                <p>* @author Martin Peloquin</p> 
                <p>* @version 1.0</p> 
                <p>*/</p> 
                <p>import java.awt.*;</p> 
                <p>import java.io.*;</p> 
                <p>import java.lang.Throwable;</p> 
                <p>import Snow.*;</p> 
                <p>public class ds</p> 
                <p>{</p> 
                <p>ConsoleDisplay display = null;</p> 
                <p>Snowbnd simage = null;</p> 
                <p>File inputfile = null;</p> 
                <p>File outputfile = null;</p> 
                <p>int saveformatvalue = 0;</p> 
                <p>//***********************************************</p> 
                <p>//**CONSTRUCTOR**********************************</p> 
                <p>//***********************************************</p> 
                <p>//Constructor: Determines if the arguments passed are valid or not then passes</p> 
                <p>// those arguments to the proper run method</p> 
                <p>public ds(String pargs[])</p> 
                <p>{</p> 
                <p>//Initialize display console</p> 
                <p>display = new ConsoleDisplay();</p> 
                <p>try{</p> 
                <p>//Check for valid number of arguments</p> 
                <p>if (pargs.length != 0) {</p> 
                <p>//Check if it is a valid input file</p> 
                <p>setInputFile(pargs[0]);</p> 
                <p>if (pargs.length &gt;= 2) {</p> 
                <p>setSaveFormat(pargs[1]);</p> 
                <p>}</p> 
                <p>//Initialize Output File Name</p> 
                <p>outputfile = new File(inputfile.getName()+".sbd");</p> 
                <p>//Initialize Snowbound Image Object and Run Conversion</p> 
                <p>simage = new Snowbnd();</p> 
                <p>run();</p> 
                <p>}</p> 
                <p>else //Display No Arguments</p> 
                <p>display.addText("ERROR: Incorrect number of arguments -- Optional Output Format Name or Integer Value&gt;");</p> 
                <p>}catch(Throwable t){</p> 
                <p>//Display All Thrown Error Messages</p> 
                <p>display.addText(t.getMessage());</p> 
                <p>}</p> 
                <p>}</p> 
                <p>//***********************************************</p> 
                <p>//**MAIN ACTION METHODS: RUN**********************************************</p> 
                <p>//***********************************************</p> 
                <p>private void run() throws Throwable</p> 
                <p>{</p> 
                <p>//Decompress Image</p> 
                <p>int pg = 0;</p> 
                <p>int stat = simage.IMG_decompress_bitmap(inputfile.toString(),pg);</p> 
                <p>if (stat &lt; 0)</p> 
                <p>throw (new Throwable("ERROR: Could not decompress Image "+inputfile.toString()));</p> 
                <p>//Display Decompressed image info</p> 
                <p>display.addText(" OK: *******Decompress Successful*******");</p> 
                <p>display.addText(" OK: Input File Name -- " + inputfile.toString());</p> 
                <p>display.addText(" OK: Input File Size -- " + inputfile.length());</p> 
                <p>stat = simage.IMGLOW_get_filetype(inputfile.toString());</p> 
                <p>if (stat &gt;= 0){</p> 
                <p>display.addText(" OK: Input File Type -- " + FormatHash.getFormatName(stat));</p> 
                <p>display.addText(" OK: Input File Type Value -- " + stat);</p> 
                <p>}</p> 
                <p>else</p> 
                <p>display.addText("ERROR: Could not retrieve input file File Type: Error Code -- " + stat);</p> 
                <p>display.addText(" OK: Input Number of Pages -- " + simage.IMGLOW_get_pages(inputfile.toString()));</p> 
                <p>display.addText(" OK: Input Page Decompressed -- " + pg);</p> 
                <p>display.addText(" OK: Width -- " + simage.getWidth());</p> 
                <p>display.addText(" OK: Height -- " + simage.getHeight());</p> 
                <p>display.addText(" OK: XDpi -- " + simage.getXdpi());</p> 
                <p>display.addText(" OK: YDpi -- " + simage.getXdpi());</p> 
                <p>//Save Image</p> 
                <p>stat = simage.IMG_save_bitmap(outputfile.toString(),saveformatvalue);</p> 
                <p>if (stat &lt; 0)</p> 
                <p>throw (new Throwable("ERROR: Could not Save Image -- Error Code: "+stat+" -- Format: "+saveformatvalue+" = "+FormatHash.getFormatName(saveformatvalue)));</p> 
                <p>//Display Saved Image Info</p> 
                <p>display.addText(" ");</p> 
                <p>display.addText(" OK: *******Save Successful*******");</p> 
                <p>display.addText(" OK: Output File Name -- "+outputfile.toString());</p> 
                <p>display.addText(" OK: Output File Size -- "+stat);</p> 
                <p>display.addText(" OK: Output File Type -- "+FormatHash.getFormatName(saveformatvalue));</p> 
                <p>display.addText(" OK: Output File Type Value -- "+saveformatvalue);</p> 
                <p>stat = simage.IMGLOW_get_pages(outputfile.toString());</p> 
                <p>if (stat &gt;= 1){</p> 
                <p>display.addText(" OK: Output Number of Pages -- " + stat);</p> 
                <p>display.addText(" OK: Output Image Save To Page -- " + (stat + 1));</p> 
                <p>}</p> 
                <p>else</p> 
                <p>display.addText("ERROR: Could not retrieve number of pages of output file: Error Code -- " + stat);</p> 
                <p>display.addText(" OK: Width -- "+simage.getWidth());</p> 
                <p>display.addText(" OK: Height -- "+simage.getHeight());</p> 
                <p>display.addText(" OK: XDpi -- "+simage.getXdpi());</p> 
                <p>display.addText(" OK: YDpi -- "+simage.getXdpi());</p> 
                <p>}</p> 
                <p>//***********************************************</p> 
                <p>//**SUPPORT METHODS******************************<br> //***********************************************</p> 
                <p>private void setInputFile(String infile) throws Throwable</p> 
                <p>{</p> 
                <p>inputfile = new File(infile);</p> 
                <p>if (!inputfile.isFile())</p> 
                <p>throw (new Throwable("ERROR: Invalid First Argumemt -- Need Image File Name"));</p> 
                <p>}</p> 
                <p>private void setSaveFormat(String sformat) throws Throwable</p> 
                <p>{</p> 
                <p>//Check if format is valid</p> 
                <p>try {</p> 
                <p>saveformatvalue = Integer.parseInt(sformat);</p> 
                <p>if (!FormatHash.isValidFormat(saveformatvalue))</p> 
                <p>throw (new Throwable("ERROR: Invalid Second Argument -- Not a Recognized Snowbound Format Integer"));</p> 
                <p>}catch (java.lang.NumberFormatException nfe2) {</p> 
                <p>if (!FormatHash.isValidFormat(sformat.toUpperCase()))</p> 
                <p>throw (new Throwable("ERROR: Invalid Second Argument -- Not a Recognized Snowbound Format Name"));</p> 
                <p>saveformatvalue = FormatHash.getFormat(sformat.toUpperCase());</p> 
                <p>}</p> 
                <p>}</p> 
                <p>//***********************************************</p> 
                <p>//**START POINT**********************************</p> 
                <p>//***********************************************</p> 
                <p>public static void main(String[] args)</p> 
                <p>{</p> 
                <p>ds ds1 = new ds(args);</p> 
                <p>}</p> 
                <p>}</p> 
                <p>Converter Sample</p> 
                <p>package converter;</p> 
                <p>import java.io.File;</p> 
                <p>import Snow.*;</p> 
                <p>public class Convert</p> 
                <p>{</p> 
                <p>public File f=null;</p> 
                <p>private String[] inputFiles=null;</p> 
                <p>private String loadDirectory=null;</p> 
                <p>private String saveDirectory=null;</p> 
                <p>private int outputFileType;</p> 
                <p>private String extension;</p> 
                <p>private Snowbnd Simage=null;</p> 
                <p>private boolean createThumbs=false;</p> 
                <p>public Convert(String loadDir, String saveDir, int fileType, String outputExtension)</p> 
                <p>{</p> 
                <p>extension=outputExtension;</p> 
                <p>Simage= new Snowbnd();</p> 
                <p>setSaveDirectory(saveDir);</p> 
                <p>setLoadDirectory(loadDir);</p> 
                <p>setOutputFileType(fileType);</p> 
                <p>f = new File(loadDir);</p> 
                <p>inputFiles = f.list();</p> 
                <p>}</p> 
                <p>public void setSaveDirectory(String save)</p> 
                <p>{</p> 
                <p>saveDirectory = save;</p> 
                <p>}</p> 
                <p>public void setLoadDirectory(String load)</p> 
                <p>{</p> 
                <p>loadDirectory = load;</p> 
                <p>}</p> 
                <p>public void setOutputFileType(int ftype)</p> 
                <p>{</p> 
                <p>outputFileType = ftype;</p> 
                <p>}</p> 
                <p>public String getSaveDirectory()</p> 
                <p>{</p> 
                <p>return saveDirectory;</p> 
                <p>}</p> 
                <p>public String getLoadDirectory()</p> 
                <p>{</p> 
                <p>return loadDirectory;</p> 
                <p>}</p> 
                <p>public int getOutputFileType()</p> 
                <p>{</p> 
                <p>return outputFileType;</p> 
                <p>}</p> 
                <p>public void setJPEGparameters(int quality, int horizontalInterleave, int verticalInterleave)</p> 
                <p>{</p> 
                <p>Simage.IMGLOW_set_comp_quality(quality);</p> 
                <p>Simage.IMGLOW_set_jpg_interleave(horizontalInterleave,<br> verticalInterleave);</p> 
                <p>}</p> 
                <p>public void setThumbnailCreation(boolean create)</p> 
                <p>{</p> 
                <p>createThumbs = create;</p> 
                <p>}</p> 
                <p>private void makeThumbnail(String outName, int outType)</p> 
                <p>{</p> 
                <p>int bits = Simage.getBitsPerPixel();</p> 
                <p>int horiz = 120;</p> 
                <p>//calculate the vertical height to maintain aspect ratio of thumbnail</p> 
                <p>int vert = (int)(((double)horiz / (double)Simage.getWidth()) * (double)Simage.getHeight());</p> 
                <p>if (bits == 1)</p> 
                <p>Simage.IMG_resize_to_gray(horiz,vert);</p> 
                <p>else</p> 
                <p>Simage.IMG_resize_bitmap_bicubic(horiz,vert);</p> 
                <p>Simage.IMG_save_bitmap(outName,outType);</p> 
                <p>}</p> 
                <p>public void performConversion()</p> 
                <p>{</p> 
                <p>int status;</p> 
                <p>String inputFileName = null;</p> 
                <p>String outputFileName = null;</p> 
                <p>String baseFileName = null;</p> 
                <p>int extnLocation;</p> 
                <p>//Loop through each file in the directory</p> 
                <p>for (int i=0; i &lt; inputFiles.length; i++)</p> 
                <p>{</p> 
                <p>//Gets the location of the . so we can pull off the extension</p> 
                <p>//if there is no extension sets it to the end of the file.</p> 
                <p>if (inputFiles[i].indexOf(".") &lt; 1)</p> 
                <p>extnLocation = inputFiles[i].length();</p> 
                <p>else</p> 
                <p>extnLocation = inputFiles[i].indexOf(".");</p> 
                <p>//extract The Base from the fileExtension</p> 
                <p>baseFileName = inputFiles[i].substring(0,extnLocation);</p> 
                <p>//create File String</p> 
                <p>inputFileName = loadDirectory + System.getProperty("file.separator") + inputFiles[i];</p> 
                <p>//decompress image</p> 
                <p>status = Simage.IMG_decompress_bitmap(inputFileName,0);</p> 
                <p>//only go forward if the image can be properly decompressed</p> 
                <p>//otherwise just move on to the next image.</p> 
                <p>if (status &gt;= 0)</p> 
                <p>{</p> 
                <p>//create outputFile String</p> 
                <p>outputFileName = saveDirectory + System.getProperty("file.separator") + baseFileName + extension;</p> 
                <p>//save new image</p> 
                <p>if (createThumbs == false)</p> 
                <p>Simage.IMG_save_bitmap(outputFileName,outputFileType);</p> 
                <p>else</p> 
                <p>makeThumbnail(outputFileName,outputFileType);</p> 
                <p>}</p> 
                <p>} //end for loop</p> 
                <p>} //end perform conversion</p> 
                <p>} //end convert class</p> 
                <p>Additional Resources<br> Learn more about Snowbound's solutions for image conversion and batch file conversion.</p>
