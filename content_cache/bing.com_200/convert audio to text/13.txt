                    <div class="post-header">

                      <div>
                        You are working as ETL Developer / SSIS Developer for Car Insurance company. They receive text files in their source folder. You need to create an SSIS Package that should convert these text files into Excel Files. The Excel File Name should be the same as Text file and Sheet Name should also be same as file name.
                      </div> 
                      <div> 
                       <br>
                      </div> 
                      <div>
                        Here are couple of samples files I am using for test
                      </div>

                       <a href="https://4.bp.blogspot.com/-B-KdtGq0ywc/VvKj7U2ZyGI/AAAAAAAAHDQ/wSonVL0IlUINULnsT7cdowV6N6pGfOdrQ/s1600/Capture.PNG" imageanchor="1" style="margin-left: 1em; margin-right: 1em;"><img border="0" height="273" src="https://4.bp.blogspot.com/-B-KdtGq0ywc/VvKj7U2ZyGI/AAAAAAAAHDQ/wSonVL0IlUINULnsT7cdowV6N6pGfOdrQ/s400/Capture.PNG" width="400"></a>

                      <h2 style="text-align: left;"> <b style="color: #333333; font-family: Arial, Tahoma, Helvetica, FreeSans, sans-serif; font-size: 13px; line-height: 18.2px;">Log File Information :&nbsp;</b><span style="background-color: yellow; color: #333333; font-family: &quot;arial&quot; , &quot;tahoma&quot; , &quot;helvetica&quot; , &quot;freesans&quot; , sans-serif; font-size: 13px; font-weight: normal; line-height: 18.2px;">In case your SSIS Package fail. Log file will be created in the same folder where your Excel File will be created.The name of log file will be same like your source file that Package was processing.</span></h2> 
                      <h2 style="text-align: left;"> <br>Solution: &nbsp;</h2> 
                      <div>
                        We care going to use Script Task in SSIS Package to convert Text/CSV/Tab delimited files to Excel Files. We are going to create variable in our SSIS Package such as FileExtension and FileDilimeter. By changing the values of these variable we can use our SSIS Package to handle different type of text files.
                      </div> 
                      <div> 
                       <br>
                      </div> 
                      <div> 
                       <br>
                      </div>

                      </div>

                      </div> 
                      <div> 
                       <br>
                      </div> 
                      <div> 
                       <b>DestinationFolderPath:</b> Folder path where you would like to create your Excel Files
                      </div> 
                      <div> 
                       <b>FileDelimiter :</b> Provide the delimiter such as comma (,), Pipe( | ) Whatever your files are using.
                      </div> 
                      <div> 
                       <b>FileExtension :</b> Provide the extension of files you would like to convert such .txt, .csv
                      </div> 
                      <div> 
                       <b>SourceFolderPath :</b> Source folder path where text files exists
                      </div>

                       <br>
                      </div>

            {

                <span class="rem" style="color: green;">//Declare Variables</span>
                <span class="kwrd" style="color: blue;">string</span> SourceFolderPath = Dts.Variables[<span class="str" style="color: #a31515;">"User::SourceFolderPath"</span>].Value.ToString();
                <span class="kwrd" style="color: blue;">string</span> DestinationFolderPath = Dts.Variables[<span class="str" style="color: #a31515;">"User::DestinationFolderPath"</span>].Value.ToString();
                <span class="kwrd" style="color: blue;">string</span> FileExtension= Dts.Variables[<span class="str" style="color: #a31515;">"User::FileExtension"</span>].Value.ToString();
                <span class="kwrd" style="color: blue;">string</span> FileDelimiter = Dts.Variables[<span class="str" style="color: #a31515;">"User::FileDelimiter"</span>].Value.ToString();
                <span class="kwrd" style="color: blue;">string</span> CreateTableStatement = <span class="str" style="color: #a31515;">""</span>;
                <span class="kwrd" style="color: blue;">string</span> ColumnList = <span class="str" style="color: #a31515;">""</span>;
                
                <span class="rem" style="color: green;">//Reading file names one by one</span>
                <span class="kwrd" style="color: blue;">string</span> SourceDirectory = SourceFolderPath;
                <span class="kwrd" style="color: blue;">string</span>[] fileEntries = Directory.GetFiles(SourceDirectory,<span class="str" style="color: #a31515;">"*"</span>+FileExtension);
                <span class="kwrd" style="color: blue;">foreach</span> (<span class="kwrd" style="color: blue;">string</span> fileName <span class="kwrd" style="color: blue;">in</span> fileEntries)
                {
                    <span class="rem" style="color: green;">// do something with fileName</span>
                    <span class="rem" style="color: green;">//MessageBox.Show(fileName);</span>
                    
                    <span class="rem" style="color: green;">//Read first line(Header) and prepare Create Statement for Excel Sheet</span>
                    System.IO.StreamReader file = <span class="kwrd" style="color: blue;">new</span> System.IO.StreamReader(fileName);
                    <span class="kwrd" style="color: blue;">string</span> filenameonly = (((fileName.Replace(SourceDirectory, <span class="str" style="color: #a31515;">""</span>)).Replace(FileExtension, <span class="str" style="color: #a31515;">""</span>)).Replace(<span class="str" style="color: #a31515;">"\\", "</span><span class="str" style="color: #a31515;">"));
                    CreateTableStatement = ("</span> Create Table [<span class="str" style="color: #a31515;">" + filenameonly + "</span>] ([<span class="str" style="color: #a31515;">" + file.ReadLine().Replace(FileDelimiter, "</span>] Text,[<span class="str" style="color: #a31515;">")) + "</span>] Text)<span class="str" style="color: #a31515;">";
                    file.Close();
                    //MessageBox.Show(CreateTableStatement.ToString());


                    //Construct ConnectionString for Excel
                    string connstring = "</span>Provider=Microsoft.ACE.OLEDB.12.0;<span class="str" style="color: #a31515;">" + "</span>Data Source=<span class="str" style="color: #a31515;">" + DestinationFolderPath + "</span>\\<span class="str" style="color: #a31515;">" + filenameonly
                        + "</span>;<span class="str" style="color: #a31515;">" + "</span>Extended Properties=\<span class="str" style="color: #a31515;">"Excel 12.0 Xml;HDR=YES;\""</span>;
                    OleDbConnection Excel_OLE_Con = <span class="kwrd" style="color: blue;">new</span> OleDbConnection();
                    OleDbCommand Excel_OLE_Cmd = <span class="kwrd" style="color: blue;">new</span> OleDbCommand();



                    <span class="rem" style="color: green;">//drop Excel file if exists</span>
                    File.Delete(DestinationFolderPath + <span class="str" style="color: #a31515;">"\\" + filenameonly + "</span>.xlsx<span class="str" style="color: #a31515;">");
                    Excel_OLE_Con.ConnectionString = connstring;
                    Excel_OLE_Con.Open();
                    Excel_OLE_Cmd.Connection = Excel_OLE_Con;

                    //Use OLE DB Connection and Create Excel Sheet
                    Excel_OLE_Cmd.CommandText = CreateTableStatement;
                    Excel_OLE_Cmd.ExecuteNonQuery();

                    //Writing Data of File to Excel Sheet in Excel File
                    int counter = 0;
                    string line;

                    System.IO.StreamReader SourceFile =
                    new System.IO.StreamReader(fileName);
                    while ((line = SourceFile.ReadLine()) != null)
                    {
                        if (counter == 0)
                        {
                            ColumnList = "</span>[<span class="str" style="color: #a31515;">" + line.Replace(FileDelimiter, "</span>],[<span class="str" style="color: #a31515;">") + "</span>]<span class="str" style="color: #a31515;">";
                            
                        }
                        else
                        {
                            string query = "</span>Insert into [<span class="str" style="color: #a31515;">" + filenameonly + "</span>] (<span class="str" style="color: #a31515;">" + ColumnList + "</span>) VALUES(<span class="str" style="color: #a31515;">'" + line.Replace(FileDelimiter, "'</span>,<span class="str" style="color: #a31515;">'") + "'</span>)<span class="str" style="color: #a31515;">";
                           // MessageBox.Show(query.ToString());
                            var command = query;
                            Excel_OLE_Cmd.CommandText = command;
                            Excel_OLE_Cmd.ExecuteNonQuery();

                        }
                        counter++;
                    }
                    Excel_OLE_Con.Close();
                    SourceFile.Close();
                    Dts.TaskResult = (int)ScriptResults.Success;
                }
            }
            catch (Exception exception)
            {


                // Create Log File for Errors
                using (StreamWriter sw = File.CreateText(Dts.Variables["</span>User::DestinationFolderPath<span class="str" style="color: #a31515;">"].Value.ToString() 
                    + "</span>\\<span class="str" style="color: #a31515;">" +"</span>ErrorLog_<span class="str" style="color: #a31515;">"+DateTime.Now.ToString("</span>yyyyMMddHHmmss<span class="str" style="color: #a31515;">")+"</span>.log"))
         
                {
                    sw.WriteLine(exception.ToString());
                    Dts.TaskResult = (<span class="kwrd" style="color: blue;">int</span>)ScriptResults.Failure;

                }
                               
            }</pre>
