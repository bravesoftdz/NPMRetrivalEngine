var FooterSystem = function() {
  this.responsiveMenus = '';
  this.responsiveFooterItems = $('#responsiveFooterMenu').find('> div');


  $('#email-sign-footer').submit(function () {
    var email = $(this).find('input[name=email]').val();
    $(this).find('input[name=emailConfirm]').val(email);
  });


  if ($(window).width() >= 768) { // desktop footer | Tay
    this.kickOffDesktop();
  } else {  // below 768 gets the mobile version | Tay
    this.kickOffMobile();
  }
  var that = this;
  that.responsiveFooterItems.not('.linkable').on('click', function () {
    var $responsiveMenus;
    var LI_HEIGHT = 58; // consider a constant int | Tay
    $(this).siblings().removeClass('tapped');
    $(this).addClass('tapped');
    if ($(window).width() >= 768) {
      $responsiveMenus = $('#responsiveMenus').find('ul');
    } else {
      $responsiveMenus = $(this).parent().find('ul');
    }
    var location = parseInt($(this).attr('data-location')); // allows the correct menu to be shown | Tay


// check if first of the month is a Monday, if so return the date, otherwise get to the Monday following the first of the month


    if ($responsiveMenus.eq(location).hasClass('activated')) {
      that._resetMenus($responsiveMenus);
      $(this).removeClass('tapped');

    } else {
      that._resetMenus($responsiveMenus);
      $responsiveMenus.eq(location).addClass('activated');
      var $li = $responsiveMenus.eq(location).find('li');
      $responsiveMenus.eq(location).css('height', $li.length * LI_HEIGHT);
    }
  });
};


FooterSystem.prototype._resetMenus = function(menus){
  menus.css('height', 0);
  menus.removeClass('activated');
};


FooterSystem.prototype.kickOffMobile = function(){
    this.responsiveMenus = $('#responsiveMenus').find('ul');
    var that = this;
    $.each(that.responsiveMenus, function(index, responsiveMenu){
        $(that.responsiveFooterItems[index]).after(responsiveMenu);
    });
    $(window).off('resize');
    $(window).resize(function(){
        if($(window).width() >= 768){
            that.kickOffDesktop();
        }
    });
};

FooterSystem.prototype.kickOffDesktop = function(){
    this.responsiveMenus = $('#responsiveFooterMenu').find('> ul');
    $('#responsiveMenus').append(this.responsiveMenus);
    $(window).off('resize');
    var that = this;
    $(window).resize(function(){
        if($(window).width() < 768){
            that.kickOffMobile();
        }
    });
};

        <h1><a href="https://www.swansonvitamins.com/blog/health-news-and-opinion/near-zero-calorie-foods">16 Near-Zero Calorie Snacks to Curb Your Hunger</a></h1>
       </div> 
       <div class="post-by-line">
         By 
        <span class="author-name"><a rel="author" href="https://www.swansonvitamins.com/blog/health-news-and-opinion">Ben H.</a></span> 
       </div> 
       <div class="post-body"> 
        <p><span style="line-height: 1.6;">There’s a myth in the diet world that just won’t die. It’s the myth of negative-caloric foods like celery or apples. </span></p> 
        <p><span style="line-height: 1.6;">This debunked theory states certain foods are so low in calories that your body actually burns more in the process of eating than it takes in. Unfortunately, the so-called “negative caloric effect” isn’t true, but that doesn’t mean there aren’t foods out there that deliver serious nutrition (and taste) in an ultra-low calorie package.</span></p> 
        <p>Just like <a href="http://www.swansonvitamins.com/blog/health-news-and-opinion/filling-foods">foods that keep you feeling fuller for longer periods of time</a> can help you maintain a healthy weight, foods that are naturally low in calories can help you get to that ideal weight by delivering needed nutrients without extra fat or sugar.</p> 
        <p>Here’s a list of near zero-calorie foods to add to your shopping list.</p> 
        <h2>Low-Calorie Vegetables</h2> 
        <p>The produce section should already be the place you spend most of your grocery dollars, and in this case it will also be the place to find the most bang for your caloric buck.</p> 
        <p>In many cases, vegetables are high in water content and fiber, yet low in calories. Plus, vegetables are incredibly versatile—you can eat them raw, steam them, grill, saute, bake, broil, etc.</p> 
        <p><span class="circle">1</span>&nbsp;<strong style="line-height: 20.8px;">Celery</strong>:&nbsp;two large celery stalks come in under 15 total calories<img alt="celery near zero calorie food" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/ec14d89b6c9eff14c1a33b1f86030b81/celery.png" style="float: right; width: 100px; height: 100px; margin: 2px 3px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">2</span><strong>Cucumber</strong>: half a cuke only adds 20 calories to your day<img alt="cucumber low calorie food" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/8c026b40c5b1dd85a207db71e5a8d6a6/cucumber.png" style="float: right; width: 100px; height: 100px; margin: 2px 3px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">3</span><strong>Tomato</strong>: a medium tomato is just 25 calories<img alt="tomato low calorie food at 25 calories" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/b832c66fa970e59162c1680e4712f188/tomato.png" style="float: right; width: 100px; height: 100px; margin: 2px 3px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">4</span><strong>Asparagus</strong>: not only is this veggie low in calories (27 per cup), but it’s also known for its natural detoxification power<img alt="asparagus low calorie food" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/ea960e451889f5dbc63e9c251f47ea3d/asparagus.png" style="float: right; width: 100px; height: 100px; margin: 2px 3px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">5</span><strong>Snap peas</strong>: satisfy your sweet tooth and enjoy half a cup of sugar snap peas for just 30 calories<img alt="healthy low cal snack food" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/2867a5fe0396a97f831aa84affd96f5e/snappeas.png" style="float: right; width: 100px; height: 100px; margin: 2px 3px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">6</span><strong>Carrot</strong>: skip the baby carrots and remind yourself what a real carrot is supposed to taste like (only 30 calories per)<img alt="carrots weight loss snack" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/7e1c316375606d99ce2c48143989b4f8/carrot.png" style="float: right; width: 100px; height: 100px;"></p> 
        <h2>&nbsp;</h2> 
        <p>&nbsp;</p> 
        <h2>Low-Calorie Fruits</h2> 
        <p>Fruits often get a bad reputation for being high in sugar. While certain fruits definitely contain more natural sugar than others (I’m looking at you, bananas), others are deceptively low in calories without sacrificing the sweetness.</p> 
        <p>Try these fruits to keep your calorie count low while also adding a bit of variety to your snack plate.</p> 
        <p><span class="circle">7</span><strong>Peaches</strong>: you can enjoy an entire fresh peach for only 37 calories<img alt="peaches are surprisingly low calorie for how sweet they taste" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/b7370be522a4ca3e19d3a1802a4889ea/peach_nectarine.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">8</span><strong>Grapefruit</strong>: this wonderfully tart fruit is big enough to split, offering two 39-calorie servings<img alt="low cal fruits include grapefruit" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/40b50d046aaea75c3cca3fc77cf4e060/grapefruit.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">9</span><strong>Strawberries</strong>: snack on a whole cup for just 50 calories<img alt="strawberries are a snack with just 50 calories" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/fb50748c6a58ac343a60a880f2121db6/strawberry.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">10</span><strong>Papaya</strong>: even though it's not as common as other fruits, this exotic fruit comes in at just 54 calories per cup<img alt="papaya calories" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/32c1fe04b878d0c048486a82ddf72a75/papaya.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">11</span><strong>Cantaloupe</strong>: sometimes called “melon” depending on where you live, ripe cantaloupe is a delight with just 56 calories per cup<img alt="cantaloupe is a fruit with low calories" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/72db942da3bb7efe6809eadfb82e5f1a/cantaloupe.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <h2>&nbsp;</h2> 
        <h2>Low-Calorie Snacks</h2> 
        <p>All of the above fresh produce obviously can be (and should be) considered snack food, but the following items are more traditional “snacks” you might find in the cupboard.</p> 
        <p><span class="circle" style="line-height: 1.6;">12</span><strong style="line-height: 1.6;">Popcorn</strong><span style="line-height: 1.6;">: not all popcorn is created equally… stick to air-popped popcorn (only 31 calories per cup) and add flavor with your favorite spices instead of melted butter<img alt="air-popped popcorn is very low in calories" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/42833ac840da7cfa0eef750a237d5628/popcorn.png" style="float: right; width: 100px; height: 100px;"></span></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">13</span><strong>Rice cakes</strong>: yes, I know, rice cakes… but when you’re craving crunchy with a hint of salt, these 35-calories snacks make a great afternoon craving killer<img alt="rice cakes near zero calorie snack" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/fb5fd9ad77c21a56d10a85233d8994e3/ricecakes.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">14</span><strong>Crunchy chickpeas</strong>: a little higher in calories than anything else on this list (120 per serving), crunchy chickpeas make up for it with high protein and fiber to keep you full<img alt="crunchy chickpeas healthy weight loss snack" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/ebe9c67b8a77212f91f2bb37b735034c/chickpeas.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">15</span><strong>Black Mission Figs</strong>: satisfy your sweet tooth and hunger for just 90 calories<img alt="figs calorie count" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/78ba66e787ef758e0b3de041d4f60442/figs.png" style="float: right; width: 100px; height: 100px;"></p> 
        <p>&nbsp;</p> 
        <p>&nbsp;</p> 
        <p><span class="circle">16</span><strong>Turkey jerky</strong>: at around 80 calories per ounce, turkey jerky makes a great afternoon snack to tide you over until supper time<img alt="turkey jerky lowest calorie jerky" src="//cdn.app.compendium.com/uploads/user/60e1f3de-9022-4822-b04f-6e65a57f5d12/7fe9e36b-9055-44e5-811d-905e8eff246f/Image/0613531040e425f6bb548da094e3229c/jerky.png" style="float: right; width: 100px; height: 100px;"></p>

        <p>Let’s be honest: the real secret here isn’t the list of foods… it’s portion control.</p> 
        <p>Lots of people get sucked into the “96% fat free” sticker on packaging, but then they go ahead and eat an entire sleeve of cookies. That’s not going to help anyone lose weight.</p>
