          <p> Despite Cadbury's pledge, many of its ordinary products already fall under the 250 calorie limit. A bar of Dairy Milk, for example, has 240 calories. While a Dairy Milk Caramel bar has only 215 calories. <br> Another major chocolate producer, Mars, was a step ahead of its competitor; more than two years ago it committed to cut the calorie content of its single-serving products to a maximum of 250 calories by the end of 2013. As a result it reduced the size of a Mars bar by seven grams to ensure there are 229 calories in a standard size Mars bar. <br> Other chocolate bars owned by Mars have also shrunk in size over the years. A Galaxy bar now contains only 228 calories. Snickers and Twix bars have a few extra calories; a standard bar of both has 250. <br> Nestle has reduced the calorie content in some of its chocolate bars, but a standard Yorkie bar still contains 299 calories. </p>

                return array.filter(function (el, index, arr) {
                    return index == arr.indexOf(el);
                });
            }

            var createCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        articleListToSkip[0] = currentArticleId;
                        $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
                        return true;
                    }
                });
            }

            var updateCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        updateCookie(currentArticleId, articleListToSkip);
                        return true;
                    }
                });
            }

            var updateCookie = function(articleIdToSkip, articleListToSkip) {
                articleListToSkip.push(articleIdToSkip);
                articleListToSkip = unique(articleListToSkip);
                $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
            }

            var chooseNextArticle = function (currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if ((article.id != currentArticleId) && ($.inArray(article.id, articleListToSkip) === -1)) {

                        //set article
                        settings.articleTitle = article.title;
                        settings.articleUrl = article.url;
                        settings.articleId = article.id;
						settings.articleImage = article.imgUrl;
                        return true;
                    }
                });
            }

            //public API
            return {
                settings: settings,
                init: init,
                display: display,
                openArticleContainer: openArticleContainer,
                closeArticleContainer: closeArticleContainer
            }
        })();

    };
