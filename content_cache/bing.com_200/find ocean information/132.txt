    function trimBoth(str) {
        return str.replace(/^\s\s*/, '').replace(/\s\s*$/, '');

    function isEmptySearch(s) {
        return ((s == null) || (s.length == 0));

    function isAWB(awb) {
        awb = trimBoth(awb);
        if (isNaN(parseInt(awb))) return false;
        if (awb.length != 10) return false;

        var check1 = awb.substring(9, 10);
        var check2 = awb.substring(0, 9) % 7;
        return check1 == check2;
    }

    function checkFormSearch( theForm, defaultValue ) {
        var trackingPage = '/content/in/en/express/tracking/tracking_results.html',
            searchField = $( '#q' );

        //get the query string
        var queryString = searchField.val();

        if ( !isEmptySearch( queryString ) && queryString != defaultValue ) {
            if ( isAWB( queryString ) ) {
                // transform field to match to tracking output
                searchField.attr( 'name', 'AWB');
                // remove unused fields for tracking
                $( 'input[name=without]', theForm ).remove();
                $( 'input[name=cqpages]', theForm ).remove();
                $( 'input[name=category]', theForm ).remove();
                // change form action
                theForm.action = trackingPage;
            }
            return true;
        }

        return false;
    }
