/**
 * sends a request to the fact-finder shopping cart information collector
 * (the url passed as first parameter) 
 * requires the getXmlHttpRequest from bc_pmc util.js 
 *
 */
function reportSCIC(query, artId, artPos, artOrigPos, pageNum, artSimi, artTitle, pageSize, origPageSize, eventName, artMasterId, domainName, indexID)
{
  if(eventName == null)
  {
    eventName = 'click';
  }
  //send information as utf-8 encoded post
  var httpRequest = getXmlHttpRequest();
  
  var params =  "";  
  params += "?query=" + encodeURIComponent(query);
  params += "&id=" + encodeURIComponent(artId);
  params += "&masterId=" + encodeURIComponent(artMasterId);
  params += "&pos=" + encodeURIComponent(artPos);
  params += "&origPos=" + encodeURIComponent(artOrigPos);
  params += "&page=" + encodeURIComponent(pageNum);
  params += "&simi=" + encodeURIComponent(artSimi);
  params += "&title=" + encodeURIComponent(artTitle);
  params += "&event=" + encodeURIComponent(eventName);
  params += "&pageSize=" + encodeURIComponent(pageSize);
  params += "&origPageSize=" + encodeURIComponent(origPageSize);
  params += "&DomainName=" + encodeURIComponent(domainName);
  params += "&IndexID=" + encodeURIComponent(indexID);
  

  httpRequest.open("GET", 'https://www.swarovski.com/is-bin/INTERSHOP.enfinity/WFS/SCO-Web_US-Site/en_US/-/USD/ViewFactfinder-SCIC' + params, false);
  httpRequest.send(null);
}

/**
 * function to report onclicks with less parameters
 * others are taken from search result
 */
function reportSCICClick(id, pos, origPos, artSimi, title, masterId)
{
  reportSCIC(
    '', 
    id, pos, origPos,
    '',
    artSimi, title, 
    '', '',
    'click', masterId, '', '');
}
