           <p>The <span class="input">using</span> statement provides a recommended alternative to the typical .Create, .Save, .Close sequence. It ensures that the <span class="input">Dispose</span> method (internal method that is used by the Open XML SDK to clean up resources) is automatically called when the closing brace is reached. The block that follows the <span class="input">using</span> statement establishes a scope for the object that is created or named in the <span class="input">using</span> statement, in this case <em>wordprocessingDocument</em>. Because the <span class="input">WordprocessingDocument</span> class in the Open XML SDK automatically saves and closes the object as part of its <span class="input">System.IDisposable</span> implementation, and because <span class="input">Dispose</span> is automatically called when you exit the block, you do not have to explicitly call <span class="input">Save</span> and <span class="input">Close</span>?as long as you use <span class="input">using</span>.</p>

<span style="color:Blue;">public</span> <span style="color:Blue;">static</span> <span style="color:Blue;">void</span> InsertAPicture(<span style="color:Blue;">string</span> document, <span style="color:Blue;">string</span> fileName)
{
    <span style="color:Blue;">using</span> (WordprocessingDocument wordprocessingDocument = 
        WordprocessingDocument.Open(document, <span style="color:Blue;">true</span>))
    {
        MainDocumentPart mainPart = wordprocessingDocument.MainDocumentPart;

        ImagePart imagePart = mainPart.AddImagePart(ImagePartType.Jpeg);

        <span style="color:Blue;">using</span> (FileStream stream = <span style="color:Blue;">new</span> FileStream(fileName, FileMode.Open))
        {
            imagePart.FeedData(stream);
        }

        AddImageToBody(wordprocessingDocument, mainPart.GetIdOfPart(imagePart));
    }
}

<span style="color:Blue;">private</span> <span style="color:Blue;">static</span> <span style="color:Blue;">void</span> AddImageToBody(WordprocessingDocument wordDoc, <span style="color:Blue;">string</span> relationshipId)

                 <span style="color:Blue;">new</span> DW.EffectExtent() { LeftEdge = 0L, TopEdge = 0L,
