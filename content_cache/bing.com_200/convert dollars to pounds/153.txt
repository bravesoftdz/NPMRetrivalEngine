     <script type="text/javascript">
            (function(){

            var converterAPI = 'http://rates.shiftforex.com/v1/convert.php';
            var converterForm;
            var converterOutput;
            var converterLoading;
            var converterRequest;
            var populatingTable = false;

            $(document).ready(function() {
              populateCurrencyTable();
              converterForm = $('form#CurrencyCalcForm');
              //converterForm.submit(currencyFormSubmit);

              $("form #updateCurrency").on("click",function(){currencyFormSubmit();});

              $("#updateCurrencyTable").on("click",function(){populateCurrencyTable();});

              converterForm.find('#two_column_main_content_InputValue').val("1");
              document.getElementById('two_column_main_content_lstFromCurrency').selectedIndex = 0;
              document.getElementById('two_column_main_content_lstToCurrency').selectedIndex = 1;
              converterOutput = converterForm.find('#converterDisplay');
              converterLoading = converterForm.find('#converterLoading');
              currencyFormSubmit();
            });

            function currencyFormSubmit(event) {
              displayLoading();

              convertCurrency()
                .done(updateCurrencyDisplay);

              return false;
              //event.preventDefault();
            }

            function convertCurrency() {
              converterRequest = {
                amount: converterForm.find('#two_column_main_content_InputValue').val(),
                baseCurrency: converterForm.find('#two_column_main_content_lstFromCurrency').val(),
                quoteCurrency: converterForm.find('#two_column_main_content_lstToCurrency').val()
              };
              return $.get(converterAPI + '?amount=' + converterRequest.amount + '&baseCurrency=' + converterRequest.baseCurrency + '&quoteCurrency=' + converterRequest.quoteCurrency);
            }

            function updateCurrencyDisplay(convertedCurrency) {
              if ((typeof convertedCurrency === 'object')
                  && (typeof convertedCurrency.error === 'undefined')
                  && (convertedCurrency.baseAmount == converterRequest.amount)
                  && (convertedCurrency.baseCurrency == converterRequest.baseCurrency)
                  && (convertedCurrency.quoteCurrency == converterRequest.quoteCurrency)) {
                var baseDisplay = formatCurrency(convertedCurrency.baseAmount)
                                  + ' '
                                  + convertedCurrency.baseCurrency;

                var quoteDisplay = formatCurrency(convertedCurrency.quoteAmount)
                                   + ' '
                                   + convertedCurrency.quoteCurrency;

	            document.getElementById("fromCurrency").innerHTML = baseDisplay;
                //converterForm.find('.converterInput').html(baseDisplay);
                document.getElementById("converterDisplay").innerHTML = quoteDisplay;
              }

              hideLoading();
            }

            function formatCurrency(amount) {
              if (Math.floor(amount * 100) / 100 === Math.floor(amount)) {
                return Math.floor(amount);
              } else {
                return Math.floor(amount * 100) / 100;
              }
            }

            function displayLoading() {
              //converterOutput.css("display","none");
              converterLoading.css("display","inline");
            }

            function hideLoading() {
              converterLoading.css("display","none");
              //converterOutput.css("display","inline");
            }

            function populateCurrencyTable()
            {
	            var baseCurrencies = ['AUD','GBP','CAD','EUR','JPY','CHF','USD'];

	            if(populatingTable)
		            return;
            	
	            function populateRow(currencyIndex)
	            {
		            document.getElementById(baseCurrencies[currencyIndex]+"label").innerHTML = "<img src='http://www.nasdaq.com/images/converter-loading.gif' />";

		            var quoteCurrencies = ['AUD','GBP','CAD','EUR','JPY','CHF','USD'];
            	
		            quoteCurrencies.splice(currencyIndex,1);
            		
		            $.when($.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[0]), 
		            $.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[1]),
		            $.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[2]),
		            $.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[3]),
		            $.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[4]),
		            $.ajax(converterAPI+'?amount=1&baseCurrency='+baseCurrencies[currencyIndex]+'&quoteCurrency='+quoteCurrencies[5]))
			            .done(function(quoteCurrency0, quoteCurrency1, quoteCurrency2, quoteCurrency3, quoteCurrency4, quoteCurrency5){
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[0]).innerHTML = formatCurrency($.parseJSON(quoteCurrency0[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[1]).innerHTML = formatCurrency($.parseJSON(quoteCurrency1[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[2]).innerHTML = formatCurrency($.parseJSON(quoteCurrency2[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[3]).innerHTML = formatCurrency($.parseJSON(quoteCurrency3[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[4]).innerHTML = formatCurrency($.parseJSON(quoteCurrency4[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+quoteCurrencies[5]).innerHTML = formatCurrency($.parseJSON(quoteCurrency5[2].responseText).quoteAmount);
				            document.getElementById(baseCurrencies[currencyIndex]+"label").innerHTML = baseCurrencies[currencyIndex];
			            if(currencyIndex<6)
				            setTimeout(function(){populateRow(currencyIndex + 1);},0);
			            else
				            populatingTable = false;
		            });
	            }
	            populateRow(0);
            }
            })();
