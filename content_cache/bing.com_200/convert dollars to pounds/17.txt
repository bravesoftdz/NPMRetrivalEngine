				if(nSourceYear>=1264 && nSourceYear<=1983)

						nPresentValue=Math.round(nPresentValue*Math.pow(10,2))/Math.pow(10,2);

						PresentValueString = String(nPresentValue);
						nPresentValueLen = PresentValueString.indexOf(".");

						var output="";
						for(var i=0; i<nPresentValueLen%3; i++)
						{
							output = output + PresentValueString.charAt(i);
						}
						for(var i=nPresentValueLen%3; i<nPresentValueLen; i++)
						{
							if(i>0 && (i-nPresentValueLen)%3==0)
							{
								output = output + ",";

						}
						for(var i=nPresentValueLen; i<PresentValueString.length; i++)
						{
