import java.awt.*;
import java.awt.event.*;
import java.text.DecimalFormat;
import javax.swing.*;


public class Hello extends JApplet implements ActionListener
{
// Declare the components

Panel nPanel, sPanel, cPanel;
Label displayLabel, exchangeLabel, amountLabel, convertedAmountLabel ;
TextField exchangeField, amountField, convertedamountField;
Button poundsButton , eurosButton, clearButton;
private double euroexchangeRate, poundsexchangeRate, euroconvertedAmount, poundsconvertedAmount, inputAmount;


// Declare the global variables

double displayTotal , newValue ;
String displayString , inputString ;



/**
*
*/
public void init()
{
// Initialise the components and add them to the screen




nPanel = new Panel(); // south Panel will just hold one Label
nPanel.setBackground(Color.yellow); // give it color so you can see it
displayLabel = new Label("Currency Converter");
//Set Layout and Add Label
nPanel.setLayout (new FlowLayout(FlowLayout.CENTER));
nPanel.add (displayLabel);



cPanel = new Panel();
cPanel.setLayout(new GridLayout(3,2));
exchangeLabel = new Label("Rates of Exchange from $ to ");
cPanel.add( exchangeLabel );
exchangeField = new TextField( 10 );
exchangeField.setEditable( false );
cPanel.add( exchangeField );

amountLabel = new Label("Amount to be Exchanged");
cPanel.add( amountLabel );
amountField = new TextField( 10 );
cPanel.add( amountField );

convertedAmountLabel = new Label("Converted Amount");
cPanel.add( convertedAmountLabel );
convertedamountField = new TextField( 10 );
convertedamountField.setEditable( false );
cPanel.add( convertedamountField );


// Add each Button and its actionListener

sPanel = new Panel(); // north panel will hold three button
sPanel.setBackground(Color.blue); // give it color so you can see it

poundsButton = new Button ( "Pounds" ) ;
add ( poundsButton ) ;
poundsButton.addActionListener ( this ) ;
getContentPane().add(cPanel);


eurosButton = new Button ( "Euros" ) ;
add ( eurosButton ) ;
eurosButton.addActionListener ( this ) ;
getContentPane().add(cPanel);


clearButton = new Button ( "Clear" ) ;
add ( clearButton ) ;
clearButton.addActionListener ( this ) ;
getContentPane().add(cPanel);


sPanel.setLayout (new FlowLayout(FlowLayout.CENTER));
sPanel.add(poundsButton);
sPanel.add(eurosButton);
sPanel.add(clearButton);




//set Layout for the Applet itself and add the panels
this.setLayout (new BorderLayout());
add(nPanel, BorderLayout.NORTH);
add(cPanel, BorderLayout.NORTH);
add(sPanel, BorderLayout.SOUTH);

}

public void actionPerformed( ActionEvent event )
{

inputAmount = Double.parseDouble( amountField.getText() );
calculateValues();
displayValues();
}

/** calculate the values */
public void calculateValues()
{
euroexchangeRate = 0.85;
poundsexchangeRate = 1.18;
euroconvertedAmount = euroexchangeRate * inputAmount;
poundsconvertedAmount = poundsexchangeRate * inputAmount;


}

/** display the values */
public void displayValues()
{
DecimalFormat df = new DecimalFormat( "0.00" );

exchangeField.setText( df.format(euroexchangeRate) + "" );
convertedamountField.setText( df.format(euroconvertedAmount) + "" );

exchangeField.setText( df.format(poundsexchangeRate) + "" );
convertedamountField.setText( df.format(poundsconvertedAmount) + "" );



}




}
