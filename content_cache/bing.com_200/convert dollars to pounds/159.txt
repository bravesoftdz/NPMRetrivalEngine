	function calcIndex(value) {
		var range = new Array();
		range[1] = 0;
		range[2] = 501;
		range[3] = 1001;
		range[4] = 3001;
		range[5] = 10001;
		range[6] = 40001;
		range[7] = 100001;
		idx = -1;
		var i = 7;
		for (i=7;i>=0;i-=1){
			if (range[i]<=value){
				idx = i;
				break;
			}
		}

		return i;
	}
