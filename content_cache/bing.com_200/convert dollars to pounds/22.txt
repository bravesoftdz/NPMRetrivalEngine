        if (navigator.userAgent.match(/(iPad|iPhone);.*CPU.*OS 7_\d/i)) {
            $('.header-container').addClass('is-ios7');
            $('.header-slide').addClass('is-ios7');
        }

        $( '.main-nav li:has(.subnav), .main-nav li:has(ul)' ).doubleTapToGo();
        $('.has-subnav a[id*="navigation-menuitem"]').off().on('click')
    };

    // Hide contact if no href attribute
    var hideContact = function() {

        var resizeTimer;

        /// Hide when mobile nav is triggered
        
        if ( !$('li.contact a').attr('href') ) {
            $('li.contact').css('display', 'none');
        };

        /// Unhide on resize
        $(window).on( 'resize', function(e) {

            clearTimeout(resizeTimer);
            
            resizeTimer = setTimeout(function() {
                if ( $(window).width() >= 1040) {

                    $('li.contact').css('display', '');
                }
            
            }, 250 );
        });
    };

    this.mobileNav = function() {
        // unBindTouch();
        $('.nav-level, .subnav').css({
            'display': '',
            'height': ''
        });
        hideContact();
        showHideMenu();
        showHideSubNav();
        handleBackButtons();
        showHideSubSubNav();
        countrySwitcher();
    };

    this.init = function() {

        // Detecting IE
        var oldIE;
        if ($('html').is('.lt-ie9')) {
            oldIE = true;
        }

        // Add responsive functionality for modern browsers
        if (!oldIE) {
            var mq = window.matchMedia('all and (min-width: 65em)');
            if (mq.matches) {
                initDesktop();
            } else {
                _self.mobileNav();
            }
            mq.addListener(function (changed) {
                if (changed.matches) {
                    initDesktop();
                } else {
                    _self.mobileNav();
                }
            });
        }
    };

};

var mobileNav = new MobileNav();

mobileNav.init();
