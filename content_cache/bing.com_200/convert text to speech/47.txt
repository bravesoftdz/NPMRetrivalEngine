        function getParamsCallback(newSearch) {
            if (newSearch === undefined) {
                newSearch = true;
                $('#devStatusRefinement').val('');
                $('#tagRefinement').val('');
            }

            var originalSearchText = 'speech';
            var searchText = encodeURIComponent($('#search_directory_bar').val());
            var stableOnly = $('#stableOnly').is(':checked');
            var licenses = $('#selectedLicenses').val();
            var tags = $('#tagRefinement').val();
            var advancedSearch = $('#usingAdvancedSearch').val();
            var developmentStatusParam = '';
            var sort = (newSearch && searchText.length > 0 && searchText != originalSearchText && searchText != ' ') ? 'Relevance' : $('#sortBy').val();
            if (advancedSearch == 'True') {
                developmentStatusParam = $('#advancedDevelopmentStatus').val();
            } else {
                developmentStatusParam = $('#devStatusRefinement').val();
            }

            return ((advancedSearch != 'True' && stableOnly) ? '&stableOnly=true' : '') +
                   ((sort != undefined && sort.length > 0) ? '&sortBy=' + sort : '') +
                   ((developmentStatusParam != null && developmentStatusParam.length > 0) ? '&devStatus=' + developmentStatusParam : '') +
                   ((tags != undefined && tags.length > 0) ? '&tagName=' + tags : '') +
                   ((advancedSearch == 'True') ? '&advSearch=true' : '') +
                   ((licenses.length > 0) ? '&licenses=' + licenses : '') +
                   (newSearch ? '' : '&refinedSearch=true');
        }
        function getSearchParams(newSearch) {
            var searchBoxVal;
            if (newSearch) {
                var searchBoxVal = encodeURIComponent($('#search_directory_bar').val());
                if (searchBoxVal == ' ') searchBoxVal = '';
                $('#previousSearchValue').val(searchBoxVal);
            }
            else {
                searchBoxVal = encodeURIComponent($('#previousSearchValue').val());
            }

            return '?query=' + searchBoxVal + getParamsCallback(newSearch);         
        }

        function refineTags(tag, shouldAdd) {
            var current = $('#tagRefinement').val();
            if (shouldAdd) {
                current += (current.length == 0) ? '%2c' + tag : tag;
                current += '%2c';
            } else {
                var regEx = new RegExp("\%2c" + tag + "\%2c");
                current = current.replace(regEx, '%2c');
                if (current == '%2c') current = '';
            }
            $('#tagRefinement').val(current);
            search(false);
        }
        
        function toggleSearch() {
            var newText = 'Simple Search';

        function selectLicenses() {
            var licenses = '';
            var fullList = '';
            var licensesTitle = '';
            $('#licenseSelection :checkbox:checked').each(function() {
                if (licenses.length > 0) { licenses += '%2c'; fullList += ', '; licensesTitle += ', '; }
                licenses += jQuery.trim($(this).val());
                fullList += '<span title=\"' + jQuery.trim($('#licenseLabel' + $(this).val()).text()) + '\">' + jQuery.trim($(this).val()) + '</span>';
                licensesTitle += jQuery.trim($('#licenseLabel' + $(this).val().replace('.', '')).text());
            });
            $('#selectedLicenses').val(licenses + "|" + licensesTitle);
            $('#licensesDisplay').empty().append(((fullList.length > 0) ? fullList : 'any') + ' ');
            OpenDialog('#licenseSelection', false);
        }
        function toggleAllLicenses() {
            var checked = ($('#licenseSelection :checkbox:checked').length == $('#licenseSelection :checkbox').length);
            if (checked === undefined)
                checked = false;
            checked = !(checked.toString().toLowerCase() == 'true');
            $('#licenseSelection :checkbox').each(function() {$(this).attr('checked', checked);$(this).click();$(this).attr('checked', checked);});
            $('#toggleAllLicensesLink').text(checked ? 'Unselect All' : 'Select All');

        }
        function preserveCheckedLicenses() {
            if ($('#selectedLicenses').val().length == 0 || $('#selectedLicenses').val() == "|") {
                $('#licenseSelection :checkbox').each(function() { $(this).attr('checked', false) });

            }
            else {
                $('#licenseSelection :checkbox').each(function() {
                    $(this).attr('checked', IsThisLicenseSelected($(this).val()));
                });
            }

            toggleLicenseCheckbox();
        }

        function IsThisLicenseSelected(code) {

            if ($('#selectedLicenses').val().length > 0 && $('#selectedLicenses').val() != "|") {
                var licenseCodes = $('#selectedLicenses').val().split("|")[0];
                licenseCodes = licenseCodes.split("%2c");
                for (x = 0; x < licenseCodes.length; x++) {
                    if (licenseCodes[x] == code) {
                        return true;
                    }
                }
                return false;
            }
            return false;
        }
