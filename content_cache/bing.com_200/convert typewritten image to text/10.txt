       <p> <a href="http://www.dailycoding.com/Utils/Converter/ImageToBase64.aspx">Online Converter Image to Base64</a> </p> 
       <p>This article will help you to learn how we can convert an image into a <a href="http://en.wikipedia.org/wiki/Base64" target="_blank">base64</a> string and base64 string back to image.</p> 
       <h3>Image to Base64 String</h3> 
       <pre class="csharpcode">
<span class="kwrd">public string</span> ImageToBase64(<span class="cls">Image</span> image, 
  System.Drawing.Imaging.<span class="cls">ImageFormat</span> format)
{
  <span class="kwrd">using</span> (<span class="kwrd">MemoryStream</span> ms = new <span class="kwrd">MemoryStream</span>())
  {
    <span class="rem">// Convert Image to byte[]</span>
    image.Save(ms, format);
    <span class="kwrd">byte</span>[] imageBytes = ms.ToArray();

    <span class="rem">// Convert byte[] to Base64 String</span>
    <span class="kwrd">string</span> base64String = Convert.ToBase64String(imageBytes);
    <span class="kwrd">return</span> base64String;
  }
}</pre> 
       <!--ads--> 
       <h3>Base64 String to Image</h3> 
       <pre class="csharpcode">
<span class="kwrd">public</span> <span class="cls">Image</span> Base64ToImage(<span class="kwrd">string</span> base64String)
{
  <span class="rem">// Convert Base64 String to byte[]</span>
  <span class="kwrd">byte</span>[] imageBytes = <span class="cls">Convert</span>.FromBase64String(base64String);
  <span class="cls">MemoryStream</span> ms = new <span class="cls">MemoryStream</span>(imageBytes, 0, 
    imageBytes.Length);

  <span class="rem">// Convert byte[] to Image</span>
  ms.Write(imageBytes, 0, imageBytes.Length);
  <span class="cls">Image</span> image = <span class="cls">Image</span>.FromStream(ms, true);
  <span class="kwrd">return</span> image;
}</pre>
