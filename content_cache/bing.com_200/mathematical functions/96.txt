<font color="#3f7f5f">/*************************************************************************

<font color="#7f0055"><b>import</b></font> java.awt.*;
<font color="#7f0055"><b>import</b></font> edu.hws.jcm.data.*;
<font color="#7f0055"><b>import</b></font> edu.hws.jcm.draw.*;
<font color="#7f0055"><b>import</b></font> edu.hws.jcm.awt.*;

<font color="#7f0055"><b>public</b></font> <font color="#7f0055"><b>class</b></font> GraphApplet3 <font color="#7f0055"><b>extends</b></font> java.applet.Applet {
      <font color="#7f0055"><b>public</b></font> <font color="#7f0055"><b>static</b></font> <font color="#7f0055"><b>void</b></font> main(String[] a){
         javax.swing.JFrame f = <font color="#7f0055"><b>new</b></font> javax.swing.JFrame();
         java.applet.Applet app = <font color="#7f0055"><b>new</b></font> GraphApplet3();
         app.init();
         
         f.getContentPane().add (app);

         f.pack();
         f.setSize (<font color="#7f0055"><b>new</b></font> Dimension (500, 500));
         f.setVisible(true);
      }  
   <font color="#7f0055"><b>private</b></font> DisplayCanvas canvas;
   
   <font color="#7f0055"><b>public</b></font> <font color="#7f0055"><b>void</b></font> stop() {
      canvas.releaseResources();
   }
   
   <font color="#7f0055"><b>public</b></font> <font color="#7f0055"><b>void</b></font> init() {
   
      Parser parser = <font color="#7f0055"><b>new</b></font> Parser();
      Variable x = <font color="#7f0055"><b>new</b></font> Variable(<font color="#2a00ff">"x"</font>);
      parser.add(x);

      canvas = <font color="#7f0055"><b>new</b></font> DisplayCanvas();
      canvas.setHandleMouseZooms(true);
      canvas.add(<font color="#7f0055"><b>new</b></font> Panner());
      
      CoordinateRect coords = canvas.getCoordinateRect();
      
      LimitControlPanel limits =
           <font color="#7f0055"><b>new</b></font> LimitControlPanel( LimitControlPanel.SET_LIMITS | LimitControlPanel.RESTORE, false);
      limits.addCoords(canvas);
      
      ExpressionInput input = <font color="#7f0055"><b>new</b></font> ExpressionInput(<font color="#2a00ff">"sin(x)+2*cos(3*x)"</font>, parser);
      Function func = input.getFunction(x);
   
      Graph1D graph = <font color="#7f0055"><b>new</b></font> Graph1D(func);
      
      VariableInput xInput = <font color="#7f0055"><b>new</b></font> VariableInput();
      VariableSlider xSlider = <font color="#7f0055"><b>new</b></font> VariableSlider( coords.getValueObject(CoordinateRect.XMIN), 
                                                      coords.getValueObject(CoordinateRect.XMAX) );
      
      Value yValue = <font color="#7f0055"><b>new</b></font> ValueMath(func,xSlider); <font color="#3f7f5f">// A Value object to represent the y-coord of the point.
</font>      
         <font color="#3f7f5f">// Instead of using a crosshair to mark a point on the graph, it is marked
</font>         <font color="#3f7f5f">//   with two gray lines and a small magenta oval.  These geometric objects
</font>         <font color="#3f7f5f">//   are represented as objects belonging to the class DrawGeometric,
</font>         <font color="#3f7f5f">//   which makes it possible to draw a variety of geometric figures on a
</font>         <font color="#3f7f5f">//   DisplayCanvas.
</font>      DrawGeometric vLine = <font color="#7f0055"><b>new</b></font> DrawGeometric(DrawGeometric.LINE_ABSOLUTE,xSlider,<font color="#7f0055"><b>new</b></font> Constant(0),xSlider,yValue);
      DrawGeometric hLine = <font color="#7f0055"><b>new</b></font> DrawGeometric(DrawGeometric.LINE_ABSOLUTE,<font color="#7f0055"><b>new</b></font> Constant(0),yValue,xSlider,yValue);
      DrawGeometric point = <font color="#7f0055"><b>new</b></font> DrawGeometric(DrawGeometric.OVAL_CENTERED,xSlider,yValue,3,3);
      vLine.setColor(Color.lightGray);
      hLine.setColor(Color.lightGray);
      point.setColor(Color.magenta);
      point.setFillColor(Color.magenta);

      
      DrawString info = <font color="#7f0055"><b>new</b></font> DrawString(<font color="#2a00ff">"x = #\nf(x) = #"</font>, DrawString.TOP_LEFT,
                                                             <font color="#7f0055"><b>new</b></font> Value[] { xSlider, yValue });
      info.setFont( <font color="#7f0055"><b>new</b></font> Font(<font color="#2a00ff">"SansSerif"</font>,Font.BOLD,12) );
      info.setColor( <font color="#7f0055"><b>new</b></font> Color(0,100,0) );
      info.setOffset(10);

      ComputeButton graphIt = <font color="#7f0055"><b>new</b></font> ComputeButton(<font color="#2a00ff">"Graph It!"</font>);
      
      setLayout(<font color="#7f0055"><b>new</b></font> BorderLayout(3,3));
      setBackground(Color.lightGray);      

          <font color="#3f7f5f">// In this version of the applet, I have built the interface from
</font>          <font color="#3f7f5f">//    regular Panels instead of JCMPanels.  This puts responsibility
</font>          <font color="#3f7f5f">//    for a lot more setup in the hands of the programmer.  The gain
</font>          <font color="#3f7f5f">//    is in efficiency.  Here, my object is to avoid recomputing the
</font>          <font color="#3f7f5f">//    graph just because the user adjusts the slider.  To do this,
</font>          <font color="#3f7f5f">//    I have to use two controllers, which listen for different user
</font>          <font color="#3f7f5f">//    actions.  (Of course, computers are so fast now that the extra
</font>          <font color="#3f7f5f">//    computation probably doesn't add a perceptible delay.  In this 
</font>          <font color="#3f7f5f">//    case, the extra design work is probably not worth the trouble.)
</font>      Panel top = <font color="#7f0055"><b>new</b></font> Panel();
      top.setLayout(<font color="#7f0055"><b>new</b></font> BorderLayout(3,3));
      Panel bottom = <font color="#7f0055"><b>new</b></font> Panel();
      bottom.setLayout(<font color="#7f0055"><b>new</b></font> BorderLayout(3,3));
      add(canvas, BorderLayout.CENTER);  <font color="#3f7f5f">// Add components directly to the applet.
</font>      add(limits, BorderLayout.EAST);
      add(bottom, BorderLayout.SOUTH);
      add(top, BorderLayout.NORTH);

      top.add(input, BorderLayout.CENTER);
      top.add(<font color="#7f0055"><b>new</b></font> Label(<font color="#2a00ff">" f(x) = "</font>), BorderLayout.WEST);
      top.add(graphIt, BorderLayout.EAST);
      
      bottom.add(xSlider, BorderLayout.CENTER);
      bottom.add(xInput, BorderLayout.EAST);
      bottom.add(<font color="#7f0055"><b>new</b></font> Label(<font color="#2a00ff">"  x = "</font>), BorderLayout.WEST);

      
      canvas.add( <font color="#7f0055"><b>new</b></font> Axes() );
      canvas.add( hLine );
      canvas.add( vLine );
      canvas.add( point );
      canvas.add( graph );
      canvas.add( info );
      canvas.add( <font color="#7f0055"><b>new</b></font> DrawBorder(Color.darkGray, 2) );
      
      Controller cc = <font color="#7f0055"><b>new</b></font> Controller();  <font color="#3f7f5f">// This controller will listen for changes
</font>      xInput.setOnUserAction(cc);        <font color="#3f7f5f">//   In the VariableSlider or VariableInput,
</font>      xSlider.setOnUserAction(cc);       <font color="#3f7f5f">//   As well as in the limits on the coordinates.
</font>      coords.setOnChange(cc);

      cc.add( <font color="#7f0055"><b>new</b></font> Tie(xSlider,xInput) ); <font color="#3f7f5f">// Ties the values of the slider and VariableInput.
</font>      
      cc.add( hLine );    <font color="#3f7f5f">// I have to tell the controller which objects need to be recomputed
</font>      cc.add( vLine );    <font color="#3f7f5f">//    when it sees some kind of change.  This includes all the 
</font>      cc.add( point );    <font color="#3f7f5f">//    objects that depend on the x-coordinate.  Note that is ALSO
</font>      cc.add( info );     <font color="#3f7f5f">//    includes xInput and xSlider, which need to be checked for 
</font>      cc.add( xInput );   <font color="#3f7f5f">//    changes in their values.  The value associated with a
</font>      cc.add( xSlider );  <font color="#3f7f5f">//    VariableSlider or VariableInput doesn't actually change
</font>                          <font color="#3f7f5f">//    until a Controller checks it.  (All this is the part of the
</font>                          <font color="#3f7f5f">//    setup that is done automatically when you build your
</font>                          <font color="#3f7f5f">//    interface from JCMPanels.)
</font>      
      Controller gc = <font color="#7f0055"><b>new</b></font> Controller();   <font color="#3f7f5f">// This controller will listen for changes
</font>      input.setOnUserAction(gc);          <font color="#3f7f5f">//   in the function definition.
</font>      graphIt.setOnUserAction(gc);
      
      gc.add(input);   <font color="#3f7f5f">// I have to add the ExpressionInput to a Controller, since the
</font>                       <font color="#3f7f5f">//   function doesn't actually change unless the ExpressionInput
</font>                       <font color="#3f7f5f">//   is checked by a Controller.
</font>      gc.add(graph);   <font color="#3f7f5f">// The graph needs to be recomputed when the function changes.
</font>      gc.add(cc);      <font color="#3f7f5f">// You can add one Controller to another.  Here, gc will call
</font>                       <font color="#3f7f5f">//   on cc to do all its checks and computations, in addition to.
</font>                       <font color="#3f7f5f">//   recomputing the graph.
</font>      
      gc.setErrorReporter(canvas);      <font color="#3f7f5f">// Set error reporters for the Controller.
</font>                                        <font color="#3f7f5f">//   This error reporter is also used by
</font>                                        <font color="#3f7f5f">//   cc, which has been added as a subcontroller
</font>                                        <font color="#3f7f5f">//   to gc.  So, it's not necessary to set a separate
</font>                                        <font color="#3f7f5f">//   error reporter for cc
</font>
      limits.setErrorReporter(canvas);  <font color="#3f7f5f">// Error reporter for the LimitControlPanel.
</font>      
   } <font color="#3f7f5f">// end init()
</font>
} <font color="#3f7f5f">// end class SimpleGraph3
</font>
