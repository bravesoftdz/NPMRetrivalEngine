   <script type="text/javascript">
$(function()
{
    var TITLE = 'Account Login';
    $('#txt-login').focus();

    $('#btn-forgotpwd').click(function(e)
    {
        e.preventDefault();
        window.open('http://smsonlinegh.com/index.php?action=forgotpwd', '_self');
    });
	
    $('#btn-login').click(function(e)
    {
        e.preventDefault();

        var login = $('#txt-login').val();
        var pwd   = $('#txt-pwd').val();

        if (login == null || login == '')
        {
            $.messager.alert(TITLE, 'Please enter your account login.', 'error');
            $('#txt-login').focus();

            return;
        }

        if (pwd == null || pwd == '')
        {
            $.messager.alert(TITLE, 'Please enter your account password.', 'error');
            $('#txt-pwd').focus();

            return;
        }

        doLogin(login,pwd,$('#chk-rmb')[0].checked ? 1:0);
    });
	
    function doLogin(login, pwd, rmb)
    {
        $.ajax(
        {
            type: "POST",
            url:  "client/login.php",
            data: {login: login, pwd: pwd, rmb:rmb},
            dataType: "json",

            beforeSend: function()
            {
                $.pgsOverlay.show('Authenticating...');
            },

            success: function(jd)
            {   
                if (jd.response_code == 0x0000)
                {
                    $.pgsOverlay.close();
                    $.messager.alert(TITLE, jd.response_msg, 'error');
                    return;
                }

                else if (jd.response_code == 0x0001)
                {
                    // authentication was successful.

                    $.pgsOverlay.setNotice("Login successful, please wait...");

                    if (jd.reqpage != null && jd.reqpage != "")
                        window.open(jd.reqpage, '_self');
                    else
                      window.open("http://smsonlinegh.com/client/sendsms.php", "_self");
                }

                else if (jd.response_code == 0x0004)
                {
                    // the account has not been activated. we need the account name.               
                    $.pgsOverlay.setNotice("Account is inactive, please wait...");
                    window.open('http://smsonlinegh.com/register.php?action=activate'+
                        '&l='+encodeURIComponent(jd.login), '_self');
                }

                else if (jd.response_code == 0x0006 /*-2 */)
                {
                    $.pgsOverlay.setNotice('Account is inactive, please wait...');
                    window.open("http://smsonlinegh.com/register.php?status=smsfailed&l=" + encodeURIComponent(jd.login), "_self");
                }

                else if (jd.response_code == 0x0005)
                {
                    // the account was not activated and an activation code was just sent.
                    $.pgsOverlay.setNotice('Account is inactive, please wait...');
                    window.open("http://smsonlinegh.com/register.php?status=complete&l=" + encodeURIComponent(jd.login), "_self");
                }

                else if (jd.response_code == 0x0003)
                {
                    $.pgsOverlay.setNotice('Account is inactive, please wait...');
                    window.open("http://smsonlinegh.com/register.php?status=maxattempts", "_self");
                }

                else
                {
                    $.pgsOverlay.close();
                    $.messager.alert(TITLE, "Unknown response was received from the server.",'error');
                    return;
                }
            },

            error: function(xhr, st, ex)
            {    
                $.pgsOverlay.close();
                $.messager.alert(TITLE, "Authentication could not be completed: " +
                     ajaxErr(xhr, st, ex), 'error');
                return;
            }
        });
    }
});

</script>

   <script type="text/javascript">
$(function()
{
	var TITLE = 'Contact Us';
	
	$('#cf-submit').click(function(e)
	{
        e.preventDefault();
		var name  = $('#cf-name').val();
		var email = $('#cf-email').val();
		var subj  = $('#cf-subject').val();
		var msg   = $('#cf-msg').val();
        var phone = $('#cf-phone').val();
		
		try
		{
			validate(name,phone,email, subj, msg);
            
            if (phone == null)
                phone = '';
            
			submitMsg(name,phone,email, subj, msg);
		}
		
		catch (ex)
		{
			$.messager.alert(TITLE, ex, 'warning');
			return;
		}
	});
	
	function submitMsg(name,phone,email, subj, msg)
	{
		$.ajax(
		{
			url:'ajax/contactus.php',
			type: 'post',
			dataType: 'json',
			data: {name:name, phone:phone, email:email, subject:subj, msg:msg},
			
			beforeSend: function()
			{
                $.pgsOverlay.show('Submitting message...');
			},
			
			complete: function()
			{
				$.pgsOverlay.close();
			},
			
			success: function(r)
			{
				if (r.response_code == 0x0000)
				{
					$.messager.alert(TITLE, r.response_msg, 'error');
					return;
				}
				
				$('#cf-name').val('');
				$('#cf-email').val('');
				$('#cf-subject').val('');
				$('#cf-msg').val('');
                $('#cf-phone').val('');
				
				$.messager.alert(TITLE, 'Your message has been successfully submitted. '+
                    'Thank you for contacting us.', 'info');
			},
			
			error: function(xhr, st, ex)
			{
				$.messager.alert(TITLE, 'Message submit error: '+ajaxErr(xhr,st,ex), 'error');
			}
		});
	}
	
	function validate(name,phone,email, subj, msg)
	{
		if (name == null || name == '')
			throw 'Please enter your name.';
            
        if (phone == null || phone == '')
            throw 'Please enter your phone number.';
		
		if (email == null || email == '')
			throw 'Please enter your email address.';
			
		if ((email != null && email != '') && !emailValid(email))
			throw 'The email address you entered is invalid.';
			
		if (subj == null || subj == '')
			throw 'Please enter the subject for your message.';
        
        if (msg == null || msg == '')
            throw 'Please enter your message or query.';
	}
});

</script>
