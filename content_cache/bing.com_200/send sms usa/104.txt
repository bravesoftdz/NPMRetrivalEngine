     <td valign="top"> <h1><b>(VBScript) Twilio: Send SMS using Basic Authentication</b></h1> <p>Demonstrates how to use Twilio to send an SMS message using Basic authentication.</p> 
      <blockquote>
       <table class="reference" width="500">
        <tbody>
         <tr>
          <td valign="top" colspan="3"> <h2 id="subhdr">Chilkat ActiveX Downloads</h2> <p class="button"><a href="http://www.chilkatsoft.com/downloads_ActiveX.asp">ActiveX for 32-bit and 64-bit Windows</a></p> </td>
         </tr> 
        </tbody>
       </table> 
      </blockquote> <pre id="exCode"><span id="keyw">Dim</span> fso, outFile
<span id="keyw">Set</span> fso = CreateObject("Scripting.FileSystemObject")
<span id="keyw">Set</span> outFile = fso.CreateTextFile("output.txt", True)

<span id="cmt">'  Demonstrates how to use Basic Authentication in a REST API call for Twilio.</span>
<span id="cmt">'  Sends an SMS text message..</span>

<span id="cmt">'  This example requires the Chilkat API to have been previously unlocked.</span>
<span id="cmt">'  See <a href="global_unlock.asp">Global Unlock Sample</a> for sample code.</span>

<span id="keyw">set</span> rest = CreateObject(<span id="lit">"<a target="_top" href="http://www.chilkatsoft.com/refdoc/xChilkatRestRef.html">Chilkat_9_5_0.Rest</a>"</span>)

<span id="cmt">'  Use Basic Authentication.</span>
<span id="cmt">'  Your Twilio Account SID is the username.</span>
<span id="cmt">'  Your Twilio Auth Token is the password.</span>
success = rest.<span id="dtyp">SetAuthBasic</span>(<span id="lit">"TWILIO_ACCOUNT_SID"</span>,<span id="lit">"TWILIO_AUTH_TOKEN"</span>)

<span id="cmt">'  Make the initial connection (without sending a request yet) to Twilio.</span>
bTls = <span id="lit">1</span>
port = <span id="lit">443</span>
bAutoReconnect = <span id="lit">1</span>
success = rest.<span id="dtyp">Connect</span>(<span id="lit">"api.twilio.com"</span>,port,bTls,bAutoReconnect)
<span id="keyw">If</span> (success &lt;&gt; <span id="lit">1</span>) <span id="keyw">Then</span>
    outFile.WriteLine(rest.<span id="dtyp">LastErrorText</span>)
    WScript.Quit
<span id="keyw">End If</span>

<span id="cmt">'  Provide the information for the SMS text message:</span>
success = rest.<span id="dtyp">AddQueryParam</span>(<span id="lit">"To"</span>,<span id="lit">"+16518675309"</span>)
success = rest.<span id="dtyp">AddQueryParam</span>(<span id="lit">"From"</span>,<span id="lit">"+15005550006"</span>)
success = rest.<span id="dtyp">AddQueryParam</span>(<span id="lit">"Body"</span>,<span id="lit">"Hey Jenny! Good luck on the bar exam!"</span>)
success = rest.<span id="dtyp">AddQueryParam</span>(<span id="lit">"MediaUrl"</span>,<span id="lit">"http://farm2.static.flickr.com/1075/1404618563_3ed9a44a3a.jpg"</span>)

<span id="cmt">'  Send the SMS text message.</span>
<span id="cmt">'  Your Twilio Account SID is part of the URI path:</span>
responseJson = rest.<span id="dtyp">FullRequestFormUrlEncoded</span>(<span id="lit">"POST"</span>,<span id="lit">"/2010-04-01/Accounts/TWILIO_ACCOUNT_SID/Messages.json"</span>)
<span id="keyw">If</span> (rest.<span id="dtyp">LastMethodSuccess</span> &lt;&gt; <span id="lit">1</span>) <span id="keyw">Then</span>
    outFile.WriteLine(rest.<span id="dtyp">LastErrorText</span>)
    WScript.Quit
<span id="keyw">End If</span>

<span id="cmt">'  When successful, the response status code will equal 201.</span>
<span id="keyw">If</span> (rest.<span id="dtyp">ResponseStatusCode</span> &lt;&gt; <span id="lit">201</span>) <span id="keyw">Then</span>
    <span id="cmt">'  Examine the request/response to see what happened.</span>
    outFile.WriteLine(<span id="lit">"response status code = "</span> &amp; rest.<span id="dtyp">ResponseStatusCode</span>)
    outFile.WriteLine(<span id="lit">"response status text = "</span> &amp; rest.<span id="dtyp">ResponseStatusText</span>)
    outFile.WriteLine(<span id="lit">"response header: "</span> &amp; rest.<span id="dtyp">ResponseHeader</span>)
    outFile.WriteLine(<span id="lit">"response body (if any): "</span> &amp; responseJson)
    outFile.WriteLine(<span id="lit">"---"</span>)
    outFile.WriteLine(<span id="lit">"LastRequestStartLine: "</span> &amp; rest.<span id="dtyp">LastRequestStartLine</span>)
    outFile.WriteLine(<span id="lit">"LastRequestHeader: "</span> &amp; rest.<span id="dtyp">LastRequestHeader</span>)
    WScript.Quit
<span id="keyw">End If</span>

<span id="cmt">'  The response is JSON.  We'll show how to get a few bits of information from it.</span>
<span id="cmt">'  A full sample JSON response is shown below..</span>

<span id="keyw">set</span> json = CreateObject(<span id="lit">"<a target="_top" href="http://www.chilkatsoft.com/refdoc/xChilkatJsonObjectRef.html">Chilkat_9_5_0.JsonObject</a>"</span>)
json.<span id="dtyp">EmitCompact</span> = <span id="lit">0</span>
success = json.<span id="dtyp">Load</span>(responseJson)

<span id="cmt">'  First show the entire JSON.</span>
outFile.WriteLine(json.<span id="dtyp">Emit</span>())

<span id="cmt">'  Now get some individual pieces of information:</span>
outFile.WriteLine(<span id="lit">"sid: "</span> &amp; json.<span id="dtyp">StringOf</span>(<span id="lit">"sid"</span>))
outFile.WriteLine(<span id="lit">"body: "</span> &amp; json.<span id="dtyp">StringOf</span>(<span id="lit">"body"</span>))
outFile.WriteLine(<span id="lit">"media: "</span> &amp; json.<span id="dtyp">StringOf</span>(<span id="lit">"subresource_uris.media"</span>))

outFile.WriteLine(<span id="lit">"Success."</span>)

<span id="cmt">'  Sample JSON response:</span>
<span id="cmt"></span>
/*

outFile.Close
