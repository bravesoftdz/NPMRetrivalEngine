    function getWeakWords(f) {
      // Create a list of words which will tend to make passwords weaker.
      var weaklings = new Array();
      siteName = "Vox Call";
      siteHost = "voxcall.com";
      weaklings = weaklings.concat(siteName.split(/\W/));
      weaklings = weaklings.concat(siteHost.split(/\W/));
      
      inputs = f.find('input');
      
      inputs.each(function(index) {
              if($(this).attr('type') == 'text') {
                  newWords = $(this).val().split(/\W/);
                  while(newWords.length > 0) {
                      newWord = newWords.pop();
                      if(newWord.length > 0) {
                          weaklings.push(newWord);
                      }
                  }
              }
          });
      return weaklings; 
    }
  
    function zxcvbnReset(pwElement) {
        if(typeof zxcvbn !== 'function') {
            // Do nothing because the zxcvbn library is not loaded yet.
            return;
        }
        parDiv = $(pwElement).parent();
        parDiv.find('.zxcvbn-score').attr('value', '');
        meter = parDiv.find('.zxcvbn-meter');
        if(meter) {
            meter.hide();
            meter.find('.zxcvbn-meter-0').hide();
            meter.find('.zxcvbn-meter-1').hide();
            meter.find('.zxcvbn-meter-2').hide();
            meter.find('.zxcvbn-meter-3').hide();
            meter.find('.zxcvbn-meter-4').hide();
        }
  
        // Recalculate and display.
        if(pwElement.value.length > 0) {
            result = zxcvbn(pwElement.value, getWeakWords($(pwElement).parents('form').first()));
            parDiv.find('.zxcvbn-score').attr('value', result.score);
            if(meter) {
                meter.show();
                if(result.score == 0) { meter.find('.zxcvbn-meter-0').show(); }
                if(result.score == 1) { meter.find('.zxcvbn-meter-1').show(); }
                if(result.score == 2) { meter.find('.zxcvbn-meter-2').show(); }
                if(result.score == 3) { meter.find('.zxcvbn-meter-3').show(); }
                if(result.score == 4) { meter.find('.zxcvbn-meter-4').show(); }
            }
        }
    }
      
    $('.zxcvbn').keyup(function() {
            return zxcvbnReset(this);

            return zxcvbnReset(this);

  function zxcvbnAddScoreElement(index, pwElement) {
      context = $(pwElement).attr('name');
      if(context.indexOf('[') > -1) {
          context = context.split("[");
          context.pop();
          context = context.join('[');
          name = context + '[zxcvbn]';
      } else {
          name = 'zxcvbn';
      }
      $(pwElement).after('<input class="zxcvbn-score" type="hidden" name="' + name + '" value="">');
  }
