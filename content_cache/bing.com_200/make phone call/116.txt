        else {
            if (window.msCommonShell) {
                if (!sur.user.isAuthenticated) {
                    localeInfoForNav = sur.LocaleValue;
                    loadShell();
                }
            }
            else {
                window.onShellReadyToLoad = function () {
                    window.onShellReadyToLoad = null;
                    if (!sur.user.isAuthenticated) {
                        localeInfoForNav = sur.LocaleValue;
                        loadShell();
                    }
                }
            }
        }
