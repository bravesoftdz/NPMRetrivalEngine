           <p><strong>Sending sms through Way2Sms in java</strong><br> Sms (Short Message Serive) is a text oriented service. It is a powerful mode of communication.<br> On daily basis you will be receiving so many smss from so and so.</p> 
           <p>In this post will see how to send an sms in java.</p> 
           <p>Java doesnt have an api to send sms directly. But it has network api through which you can connect to sms servers directly and communicate with it.</p> 
           <p>Fortunatley we have Way2Sms (free site that allows us to send sms to any mobile with in india).<br> Pre-requisites:<br> Way2Sms account (its free)</p> 
           <p>Below is the code to send sms though way2sms</p> 
           <p>I have split the code in 3 parts.<br> credentials: holds sensitive information like username, password etc.<br> URLConnector: that connects to way2sms server, send/receive data from it.<br> Way2Sms: is the entry point that actually send credentials over connector and send/receive data from the way2sms server<br> //credentials<br> package way2sms;</p> 
           <p>import java.util.ArrayList;<br> import java.net.URLEncoder;</p> 
           <p>public class Credentials {<br> private ArrayList list = new ArrayList();</p> 
           <p>public void set(String name, String value) {<br> StringBuffer buffer = new StringBuffer();</p> 
           <p>buffer.append(name);<br> buffer.append(=);<br> buffer.append(getUTF8String(value));</p> 
           <p>add(buffer.toString());<br> }</p> 
           <p>public void append(String name, String value) {<br> StringBuffer buffer = new StringBuffer();</p> 
           <p>buffer.append(&amp;);<br> buffer.append(name);<br> buffer.append(=);<br> buffer.append(getUTF8String(value));</p> 
           <p>add(buffer.toString());<br> }</p> 
           <p>private void add(String item) {<br> list.add(item);<br> }</p> 
           <p>private String getUTF8String(String value) {<br> String encodedValue = null;</p> 
           <p>try {<br> encodedValue = URLEncoder.encode(value, UTF-8);<br> } catch(Exception exception) {<br> //<br> System.out.println(Encoding error);<br> //<br> }</p> 
           <p>return encodedValue;<br> }</p> 
           <p>public boolean isEmpty() {<br> return list.isEmpty();<br> }</p> 
           <p>public void reset() {<br> list.clear();<br> }</p> 
           <p>public String getUserCredentials() {<br> StringBuffer buffer = new StringBuffer();<br> int size = list.size();</p> 
           <p>for(int i = 0; i &lt; size; i++)<br> buffer.append(list.get(i));</p> 
           <p>return buffer.toString();<br> }<br> }</p> 
           <p>//url connector<br> package way2sms;</p> 
           <p>import java.net.URL;<br> import java.net.HttpURLConnection;<br> import java.io.DataOutputStream;<br> import java.io.InputStream;<br> import java.io.BufferedReader;<br> import java.io.InputStreamReader;</p> 
           <p>public class URLConnector {<br> private static HttpURLConnection connection;</p> 
           <p>public static void connect(String urlPath, boolean redirect, String method, String cookie, String credentials) {<br> try {<br> URL url = new URL(urlPath);<br> connection = (HttpURLConnection) url.openConnection();</p> 
           <p>connection.setInstanceFollowRedirects(redirect);</p> 
           <p>if(cookie != null)<br> connection.setRequestProperty(Cookie, cookie);</p> 
           <p>if(method != null &amp;&amp; method.equalsIgnoreCase(POST)) {<br> connection.setRequestMethod(method);<br> connection.setRequestProperty(Content-Type, application/x-www-form-urlencoded);<br> }</p> 
           <p>connection.setUseCaches (false);<br> connection.setDoInput(true);<br> connection.setDoOutput(true);</p> 
           <p>if(credentials != null) {<br> DataOutputStream wr = new DataOutputStream (connection.getOutputStream ());<br> wr.writeBytes (credentials);<br> wr.flush ();<br> wr.close ();<br> }<br> } catch(Exception exception) {<br> //<br> System.out.println(Connection error);<br> //<br> }<br> }</p> 
           <p>public static String getCookie() {<br> String cookie = null;</p> 
           <p>if(connection != null) {<br> String headerName=null;</p> 
           <p>for (int i = 1; (headerName = connection.getHeaderFieldKey(i)) != null; i++) {<br> if (headerName.equals(Set-Cookie)) {<br> cookie = connection.getHeaderField(i).split(;)[0];<br> break;<br> }<br> }<br> }</p> 
           <p>return cookie;<br> }</p> 
           <p>public static int getResponseCode() {<br> int responseCode = -1;</p> 
           <p>if(connection != null) {<br> try {<br> responseCode = connection.getResponseCode();<br> } catch(Exception exception) {<br> //<br> System.out.println(Response code error);<br> //<br> }</p> 
           <p>}</p> 
           <p>return responseCode;<br> }</p> 
           <p>public static String getResponse() {<br> StringBuffer response = new StringBuffer();</p> 
           <p>if(connection != null) {<br> try {<br> InputStream is = connection.getInputStream();<br> BufferedReader rd = new BufferedReader(new InputStreamReader(is));</p> 
           <p>String line = null;<br> while((line = rd.readLine()) != null) {<br> response.append(line);<br> response.append(\r);<br> }</p> 
           <p>rd.close();<br> } catch(Exception exception) {<br> //<br> System.out.println(Response error);<br> //<br> }<br> }</p> 
           <p>return response.toString();<br> }</p> 
           <p>public static String getErrorMessage() {<br> StringBuffer errorMessage = new StringBuffer();</p> 
           <p>if(connection != null) {<br> try {<br> InputStream is = connection.getErrorStream();<br> BufferedReader rd = new BufferedReader(new InputStreamReader(is));</p> 
           <p>String line = null;<br> while((line = rd.readLine()) != null) {<br> errorMessage.append(line);<br> errorMessage.append(\r);<br> }</p> 
           <p>rd.close();<br> } catch(Exception exception) {<br> //<br> System.out.println(Error in getting error message);<br> //<br> }<br> }</p> 
           <p>return errorMessage.toString();<br> }</p> 
           <p>public static void disconnect() {<br> if(connection != null)<br> connection.disconnect();<br> }<br> }</p> 
           <p>//Way2Sms (entry point)<br> package way2sms;public class Way2Sms {<br> private static int responseCode = -1;<br> private static String userCredentials = null;<br> private static String cookie = null;<br> private static String actionStr = null;<br> private static Credentials credentials = new Credentials();</p> 
           <p>public static void main(String[] args) {<br> login(your mobile no, password);<br> sendSMS(receiversMobNo, msg);<br> System.out.println(Message has been sent successfully!);<br> }</p> 
           <p>private static void login(String uid, String pwd) {<br> credentials.set(username, uid); //mobile no<br> credentials.append(password, pwd); //password<br> credentials.append(button, Login);<br> userCredentials = credentials.getUserCredentials();</p> 
           <p>URLConnector.connect(<a href="http://site5.way2sms.com/Login1.action&amp;#8221" rel="nofollow">http://site5.way2sms.com/Login1.action&amp;#8221</a>;, false, POST, null, userCredentials);<br> cookie = URLConnector.getCookie();<br> responseCode = URLConnector.getResponseCode();<br> if(responseCode != 302 &amp;&amp; responseCode != 200)<br> exit(URLConnector.getErrorMessage());<br> URLConnector.disconnect();<br> }</p> 
           <p>private static void getActionString() {<br> URLConnector.connect(<a href="http://site5.way2sms.com/jsp/InstantSMS.jsp&amp;#8221" rel="nofollow">http://site5.way2sms.com/jsp/InstantSMS.jsp&amp;#8221</a>;, false, GET, cookie, null);<br> responseCode = URLConnector.getResponseCode();<br> if(responseCode == 302 || responseCode == 200) {</p> 
           <p>String str = URLConnector.getResponse();</p> 
           <p>String aStr = name=\Action\ id=\Action\;<br> int aStrLen = aStr.length();</p>
