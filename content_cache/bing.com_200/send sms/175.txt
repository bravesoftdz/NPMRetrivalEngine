        var dest_details = {};

        deferredScriptQueue.push(function() {
            getCallCosts('');

        function getCallCosts(prefix) {
            if (prefix) {
                prefix = '/' + prefix;
            }

            $.ajax({
                type: 'GET',
                dataType: 'json',
                url: '//controlpanel.voipfone.co.uk/api/srv?callCostCalculator' + prefix,
                success: function(x, status, xhr) {
                    if (x[0] == "ok") {
                        $.each(x[1]['callCostCalculator'+prefix], function(index, value) {
                            var destination = value[0];
                            var cost = value[1];
                            var con_cost = value[2];
                            var min_cost = value[3];
                            var minimum_cost = min_cost ? ((min_cost / 1.2) * 100).toFixed(0) + ' pence' : 'Free';
                            var connection_cost = con_cost ? ((con_cost / 1.2) * 100).toFixed(0) + ' pence' : 'Free';
                            var price = parseFloat((((cost + value[5]) / 1.2) * 100).toFixed(1));
                            dest_details[destination] = {};
                            dest_details[destination]['destname'] = destination;
                            dest_details[destination]['minimum_cost'] = minimum_cost;
                            dest_details[destination]['connection_cost'] = connection_cost;
                            dest_details[destination]['price'] = price;
                            dest_details[destination]['ppc'] = value[4];
                            dest_details[destination]['ppm'] = value[5];
                            dest_details[destination]['ppm60'] = value[6];
                            dest_details[destination]['gate'] = value[7];
                        });
                        if(prefix)
                        {
                            if(x[1]['callCostCalculator'+prefix].length > 0)
                            {
                                $('#message').html('');
                                loadTable(dest_details[x[1]['callCostCalculator'+prefix][0][0]]);
                            }
                            else 
                            {
                                $('#price').hide();
                                $('#price08').hide();
                                $('#message').html('Sorry, we cannot find that number. Please try again.');
                            }
                        }
                        else
                        {
                            loadList();
                        }
                    }
                }
            });
        }

        function loadList() {
            var a = []
            $.each(dest_details, function(d, c) {
                a.push('<option value="' + c + '">' + d + '</option>');
            });
            $('#dest').children().last().remove();
            $('#dest').change(function() {
                var e = $(this);
                if (e.val() != '') {
                    var selectedOption = dest_details[e.find('option:selected').text()];
                    $('#message').html('');
                    loadTable(selectedOption);
                }
            });
            $('#dest').append(a.join(''));
            $('#checkprefix').removeAttr('disabled');
        }

        function loadTable(selectedOption) {
            var getCharges = generateCharges(selectedOption["ppm"], selectedOption["ppc"], selectedOption["ppm60"], selectedOption["price"]);
            var access_charge = getCharges[0];
            var service_charge = getCharges[1];

            if (selectedOption["gate"].indexOf('banned') > -1)
            {
                $('#destname').html(selectedOption["destname"] + "<br><em style='display: block; padding:5px;'>You won't be able to call this number from Voipfone as we don't allow calls to the majority of premium rate numbers.</em>");
                $('#price #cost').text('N/A');
                $('#cost_con').text('N/A');
                $('#cost_min').text('N/A');
                $('#cost_pm').hide();
                $('#price').show();
                $('#price08').hide();
            }
            else
            {
                if (access_charge && service_charge) {
                    $('#price08 #destname08').text(selectedOption['destname']);
                    $('#service').html(service_charge);
                    $('#access').html(access_charge);
                    $('#price08').show();
                    $('#price').hide();
                } else {
                    var cost = selectedOption["price"];

                    if (cost == 0) {
                        cost = 'Free';
                        $('#cost_pm').hide();
                    } else {
                        $('#cost_pm').show();
                    }
                    $('#destname').text(selectedOption['destname']);
                    $('#price #cost').text(cost);
                    $('#cost_con').text(selectedOption['connection_cost']);
                    $('#cost_min').text(selectedOption['minimum_cost']);
                    $('#price').show();
                    $('#price08').hide();
                }
            }
        }

        function loadPrefix() {
            var prefix = $('#prefix').val();
            if (prefix) {
                if (prefix.length > 2) {
                    // Deal with Americans who don't bother with the + before the 1
                    if ((prefix.substr(0, 1) == '1') && ((prefix.substr(1, 1) == '-') || (prefix.substr(1, 1) == ' '))) {
                        prefix = "001" + prefix.substr(2);
                    }

                    // Convert any leading + into 00
                    if (prefix.substr(0, 1) == '+') {
                        prefix = '00' + prefix.substr(1);
                    }

                    // Remove all non-numeric characters
                    prefix = prefix.replace(/\D/g, '');

                    // Convert UK numbers into 0044 international format
                    if (prefix.substr(0, 2) != '00') {
                        if (prefix.substr(0, 2) == '01' || prefix.substr(0, 2) == '02' || prefix.substr(0, 2) == '03' || prefix.substr(0, 2) == '05' || prefix.substr(0, 2) == '07' || prefix.substr(0, 2) == '08' || prefix.substr(0, 2) == '09') {
                            prefix = '0044' + prefix.substr(1);
                        } else {
                            prefix = prefix;
                        }
                        if (prefix.substr(0, 5) == '00442') {
                            prefix = '00442';
                        }
                    }

                    var dest = '';
                    var price = 'N/A';
                    var result = false;
                    if ((!result) && prefix.length > 0 && !isNaN(prefix)) {
                        getCallCosts(prefix);
                    }
                }
            }
            return false;
        }

        function generateCharges(ppm, ppc, ppm60, cost) {
            var access_charge = '';
            var service_charge = '';
            var price_per_call = ((ppc / 1.2) * 100).toFixed(1);
            var price_per_minute_after_60s = ((ppm60 / 1.2) * 100).toFixed(1);

            if ((ppc > 0) || (ppm > 0) || (ppm60 > 0)) {
                access_charge = '7 pence per minute (ex VAT)';

                if (ppc > 0) {
                    if (ppm > 0) {
                        service_charge = price_per_call + ' pence per call, plus ' + cost + ' pence per minute (ex VAT)';
                    } else if (ppm60 > 0) {
                        service_charge = price_per_call + ' pence per call, plus ' + price_per_minute_after_60s + ' pence per minute after the first 60 seconds (ex VAT)';
                    } else {
                        service_charge = price_per_call + ' pence per call (ex VAT)';
                    }
                } else {
                    if (ppm > 0) {
                        service_charge = cost + ' pence per minute (ex VAT)';
                    } else if (ppm60 > 0) {
                        service_charge = price_per_minute_after_60s + ' pence per minute after the first 60 seconds (ex VAT)';
                    } else {
                        service_charge = 'Free';
                    }
                }
            }
            return [access_charge, service_charge];
        }

function isValidEmail(email) {
    var valid = true;
    var parts = email.split("@");
    if (parts.length == 2) {
        var name = parts[0];
        var domain = parts[1];
        if (name.length == 0) {
            valid = false;
        }
        if (name.indexOf('.') == 0 || name.lastIndexOf('.') == name.length - 1) {
            valid = false;
        }
        if (domain.indexOf('.') == 0 || domain.lastIndexOf('.') == domain.length - 1) {
            valid = false;
        }
        if (domain.split('.').length <= 1) {
            valid = false;
        }
    } else {
        valid = false;
    }
    return valid;
}
