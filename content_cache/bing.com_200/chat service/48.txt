                    <h3 class="post-title entry-title" itemprop="name"> Membuat Aplikasi Client Server Chating Menggunakan Java </h3> 
                    <div class="post-header"> 
                     <div class="post-header-line-1"></div>

                    <div class="post-body entry-content" id="post-body-7291452080800069278" itemprop="articleBody">
                      Program ini dibuat menggunakan NetBeans IDE 6.9, terdiri dari 5 packages. Sekarang saya akan membuat packages yang pertama dengan nama “Chat” (tanpa tanda kutip). Di dalam packages Chat terdapat 2 class java, yaitu ChatDisplay.java dan ChatRoom.java
                     <br> Sekarang kita buat dulu class ChatDisplay.java
                     <span id="more-339"></span>
                     <br> =====================================
                     <br> package Chat;
                     <br> import javax.swing.JTextArea;
                     <br> import javax.swing.JScrollPane;
                     <br> import javax.swing.JPanel;
                     <br> import java.awt.BorderLayout;
                     <br> public class ChatDisplay extends JPanel {
                     <br> private JTextArea chat;
                     <br> public ChatDisplay() {
                     <br> setLayout(new BorderLayout());
                     <br> chat=new JTextArea();
                     <br> chat.setEditable(false);
                     <br> chat.setLineWrap(true);
                     <br> chat.setWrapStyleWord(true);
                     <br> add(new JScrollPane(chat));
                     <br> }
                     <br> public synchronized void updateChat(String msg) {
                     <br> chat.append(“\n” + msg);
                     <br> chat.setCaretPosition(chat.getText().length());
                     <br> }
                     <br> }
                     <br> =========================================
                     <br> Setelah itu kita buat class ChatRoom.java
                     <br> =========================================
                     <br> package Chat;
                     <br> import User.UserDisplay;
                     <br> import javax.swing.JFrame;
                     <br> import javax.swing.JSplitPane;
                     <br> public class ChatRoom extends JFrame {
                     <br> private ChatDisplay chatDisplay;
                     <br> protected UserDisplay userDisplay;
                     <br> public ChatRoom() {
                     <br> super(“by jogjavasia.com”);
                     <br> chatDisplay=new ChatDisplay();
                     <br> userDisplay=new UserDisplay();
                     <br> JSplitPane splitPane=new JSplitPane(JSplitPane.HORIZONTAL_SPLIT, true);
