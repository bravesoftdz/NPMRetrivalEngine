         <h2>Prerequisites</h2>

package info.java.tips.regex;

import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class RegexDemo {
	// Visa card numbers start with a 4. New cards have 16 digits. Old cards have 13.
	private static final String VISA_REGEX = "^4[0-9]{12}(?:[0-9]{3})?$";

	// MasterCard numbers start with the numbers 51 through 55. All have 16 digits.
	private static final String MASTER_CARD_REGEX = "^5[1-5][0-9]{14}$";
	
	// American Express card numbers start with 34 or 37 and have 15 digits.
	private static final String AMERICAN_EXPRESS_REGEX = "^3[47][0-9]{13}$";
	
	// Diners Club card numbers begin with 300 through 305, 36 or 38. All have 14 digits.
	// There are Diners Club cards that begin with 5 and have 16 digits.
	// These are a joint venture between Diners Club and MasterCard, and
	// should be processed like a MasterCard.
	private static final String DINERS_CLUB_REGEX = "^3(?:0[0-5]|[68][0-9])[0-9]{11}$";

	// Discover card numbers begin with 6011 or 65. All have 16 digits.
	private static final String DISCOVER_REGEX = "^6(?:011|5[0-9]{2})[0-9]{12}$";

	// JCB cards beginning with 2131 or 1800 have 15 digits. JCB cards beginning with 35 have 16 digits.
	private static final String JCB_REGEX = "^(?:2131|1800|35\\d{3})\\d{11}$";

	public static void main(String[] args) {

		validate("4234567890123456", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX,
				DISCOVER_REGEX, JCB_REGEX);
		
		validate("4234567890123", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("5234567890123456", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("343456789012345", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("38345678901234", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("37345678901234", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("6534567890123456", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
		
		validate("180056789012345", VISA_REGEX, MASTER_CARD_REGEX, AMERICAN_EXPRESS_REGEX, DINERS_CLUB_REGEX, DISCOVER_REGEX, JCB_REGEX);
	}

	private static void validate(String input, String... regexes) {
		boolean flg = false;
		for (String regex : regexes) {
			Pattern pattern = Pattern.compile(regex);

			Matcher matcher = pattern.matcher(input);

			boolean found = false;
			if (matcher.find()) {
				found = true;				
				flg = true;
			}

			if (found) {
				switch (regex) {
				case VISA_REGEX:
					if (input.length() == 16) {
						System.out.println(input + " is new VISA card with 16 digits");
					} else if (input.length() == 13){
						System.out.println(input + " is old VISA card with 13 digits");
					}
					break;
				case MASTER_CARD_REGEX:
						System.out.println(input + " is MASTER_CARD card");
						break;	
				case AMERICAN_EXPRESS_REGEX:
					System.out.println(input + " is AMERICAN EXPRESS card");
					break;	
				case DINERS_CLUB_REGEX:
					System.out.println(input + " is DINERS CLUB card");
					break;
				case DISCOVER_REGEX:
					System.out.println(input + " is DISCOVER card");
					break;
				case JCB_REGEX:
					System.out.println(input + " is JCB card");
					break;
				}
			} 
		}
		
		if (!flg) {
			System.out.println(input + " is unknown Card Type");
		}
	}
