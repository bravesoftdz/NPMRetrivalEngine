                              I have not tested your entire array of words, but this should get you started.
                             <br> 
                             <br> NOTES:
                             <br> 1. You did not pass a word to 'translate()' for it to check
                             <br> 2. You did not have a place to store the returned value of the translation
                             <br> 3. You should not use reserved words as variables (ie, text as a function parameter)

                              <pre class="bbcode_code" style="height:372px;">&lt;html&gt;
&lt;h2&gt; English to Spanish Translator &lt;/h2&gt;
&lt;head&gt;
&lt;script type = "text/javascript"&gt;

PHRASES = [
  ["hello", "hola"],
  ["hi, hola"],
  ["how are you", "como estas"],
  ["sorry", "lo siento"],
  ["thank you", "gracias"],
  ["please", "por favor"],
  ["how are you doing today"],
  ["como estas hoy"],
  ["I love you", "te amo"],
  ["hey, eh"],
  ["welcome", "bienvenido"],
  ["my name is", "mi nombre es"],
  ["good morning", "buenos dias"],
  ["goodbye","adios"],
  ["i don't know", "no se"],
  ["help", "ayudame"],
  ["happy birthday", "feliz cumpleanos"]
];

function capitalize (str) { return str.charAt(0).toUpperCase() + str.substring(1); }

function translate (txt) {  
  for (var i=0; i&lt;PHRASES.length; i++) {
    var toReplace = new RegExp("\\b" + PHRASES[i][0] + "\\b", "i");
    var index = txt.search(toReplace);
    while (index != -1) {
      if (txt.charAt(index) &gt;= "A" &amp;&amp; txt.charAt(index) &lt;= "Z") {
        txt = txt.replace(toReplace, capitalize(PHRASES[i][1]));
      } else {
        txt = txt.replace(toReplace,PHRASES[i][1]);
      }
      index = txt.search(toReplace);
    }
  }
  return txt;
}
&lt;/script&gt;

&lt;/head&gt;

&lt;p&gt;
&lt;input type="button" value="Translate"
 onClick = "document.getElementById('outText').value = translate(document.getElementById('inText').value)" /&gt;
&lt;p&gt;
&lt;input type="text" name="outText" id="outText" value="" /&gt;
