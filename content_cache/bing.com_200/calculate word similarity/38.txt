    <p style="margin:0 0.1em; font-size:0.95em;">A polygon is a 2 dimensional (2D) plane figure with atleast three sides and angles. The word 'poly' means 'many' and 'gon' means 'angle'. It is a closed shape and are made of straight lines. There are many types of polygons with each having its own property. This online polygon calculator helps to calculate perimeter and area (using length of the side, radius of circumcircle, apothem). Enter the values and the calculator will provide you with the area and perimeter of polygon.</p>

function isNum(args)
{
args = args.toString();

if (args.length == 0)
return false;

for (var i = 0; i<args.length; i++)
{
	if ((args.substring(i,i+1) < '0' || args.substring(i, i+1) > '9') && args.substring(i, i+1) != '.')
		{
		return false;
		}
}

return true;

}


function area(varr,c)
{
	var s = document.getElementById('len').value ;
	var n = document.getElementById('no').value ;

	if(isNum(s) && isNum(n))
	{
		document.getElementById('are').value  = Math.round((((Math.pow(s,2)) * n)/(4 * (Math.tan(Math.PI/n))))*1000)/1000;
	}
	else
	{
		if(c == '1' && !isNum(s))
		{
			s = s.substring(0,s.length-1);

		{
			n = n.substring(0,n.length-1);
			document.getElementById('no').value  = n;
		}
	}
}
