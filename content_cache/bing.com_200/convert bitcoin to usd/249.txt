	var periodData = null;

	function drawPeriodChart()
	{	
		var margin = {top: 20, right: 1, bottom: 30, left: 1},
	    width = $("#periodChart").width() - margin.left - margin.right,
	    height = $("#periodChart").height() - margin.top - margin.bottom;

		var parseDate = d3.time.format("%Y-%m-%d %H").parse;
		var x = d3.time.scale().range([0, width]);
		var y = d3.scale.linear().range([height, 0]);
		var yVolume = d3.scale.linear().range([height, 0]);
		
		var xAxis = d3.svg.axis().scale(x).orient("bottom");
		var yAxis = d3.svg.axis().scale(y).orient("left");
		var yAxisVolume = d3.svg.axis().scale(yVolume).orient("right");

		// EMA 20
		var ema20 = d3.svg.line()
			.defined(function(d) { return d[9] != null; })
		    .x(function(d) { return x(d[0]); })
		    .y(function(d) { return y(d[9]); });

		// volume
		var areaVolume = d3.svg.area()
			.interpolate("basis")
		    .x(function(d) { return x(d[0]); })
		    .y0(height)
		    .y1(function(d) { return yVolume(d[6]); });

		// clear old graphs
		$("#periodChart svg").remove();

		// show new graph
		var svg = d3.select("#periodChart").append("svg")
		  .attr("width", width 	+ margin.left + margin.right)
	      .attr("height", height + margin.top + margin.bottom)
	 	.append("g")
	      .attr("transform", "translate(" + margin.left + "," + margin.top + ")");

		// show data
		var data = periodData;
		data.forEach(function(d) {
		    d[0] = parseDate(d[0]);
		});
		  
		x.domain(d3.extent(data, function(d) { return d[0]; }));
		y.domain([d3.min(data, function(d) { return d[1]; }), d3.max(data, function(d) { return d[4]; })]); 
		yVolume.domain([0, d3.max(data, function(d) { return d[6] * 3; })]);

		svg.append("g")
	      .attr("class", "x axis")
	      .attr("transform", "translate(0," + height + ")")
	      .call(xAxis);
		
		svg.append("path")
	      .datum(data)
	      .attr("class", "periodVolume")
	      .attr("d", areaVolume);

		/* Candlestick. */
		var candleWidth = (width / data.length) - 1;

		// open - close BODY
		svg.selectAll(".rect")
		    .data(data)
		  .enter().append("rect")
		    .attr("x", function(d) { return x(d[0]) - (candleWidth / 2); })
		    .attr("y", function(d) { return y(Math.max(d[2], d[3])); })
		    .attr("width", candleWidth)
		    .attr("height",  function(d) { return Math.abs(y(d[2]) - y(d[3])); })
		    .attr("class", function(d) { return "candlestick " + (d[2] < d[3] ? "candleUp" : "candleDown"); });

		// low - high LINE
		svg.selectAll(".line")
		    .data(data)
		  .enter().append("line")
		    .attr("x1", function(d) { return x(d[0]); })
		    .attr("y1", function(d) { return y(d[1]); })
		    .attr("x2", function(d) { return x(d[0]); })
		    .attr("y2", function(d) { return y(d[4]); })
		    .attr("class", function(d) { return "candlestick " + (d[2] < d[3] ? "candleUp" : "candleDown"); });
		
		svg.append("path")
	      .datum(data)
	      .attr("class", "line ema")
	      .attr("d", ema20);
		
		svg.append("g")
		      .attr("class", "y axis")
		      .attr("transform", "translate(" + width + " ,0)")  
		      .call(yAxis)
		  .append("text")
		      .attr("y", -10)
		      .style("text-anchor", "end")
		      .text("Price USD");

		svg.append("g")
	      .attr("class", "y axis")
	      .call(yAxisVolume)
	  .append("text")
	       .attr("y", -10)
	      .text("Volume BTC");	        
	}	


	$(document).ready( function() {

			periodData = data;
			drawPeriodChart();
		});
	});
