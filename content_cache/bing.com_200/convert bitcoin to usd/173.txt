    function regExpSearch(text) {
        var search_cate = '' + text;
        try {
            search_cate = search_cate.replace(searchpattern, '');
            search_cate = search_cate.replace(/\"|,|-|;/g, '');
            search_cate = search_cate.replace(/\\s+/g, ' ');
        } catch (I) {
            //console.log("exception in regExpSearch " + text);
        }
        return search_cate.toLowerCase();
    }

    function regExpCurrency(text) {

    }

    var realnewsxyz = function () {
        var $divs = $('.real_event');
        var $checked = $('.filter-host input:checked');
        var $checked_cate = $('.filter-cate input:checked');
        var $checked_add = $('.filter-addition input:checked');

        //alert("xyz = divs  = " + $divs.length + ", checked host = " + $checked.length + ", checked curr = " + $checked_cate.length + " addition = " + $checked_add.length);
        if ($checked.length && $checked_cate.length) {
            var selector = '';
            $($checked).each(function (index, element) {
                selector += ",[data-host~='" + element.id + "']";
            });
            var curr_selector = '';
            $($checked_cate).each(function (index, element) {
                curr_selector += ",[data-currency~='" + element.id + "']";
            });
            var selector_media = '';
            $($checked_add).each(function (index, element) {
                selector_media += ",[data-media~='" + element.id + "']";
            });

            try {
                selector = selector.substring(1);
                curr_selector = curr_selector.substring(1);
                selector_media = selector_media.substring(1);
                //alert(selector_media);
                //alert(selector);
                for (var i = 0, count = $divs.length; i < count; i++) {
                    $($divs.get(i)).hide();
                    if (selector_media === '') {
                        if ($($divs.get(i)).is(selector) && $($divs.get(i)).is(curr_selector)) {
                            $($divs.get(i)).show();
                        }
                    } else {
                        if ($($divs.get(i)).is(selector) && $($divs.get(i)).is(curr_selector) && $($divs.get(i)).is(selector_media)) {
                            $($divs.get(i)).show();
                        }
                    }

                }
            } catch (I) {
                //alert(I);
                return;
            }
        } else {
            for (var i = 0, count = $divs.length; i < count; i++) {
                $($divs.get(i)).hide();
            }
        }
    };
