  <script>
function updatePrice(){
	$.get("ajax/getprices.php?currency="+encodeURIComponent($("select[name='currency']").val()), function(data){
		currentprice=data;
		if($("input[name='fiat']").is(":focus")){
			$("input[name='cyptovalue']").val(($("input[name='fiat']").val()/currentprice).toFixed(8));
		}else{
			$("input[name='fiat']").val((currentprice*$("input[name='cyptovalue']").val()).toFixed(8));
		}
  	});
}
function init(){
	updatePrice();
	self.setInterval(function(){updatePrice()},15000);
	$("input[name='fiat']").keydown(function() {
		if($("input[name='fiat']").val()==="0" || isNaN($("input[name='fiat']").val())){
			$("input[name='fiat']").val(1);
			return;
		}
	});
	$("input[name='cyptovalue']").keydown(function() {
		if($("input[name='cyptovalue']").val()==="0" || isNaN($("input[name='cyptovalue']").val())){
			$("input[name='cyptovalue']").val(1);
			return;
		}
	});
	$("input[name='fiat']").keyup(function() { 
		$("input[name='cyptovalue']").val(($("input[name='fiat']").val()/currentprice).toFixed(8));
  	});
  	$("input[name='cyptovalue']").keyup(function() {
	  	$("input[name='fiat']").val(($("input[name='cyptovalue']").val()*currentprice).toFixed(8));
  	});
  	$("select[name='currency']").change(function() {
  		updatePrice();
  	});
}
</script>
