       <h3>The API URL</h3> 
       <div class="well well-sm">
         https://apilayer.net/api/convert?from=
        <code>{from}</code>&amp;to=
        <code>{to}</code>&amp;amount=
        <code>{amount}</code> 
       </div> 
       <table class="table table-striped table-bordered table-condensed"> 
        <thead> 
         <tr> 
          <th>Variable</th> 
          <th>Explanation</th> 
         </tr> 
        </thead> 
        <tbody> 
         <tr> 
          <td><code>{from}</code></td> 
          <td>Code of currency to convert from - e.g. USD</td> 
         </tr> 
         <tr> 
          <td><code>{to}</code></td> 
          <td>Code of currency to convert to - e.g. EUR</td> 
         </tr> 
         <tr> 
          <td><code>{amount}</code></td> 
          <td>the amount to be converted - e.g. 10</td> 
         </tr> 
        </tbody> 
       </table> 
       <h3>The Server Response</h3> 
       <table class="table table-striped table-bordered table-condensed"> 
        <thead> 
         <tr> 
          <th>Property</th> 
          <th>Type</th> 
          <th>Explanation</th> 
         </tr> 
        </thead> 
        <tbody> 
         <tr> 
          <td>success</td> 
          <td><code>Boolean</code></td> 
          <td>Whether the request was successful</td> 
         </tr> 
         <tr> 
          <td>from</td> 
          <td><code>String(3)</code></td> 
          <td>The currency (3-letter code) to be converted from</td> 
         </tr> 
         <tr> 
          <td>to</td> 
          <td><code>String(3)</code></td> 
          <td>The currency (3-letter code) to be converted to</td> 
         </tr> 
         <tr> 
          <td>amount</td> 
          <td><code>Decimal</code></td> 
          <td>The amount to be <strong>converted</strong></td> 
         </tr> 
         <tr> 
          <td>timestamp</td> 
          <td><code>timestamp</code></td> 
          <td>The exact time the rates used for the conversion were collected</td> 
         </tr>

          <td><code>Decimal</code></td>

       <h3> <span class="col-sm-6 pull-right"> <select name="" class="form-control input-sm pull-right" id="example_toggle"> <option value="js">JavaScript (jQuery + JSONP)</option> <option value="php">PHP</option> <option value="python">Python</option> </select> </span> Code Examples </h3> 
       <pre class="prettyprint" data-code-example="js">
// NOTE: Sample code uses jQuery to handle jsonp
	
    var access_key = 'YOUR_ACCESS_KEY';
    var from = 'USD';
    var to = 'EUR';
    var amount = '1';

$.ajax({

	url: 'https://apilayer.net/api/convert?access_key='+access_key+'&amp;from='+from+'&amp;to='+to+'&amp;amount='+amount,
	dataType: "jsonp",
	success: function(response) {

		if (response.success) {

			alert('1 USD is worth ' + parseFloat(response.rate).toFixed(2) + ' EUR');
		}
	}
});

import urllib
import json

url = "https://apilayer.net/api/convert?access_key=YOUR_ACCESS_KEY&amp;from=USD&amp;to=EUR&amp;amount=1"

url = urllib.urlopen(url)

result = url.read()

url.close()

result = json.loads(result)

if result.success:
	print "1 USD is worth %.2f EUR" % (result.result)
