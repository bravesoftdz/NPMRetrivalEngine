(function() {
// Check if cookies are enabled
function cookiesEnabled() {
return $.cookie('check', 'valid', { expires: 1 }) && $.cookie('check') == 'valid';
}
// Set the language of the domain and redirect to the domain
function setDomainLanguage(language) {
if($.cookie("lang") != null) {
$.cookie("lang", null, { path: '/', domain: 'ch.mathworks.com'});
}
$.cookie("lang", 'lang=' + language, {path: '/', expires: 365, domain: 'ch.mathworks.com'});
}
// do not display the domain selector if querystring is not present
// or if query string has nocookie value
// or if cookies are disabled
var displayDomainSelector = true;
function detectDomain() {
var queries = {};
$.each(document.location.search.substr(1).split('&'), function(c,q){
if(q != undefined && q != "") {
var i = q.split('=');
if(i[1] != undefined) {
queries[i[0].toString()] = i[1].toString();
}
}
});
var requested_domain = (queries['requestedDomain'] != undefined) ? queries['requestedDomain'] : "";
var nocookie = (queries['nocookie'] != undefined) ? queries['nocookie'] : "";
var client = (queries['client'] != undefined) ? queries['client'] : "";
if(requested_domain == "" || nocookie == "true" || !cookiesEnabled() || client == "addons" || window.overrideDomainSelector) {
displayDomainSelector = false;
}
}
detectDomain();
function extractDomain(url) {
var domain;
//find & remove protocol (http, ftp, etc.) and get domain
if (url.indexOf("://") > -1) {
domain = url.split('/')[2];
} else {
domain = url.split('/')[0];
}
//find & remove port number
domain = domain.split(':')[0];
return domain;
}
function getDomainContent(returnContent) {
// Get the location information from the marketing site
$.ajax({
url : "/includes_content/domainSelect/location.html",
type : "get",
async: false,
success : function(data) {
$("#location_content").html(data);
var button_link = $("#button_link").text();
var current_location = $("#current_location").text();
var location_key = $("#location_key").text();
//Set the location in the text and button
$(".text_location").text(current_location);
$("#domain_select_button").text("Select " + current_location);
// if the current location is different from the browser location, update the button functionality
var browser_domain = extractDomain(window.location.href);
if(button_link.indexOf(browser_domain) == -1) {
// Get the current page link from the links within the selector
new_button_link = $("." + location_key).attr("href");
$("#domain_select_button").attr("href",new_button_link);
$("#domain_select_button").removeAttr("data-dismiss");
}
}
});
if(returnContent) {
return $('#country-unselected').html();
}
}
$(document).ready(function () {
//setLocation();
// determine if links need to be rewritten
if ( $( "#country-unselected li a" ).length ) {
$("#country-unselected li a").each(function() {
var link_url = $(this).attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$(this).attr("href", link_url);
});
}
if ( $( "li.headernav_login > a" ).length ) {
var login_url = $("li.headernav_login > a").attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$("li.headernav_login > a").attr("href", login_url);
}
if ( $( "li.mwa-nav_logout > a" ).length ) {
var login_url = $("li.mwa-nav_logout > a").attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$("li.mwa-nav_logout > a").attr("href", login_url);
}
if ( $( ".mwa-nav li a" ).length ) {
$(".mwa-nav li a").each(function() {
var link_url = $(this).attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$(this).attr("href", link_url);
});
}
$('body').on('click','.footernav_country a',function(){
getDomainContent(false);
});
// set the click function for Switzerland/English link
$('body').on('click','#swiss_english',function(){
setDomainLanguage('en');
});
// set the click function for Switzerland/French link
$('body').on('click','#swiss_french',function(){
setDomainLanguage('fr');
});
// set the click function for the Switzerland/German link
$('body').on('click','#swiss_german',function(){
setDomainLanguage('de');
});
});
// onload functionality for the selector
$(document).ready(function () {
if(displayDomainSelector) {
getDomainContent(false);
$('#country-unselected').modal('show');
}
});
})(jQuery);

          <h3 class="title" id="bvie52m-1">How Array Dimensions Affect Conversion</h3>
          <p>The term <span class="emphasis"><em>dimension</em></span> means the number of subscripts required to address the elements of an array. For example, a 5-by-1 array has one dimension, because you index individual elements using one array subscript.</p>
          <p>In converting MATLAB to Java arrays, MATLAB handles dimension in a special manner. For a MATLAB array, dimension is the number of nonsingleton dimensions in the array. For example, a 10-by-1 array has dimension 1. A 1-by-1 array has dimension 0—you cannot index into a scalar value. In Java code, the number of nested arrays determines the dimension. For example, <code>double[][]</code> has dimension 2, and <code>double</code> has dimension 0.</p>
          <p>If the number of dimensions of the Java array matches the number of dimensions in MATLAB array <code>n</code>, then the converted Java array has <code>n</code> dimensions. If the Java array has fewer than <code>n</code> dimensions, then the conversion drops singleton dimensions, starting with the first one. The conversion stops when the number of remaining dimensions matches the number of dimensions in the Java array. If the Java array has more than n dimensions, then MATLAB adds trailing singleton dimensions.</p>
         </div>

         <div itemprop="content">
          <span id="bvie2dz-1" class="anchor_target"></span>
          <h3 class="title" id="bvie2dz-1">Convert Numbers to Integer Arguments</h3>
          <p>When passing an integer type to a Java method that takes a Java integer parameter, the MATLAB conversion is the same as the Java conversion between integer types. In particular, if the integer is out-of-range, it does not fit into the number of bits of the parameter type. For out-of-range integers, MATLAB discards all lowest <code>n</code> bits. The value <code>n</code> is the number of bits in the parameter type. This conversion is unlike the conversion between MATLAB integer types, where out-of-range integers are converted to the maximum or minimum value represented by the destination type.</p>
          <p>If the argument is a floating-point number, then MATLAB does not convert it to an integer in the same manner as Java. MATLAB first converts a floating-point number to a 64-bit signed integer with the fractional part truncated. Then the number is processed as if it were an <code>int64</code> argument.</p>
          <p>A floating point number is too large to be represented in a 64-bit integer when it is (outside the range from -2<sup>63</sup>–2<sup>63</sup>. In which case, MATLAB uses the following conversions:</p>
          <ul type="disc">
           <li><p>Convert <code>int</code>, <code>short</code>, and <code>byte</code> parameter values to <code>0</code>.</p></li>
           <li><p>Convert <code>long</code> parameter values to <code>java.lang.Long.MIN_VALUE</code>.</p></li>
           <li><p>Convert <code>Inf</code> and <code>-Inf</code> values to <code>-1</code>.</p></li>
           <li><p>Convert <code>NaN</code> values to <code>0</code>.</p></li>
          </ul>
         </div>

          <h3 class="title" id="bvie2x6">Pass String Arguments</h3>
          <p>To call a Java method with an argument defined as <code>java.lang.String</code>, pass a MATLAB string or character vector. MATLAB converts the argument to a Java <code>String</code> object. You also can pass a <code>String</code> object returned by a Java method.</p>
          <p>If the method argument is an array of type <code>String</code>, then pass a string array or a cell array of character vectors. MATLAB converts the input to a Java array of <code>String</code> objects, with dimensions adjusted as described in <a href="passing-data-to-java-methods.html#bvie52m-1" class="intrnllnk">How Array Dimensions Affect Conversion</a>.</p>
         </div>

For example, MATLAB does not convert <code>double</code> to <code>java.lang.Double</code> for a parameter of type <code>Double</code>.</p>

          <h4 class="title" id="bvie4v1">Pass Array of Objects</h4>
          <p>To call a method with an argument defined as <code>java.lang.Object</code> or an array of <code>java.lang.Object</code>, pass either a Java array or a MATLAB cell array. MATLAB automatically converts the cell array elements to their Java types. A Java array is an array returned from a Java constructor. You also can construct a Java array in MATLAB using the <a href="../../matlab/ref/javaarray.html"><code>javaArray</code></a> function.</p>
          <span id="bvie5g1" class="anchor_target"></span>
          <h4 class="title" id="bvie5g1">Pass Cell Array of Java Objects</h4>
          <p>To create a cell array of Java objects, use the MATLAB syntax <code>{a1,a2,...}</code>. You index into a cell array of Java objects in the usual way, with the syntax <code>a{m,n,...}</code>. For example, create cell array <code>A</code>:</p>

A = {a1,a2}
</pre>
            </div>
           </div>
          </div>

            <div class="codeoutput">
             <pre>A =

  1×2 cell array

    [1×1 java.lang.Double]    [1×1 java.lang.Float]</pre>

         <div itemprop="content">
          <span id="bvie520-1" class="anchor_target"></span>
          <h3 class="title" id="bvie520-1">Pass Empty Matrices and Nulls</h3>
          <ul type="disc">
           <li><p>MATLAB converts an empty matrix to a Java <code>null</code>.</p></li>
           <li><p>MATLAB converts the empty MATLAB character (<code>''</code>) to an empty (not <code>null</code>) Java <code>String</code> object.</p></li>
           <li><p>Empty (<code>0</code>-length) Java arrays remain unchanged.</p></li>
           <li><p>MATLAB converts <code>&lt;missing&gt;</code> values in strings to <code>null</code>.</p></li>
          </ul>
         </div>
         <div class="clearfix"></div>
         <div itemprop="content">
          <span id="bviubld" class="anchor_target"></span>
          <h3 class="title" id="bviubld">Overloaded Methods</h3>
          <p>When calling an overloaded method on a Java object, MATLAB compares the arguments you pass to the arguments defined for the methods. In this discussion, the term <span class="emphasis"><em>method</em></span> includes constructors. MATLAB determines the method to call and converts the arguments to Java types according to the Java conversion rules. For more information, see <a href="passing-data-to-java-methods.html#bvie4v1" class="intrnllnk">Pass Array of Objects</a>.</p>
          <p>When you call a Java method, MATLAB:</p>
          <ol type="1">
           <li><p><span id="bviubm_-1" class="anchor_target"></span>Checks to make sure that the object or class (for a static method) has a method by that name.</p></li>
           <li><p><span id="bviubm_-2" class="anchor_target"></span>Determines whether the invocation passes the same number of arguments of at least one method with that name.</p></li>
           <li><p><span id="bviubm_-3" class="anchor_target"></span>Makes sure that each passed argument can be converted to the Java type defined for the method.</p></li>
          </ol>
          <p>If all the preceding conditions are satisfied, then MATLAB calls the method.</p>
          <p>In a call to an overloaded method, if there is more than one candidate, MATLAB selects the one with arguments that best fit the calling arguments. First, MATLAB rejects methods that have argument types incompatible with the passed arguments. For example, if the method has a <code>double</code> argument, a <code>char</code> argument is incompatible.</p>
          <p>MATLAB then selects the method with the highest fitness value, which is the sum of the fitness values of all its arguments. The fitness value for each argument is the fitness of the base type minus the difference between the MATLAB array dimension and the Java array dimension. For information about array dimensionality, see <a href="passing-data-to-java-methods.html#bvie52m-1" class="intrnllnk">How Array Dimensions Affect Conversion</a>. If two methods have the same fitness, then the first one defined in the Java class is chosen.</p>
         </div>
