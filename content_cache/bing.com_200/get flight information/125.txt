{		
	var queryparams=url.substring(url.indexOf('?')+1);

	var params=queryparams.split('&');
	
	var methodname = methodNamePrefix || 'Get';
	for(var i=0;i<params.length;i++)

function LoadScriptAsync(scriptURL, scriptConst, callback) {
    // create a script element
    var s = document.createElement('script');
    s.type = 'text/javascript';
    s.language = "javascript";
    s.async = true;
    s.src = scriptURL;

    // add script element to the head node
    document.getElementsByTagName("head")[0].appendChild(s);

    // wait for script to be loaded
    if (scriptConst) {
        WaitForConstLoad(scriptConst, callback);
    }
}

function WaitForConstLoad(a_const, callback, tries) {
    var maxWaitTime = 20000, // ms
        tryEvery = 100;      // ms

    if (window[a_const]) {
        callback();
    } else {
        tries = tries || 0;
        if (tries < maxWaitTime / tryEvery) {
            setTimeout(function() {
                WaitForConstLoad(a_const, callback, tries || 1);
            }, tryEvery);
        } else {
            // Give up.
        }
    }

                if (activeToolTip != null) hideToolTip(activeToolTip);

				case 'Relative':
					div.style.top = position[0];
					div.style.left = position[1];
				break 
				case 'Absolute':
					
					div.style.top = DeltaTop + 'px';
					div.style.left = DeltaLeft + 'px'; 
				break
				default:      
					div.style.top = position[0];
					div.style.left = position[1]; 
				}

                    activeToolTip = null;

				function getToolTipPosition(evt,ToolTipID,Location)
				{
				var position = [];
				var elPos = getAbsoluteElementLocation(document.getElementById(evt));
					switch(Location)
					{
					case 'TopLeft':
						 
						position[0] = (elPos[0] - document.getElementById(ToolTipID).offsetHeight)+'px';
						position[1] = (elPos[1] - document.getElementById(ToolTipID).offsetWidth)+'px';
						return position;
					break    
					case 'TopRight':
						
						position[0] = (elPos[0] - document.getElementById(ToolTipID).offsetHeight)+'px';
						position[1] = (elPos[1] + document.getElementById(evt).offsetWidth)+'px';
						return position;
					break
					case 'BottomRight':
						
						position[0] = (elPos[0] + document.getElementById(evt).offsetHeight)+'px';
						position[1] = (elPos[1] + document.getElementById(evt).offsetWidth)+'px';
						return position;	
					break
					case 'BottomLeft':
						
						position[0] = (elPos[0] + document.getElementById(evt).offsetHeight)+'px';
						position[1] = (elPos[1] - document.getElementById(ToolTipID).offsetWidth)+'px';
						return position;	
					break
					default:
						
						position[0] = (elPos[0] + document.getElementById(evt).offsetHeight)+'px';
						position[1] = (elPos[1] + document.getElementById(evt).offsetWidth)+'px';
						return position;
					}

			function getAbsoluteElementLocation(el)
				{
				var pos = [];
				var box;
				var parent;
				if (el.getBoundingClientRect) { // IE
					box = el.getBoundingClientRect();
					var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
					var scrollLeft = document.documentElement.scrollLeft || document.body.scrollLeft;
			 
					return [box.top + scrollTop,box.left + scrollLeft];
				}
				else 
					{ 
					pos = [el.offsetTop,el.offsetLeft];
					parent = el.offsetParent;
					if (parent != el) {
						while (parent) {
						pos[0] += parent.offsetTop;
						pos[1] += parent.offsetLeft;
						parent = parent.offsetParent;
						}
					}

				}
				return pos;
				}
