  function airportUrlOptimized( cod, label ){
    var ret = label
     .split( "/" ).join( "-" )
     .split( ", " ).join( "-" )
     .split( " (" ).join( "-(" )
     .split( " " ).join( "_" )
     .split( "'" ).join( "" ) ;
    return "/airport/" + cod + "-" + ret + "/" ;
  }
  
function airportCoo( cod ){

  + airportUrlOptimized( cod, a[ 0 ] ) + "delay"
  + "\" target=\"_top\" title=\"" + cod + "\" alt=\"" + cod + "\" />\n" ;
