function isMiniSearchFieldSet( sMiniSearchField, sMiniSearchID )
{
    var sAnyOption = "-1"
    var sValue = sAnyOption;
    var oElement = document.getElementById( sMiniSearchField + sMiniSearchID );
    sMiniSearchField
    //alert( 'isMiniSearchFieldSet:- Field: ' + sMiniSearchField + sMiniSearchID );
    //alert( 'isMiniSearchFieldSet:- tagName: ' + oElement.tagName );
    if ( oElement != null )
    {
      if ( "SELECT" == oElement.tagName )
      {
        if ( oElement.options[oElement.selectedIndex] )
          sValue = oElement.options[oElement.selectedIndex].value;
      }
      else if ( "INPUT" == oElement.tagName)
         sValue = oElement.value;
    }
    if ( sAnyOption.match( sValue ) )
        return false;
    else
        return true;
}

  function validatePartyMakeup( sMiniSearchID )
  {
    var bValid = true;
    var nRoomCount = $('#PackageAvailability'+sMiniSearchID).val();
    for(i = 0; i < nRoomCount; i++ )
    {
      var nChildCount = $('#NoOfChildren'+i+sMiniSearchID).val();
      for( iChild = 0; iChild < nChildCount; iChild++ )

        var nChildAge = $('#'+minorAgeDiv( 'NoOfChildren', nRoomCount, iChild, sMiniSearchID ));
        if ( nChildAge = '-1' )
        {
          bValid = false;
        }
      }

      var nInfantCount = $('#NoOfInfants'+i+sMiniSearchID).val();
      for( iInfant = 0; iInfant < nInfantCount; iInfant++ )

  function summaryText( sMiniSearchID )
  {
    var nRoomCount = $('#PackageAvailability'+sMiniSearchID).val();
    var sSummaryText = summaryTestHelper( nRoomCount, 'Room', 'Rooms' ) + ' ';
    var nTotalAdultCount = 0;
    var nTotalChildCount = 0;
    
    for(i = 0; i < nRoomCount; i++ )
    {
      var nAdultCount = parseInt( $('#NoOfAdults'+i+sMiniSearchID).val() );
      var nChildCount = parseInt( $('#NoOfChildren'+i+sMiniSearchID).val() );
      var nInfantCount = parseInt( $('#NoOfInfants'+i+sMiniSearchID).val() );
      
      nTotalAdultCount = nTotalAdultCount + nAdultCount;
      nTotalChildCount = nTotalChildCount + nChildCount;
      nTotalChildCount = nTotalChildCount + nInfantCount;
    }
    
    if ( nTotalAdultCount > 0 && nTotalChildCount > 0 )
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalAdultCount, 'Adult', 'Adults' ) +', '
        + summaryTestHelper( nTotalChildCount, 'Child', 'Children' ) + ')';
    }
    else if ( nTotalAdultCount > 0 )
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalAdultCount, 'Adult', 'Adults' ) + ')';
    }
    else
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalChildCount, 'Child', 'Children' ) + ')';
    }
    
    //alert( sSummaryText );
    $('#guests'+sMiniSearchID).val( sSummaryText );
    return sSummaryText;
  }
  
  function summaryTestHelper( nCount, sSingular, sPlural )
  {
    if ( ! nCount )
      return '';
    else if ( nCount == 1 )
      return nCount + ' ' + sSingular;
    else
      return nCount + ' ' + sPlural;
  }

  function validatePartyMakeup( sMiniSearchID )
  {
    var bValid = true;
    var nRoomCount = $('#PackageAvailability'+sMiniSearchID).val();
    for(i = 0; i < nRoomCount; i++ )
    {
      var nChildCount = $('#NoOfChildren'+i+sMiniSearchID).val();
      for( iChild = 0; iChild < nChildCount; iChild++ )

        if ( nChildAge = '-1' )

        }
      }

      var nInfantCount = $('#NoOfInfants'+i+sMiniSearchID).val();
      for( iInfant = 0; iInfant < nInfantCount; iInfant++ )

          bValid = false;
        }

  }
  
  function summaryText( sMiniSearchID )
  {
    var nRoomCount = $('#PackageAvailability'+sMiniSearchID).val();
    var sSummaryText = summaryTestHelper( nRoomCount, 'Room', 'Rooms' ) + ' ';
    var nTotalAdultCount = 0;
    var nTotalChildCount = 0;
    
    for(i = 0; i < nRoomCount; i++ )
    {
      var nAdultCount = parseInt( $('#NoOfAdults'+i+sMiniSearchID).val() );
      var nChildCount = parseInt( $('#NoOfChildren'+i+sMiniSearchID).val() );
      var nInfantCount = parseInt( $('#NoOfInfants'+i+sMiniSearchID).val() );
      
      nTotalAdultCount = nTotalAdultCount + nAdultCount;
      nTotalChildCount = nTotalChildCount + nChildCount;
      nTotalChildCount = nTotalChildCount + nInfantCount;
    }
    
    if ( nTotalAdultCount > 0 && nTotalChildCount > 0 )
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalAdultCount, 'Adult', 'Adults' ) +', '
        + summaryTestHelper( nTotalChildCount, 'Child', 'Children' ) + ')';
    }
    else if ( nTotalAdultCount > 0 )
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalAdultCount, 'Adult', 'Adults' ) + ')';
    }
    else
    {
      sSummaryText = sSummaryText + '(' 
        + summaryTestHelper( nTotalChildCount, 'Child', 'Children' ) + ')';
    }
    
    //alert( sSummaryText );
    $('#guests'+sMiniSearchID).val( sSummaryText );
    return sSummaryText;
  }
  
  function summaryTestHelper( nCount, sSingular, sPlural )
  {
    if ( ! nCount )
      return '';
    else if ( nCount == 1 )
      return nCount + ' ' + sSingular;
    else
      return nCount + ' ' + sPlural;
  }

function loadResort($sRegionCode, $sDestinationCode, $sDestinationName, $sDestinationImage, $sDestinationMapURL, $sRegionType, $sRegionName, $sDestinationType) {
	//alert( "calling load resort" );

	var $oDestMap = $("div#destcontent-" + $sRegionCode);
	//alert ( '$oDestMap: ' + $oDestMap );
	
	var $oDestMapContainer = $oDestMap.closest("div#destmap");
	var $sURLTabs = "url(//holidays.kulula.com/" + $sDestinationImage + ")";
	$oDestMapContainer.css("background-image", $sURLTabs );

	
	var $sDestinationPrompt = '';
	$oDestMap.empty();
	$oDestMap.append('<div id="destloading"><img src="//holidays.kulula.com/images/branding/kulula-hol/holding/k-loader-mini-v00.gif" /></div>');
	var sMyURL = '/specific/kulula-hol/includes/destinations-ajax.jsp;jsessionid=3F04E8B7C3D2BA8E379601F581B90D7C?Search_Destination_4=' + $sDestination
        + '&destinationName='+ $sDestinationName
        + '&destinationPrompt='+ $sDestinationPrompt
        + '&regionCode='+ $sRegionCode
        + '&regionType='+ $sRegionType
        + '&regionName='+ $sRegionName
        + '&destinationType='+ $sDestinationType
        + '&isSecurePage='+ "false";
	//log( sMyURL );
	//alert( sMyURL );
	$oDestMap.load("/specific/kulula-hol/includes/destinations-ajax.jsp;jsessionid=3F04E8B7C3D2BA8E379601F581B90D7C", { Search_Destination_4 : $sDestination,
		destinationName: $sDestinationName,
		destinationPrompt: $sDestinationPrompt,
		regionCode: $sRegionCode,
		regionType: $sRegionType,
		regionName: $sRegionName,
		destinationType: $sDestinationType,
        isSecurePage: "false" },
        function() {
        	
        	$oMapLink = $("#map-link-" + $sRegionCode);
            $oMapLinkParent = $("#maplink" );
         //alert( "sDestinationMapURL" + $sDestinationMapURL );
        	if ( typeof( $sDestinationMapURL ) == "undefined"
                || $sDestinationMapURL.length == 0
                || $sDestinationMapURL == "null" )
        	{
        	   //alert( "sDestinationMapURL" + $sDestinationMapURL );
        		//$oMapLink.attr("href", "#" );
              //  $oMapLink.attr("style", "display:none" );
                //$oMapLinkParent.attr("style", "display:none" );
        	}
        	else
        	{
                //alert( $sDestinationMapURL );
        		$oMapLink.attr("href", $sDestinationMapURL + "&iframe=true&width=100%&height=100%" );
        		$oMapLink.attr("style", "display:block" );
                $oMapLinkParent.attr("style", "display:block" );
        		$("a[rel^='prettyPhoto']").prettyPhoto({theme:'light_rounded',show_title: false});
        	}
        }

    );
}
