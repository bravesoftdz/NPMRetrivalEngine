  <script>
    var jsRoot = "./";

    var isBootInitialized = false;
    function startApp() {
        if (isBootInitialized) {
            return;
        }

        isBootInitialized = true;
        var script = document.createElement("script");
        script.type = "text/javascript";
        script.src = jsRoot + "js/boot.js";
        script.async = true;
        document.body.appendChild(script);
    }

    function increaseProgress() {
        step++;
        progress = step/total * 100;
        progressBar.style.width = progress + "%";
    }

    var progressBar = document.getElementById("progressBar");
    var progress = 0;
    var total = 4;
    var step = 0;

    var lang;
    var urlLang = /^\?.*lang=(\w+)&?.*$/.exec(window.location.search);
    if (urlLang && urlLang[1] && urlLang[1] !== "null") {
        lang = urlLang[1].toLowerCase();
    }
    if (!lang) {
        try {
            lang = localStorage.getItem("lang");
        } catch(e) {}
    }

    window._lang = lang || "en";

    if(window._lang == 'ar') {
        document.body.parentElement.setAttribute('dir', 'rtl');
        document.body.classList.add('rtl');
    }

    window._airline = "ai";
    window._iTravelWeb = false;
    window.isDEV = false;
    window._env = "PROD";

    var require = {
        waitSeconds: 300,
        baseUrl: jsRoot,
        "i18n": {
            airline: "ai",
            defaultLocale: "en",
            locale: window._lang
        }
    };
    

    window._isOnLoadCalled = false;
    window.onload = function () {
        window._isOnLoadCalled = true;
    };

    if (window.applicationCache) {
        var appCacheTimeout = setTimeout(startApp, 3000);

        window.applicationCache.onchecking = function() {
            clearTimeout(appCacheTimeout);
        };
        window.applicationCache.onnoupdate = startApp;
        window.applicationCache.oncached = startApp;
        window.applicationCache.onobsolet = startApp;
        window.applicationCache.onerror = startApp;

        window.applicationCache.onupdateready = function() {
            try {
                window.applicationCache.swapCache();
            } catch(err) {}

            startApp();
        };
    } else {
        startApp();
    }
</script>
