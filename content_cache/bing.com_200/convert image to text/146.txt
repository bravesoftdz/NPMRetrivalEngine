&lt;script type="text/javascript"&gt;
var input_id = "bin_text";
var answer_id = "answer";

function convertToASCII() {
	var bin_text = document.getElementById(input_id);
	var answer = document.getElementById(answer_id);

	if (!answer) {
		alert("Error: No element with id \""+answer_id+"\".");
		return;
	}
	if (bin_text)
		var text = bin_text.value;
	else {
		error("No element with id \""+input_id+"\".");
		return;
	}
	var divisible = text.length % 8;
	var nonBinary = /[^0|1]/.test(text);
	if (text.length &gt; 0 &amp;&amp; divisible == 0 &amp;&amp; !nonBinary) {
		var regex = /[0|1]{8}/g;
		var str = text.match(regex);
		var code = 0;
		var placeVal, exp, digit;
		var ascii = '';
		while (str.length &gt; 0) {
			code = 0;
			for (var i=0; i&lt;str[0].length; i++) {
				placeVal = 7-i;
				exp = Math.pow(2, i);
				digit = str[0].charAt(placeVal);
				code += exp*digit;
			}
			str.shift();
			ascii += String.fromCharCode(code);
		}
		answer.innerHTML = "&lt;p class=\"binary\"&gt;" + ascii + "&lt;/p&gt;";
	}
	else {
		error("Malformed binary.");
		return;
	}

	function error(errText) {
		answer.innerHTML = "&lt;span class=\"error\"&gt;Error: " + errText + "&lt;/span&gt;";
	}
}
&lt;/script&gt;

             <p>Something not really used in JS is the second parameter of parseInt().</p> 
             <p> You can use it to convert between bases:</p> 
             <p> eg:</p> 
             <pre><code class="language-js">// binary to decimal
parseInt('101', 2); // 5</code></pre> 
             <p>So converting between binary and ASCII would be as simple as:</p> 
             <pre><code class="language-js">// binary to unicode character
function bin2ascii(bin) {
  return String.fromCharCode(parseInt(bin, 2));
}</code></pre>
