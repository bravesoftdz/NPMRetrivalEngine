		function fnRender() {
			var numHeight = document.getElementById("numHeight");
			var numWidth = document.getElementById("numWidth");
			var numX = document.getElementById("numX");
			var numY = document.getElementById("numY");
			var numTextSize = document.getElementById("numTextSize");
			var clrText = document.getElementById("clrText");
			var clrBackground1 = document.getElementById("clrBackground1");
			var clrBackground2 = document.getElementById("clrBackground2");
			var cnvCanvas = document.getElementById("cnvCanvas");
			var txtText = document.getElementById("txtText");
			var ddlFont = document.getElementById("ddlFont");

			cnvCanvas.width = numWidth.value;
	 		cnvCanvas.height = numHeight.value;

			var ctxContext = cnvCanvas.getContext("2d");

			// Create a gradient
			var grdGradient = ctxContext.createLinearGradient(0,0,200,0);
			grdGradient.addColorStop(0,clrBackground1.value);
			grdGradient.addColorStop(1,clrBackground2.value);


			// Fill with the gradient
			ctxContext.fillStyle = grdGradient;
			ctxContext.fillRect(0,0,cnvCanvas.width,cnvCanvas.height);

			// Draw the entered text
			ctxContext.fillStyle = clrText.value;
			ctxContext.font = numTextSize.value+ "px " + ddlFont.value;

			//ctxContext.fillText(txtText.value,numX.value,numY.value);
			ctxContext.fillText(txtText.value,numX.value,numY.value);
		}
