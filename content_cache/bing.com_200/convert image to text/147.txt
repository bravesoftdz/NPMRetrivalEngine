function no_show_error(){
	var url=$("#url").val();
	if(url!=""){
		$("#p2i_htmlerror").prop("disabled","disabled");
		$("#p2i_htmlerror").prop("checked","");
		if(url.indexOf("http")<0){
			url="http://"+url;
			$("#url").val(url);
			p2i_htmlerror = 0;
		}
	}else{
		$("#p2i_htmlerror").removeProp("disabled");
		$("#p2i_htmlerror").prop("checked","checked");
	}
}

function submitIt()
{	$("#share").show();
    $('#error_msg').hide();

	var p2i_html=$.trim($("#p2i_html").val());
	if(p2i_html=="")
	{	$("#share").hide();
	    $('#error_msg').text('Please enter your HTML code.');
        $('#error_msg').show();
		return;
	}

	//begin
	loading(0);
	var size=$(".mobilesize").val();
    var chooseDevice=$(".chooseDevice").val();
    var p2i_htmlerror = $("#p2i_htmlerror").prop("checked") ? 1: 0;
	var url=$("#url").val();
	if(url!=""){
		p2i_htmlerror=0;
	}
	if(chooseDevice==3)
	{
		device=1;
	}else if(chooseDevice==16)
	{
		device=4;
	}else if(chooseDevice==4)

		device=2;
	}else if(chooseDevice==6)
	{
		device=3;
	}else if(chooseDevice==17)
	{
		device=6;
	}else
	{
		device=0;
	}
	call(p2i_html,device,size,0, p2i_htmlerror,url);
}
function call(p2i_html,device,size,forceSize, p2i_htmlerror,url)
{
	buildTimeout_end= Date.parse(new Date());
    if((1*buildTimeout_end-1*buildTimeout_begin)/1000>135)
    {	$("#share").hide();
    	$('#error_msg').text('Time out,Please angin.');
        $('#error_msg').show();
		$(".toolsDescription").css('margin-top','-155px');
		loading(2);
		return;
    }
	var t=$("[name='mobielvh']:checked").val();
	if(t==1)
	{
	   vsize=arr[size][1]+"x0";
	   h=arr[size][0];
	}else
	{
	   vsize=arr[size][0]+"x0";
	   h=arr[size][1];
	}
	$.post('/api/html_to_image', {p2i_html: p2i_html, p2i_device: device, p2i_size:vsize,p2i_url:url,flag:"mobile_emulator", p2i_htmlerror:p2i_htmlerror}, function(msg){
		if (msg == 'INVLIAD_URL') {
			$("#share").hide();
			$('#error_msg').text('Please enter a valid url.');
            $('#error_msg').show();
			loading(2);
		} else if (msg == 'TIMES_LIMIT') {
            var str ='<a href="http://www.page2images.com/contact_us" >contact us</a>';
			$("#share").hide();
		    $("#error_msg").html("Sorry. You have used 50 times free API calls today. Please try it tomorrow.");
            $('#error_msg').show();
			$(".toolsDescription").css('margin-top','-155px');
			loading(2);
		} else if (msg == 'API_ERROR') {
			$("#share").hide();
			$('#error_msg').text('Server is busying, please again!');
            $('#error_msg').show();
			$(".toolsDescription").css('margin-top','-155px');
			loading(2);
		} else {
			ret = $.parseJSON(msg);
			var t=$("[name='mobielvh']:checked").val();
			if(t==1)
		    {
				display = $('.gv_phone');
		    }else
		    {
		    	display = $('.gh_phone');
		    }

			switch(ret.status) {
			 	case 'error':
					$("#share").hide();
                    $("#error_msg").html(ret.msg);
                    $('#error_msg').show();
					$(".toolsDescription").css('margin-top','-155px');
			 		loading(2);
			 		issubmit=true;
			 		break;
			 	case 'finished':
					$("#share").show();
				 	getImageSizeAndLoad(ret.image_url,url,device,size,h,display);
					downloadAndCopy(ret.image_url);
			 		break;
			 	case 'processing':
					$("#share").show();
			 		buildSumTime = 50;
                    if(ret.estimated_need_time>50) ret.estimated_need_time=50;
			 		prolength = parseInt ((1 - ret.estimated_need_time/buildSumTime) * 100);
			 		set_progress(display, display_messages[parseInt(Math.random()*6)], prolength, 0);
			 		setTimeout(function(){call(p2i_html,device,size,forceSize,p2i_htmlerror,url);}, 5000);
			 		break;
			}
		}
	});
}

	            '</div>' +
	    	'</div>' +
	    '</div></div>'
    );
	 $('.phoneshow:visible #roundbar_'+ device).roundbar({'total': total});
	 h1=$(".phoneshow:visible").height();
	 h2=$(".phoneshow:visible .progress_main").height();
	 $(".phoneshow:visible .progress_main").css("margin-top",(h1-h2-30)/2+"px");
	 var t=$("[name='mobielvh']:checked").val();
	if(t==1)
    {
		$(".phoneshow:visible .phoneshowdiv").height(nx*1-5);
    }else
    {
    	$(".phoneshow:visible .phoneshowdiv").height(ny*1-5);
    }
}

function resizeIt()
{

    var size=$(".mobilesize").val();
    if(arr[size][0]>700)
    {
	    nx=Math.ceil(arr[size][0]/2);
	    ny=Math.ceil(arr[size][1]/2);
    }else
    {
        nx=Math.ceil(arr[size][0]);
	    ny=Math.ceil(arr[size][1]);
    }
	var t=$("[name='mobielvh']:checked").val();
	if(t==1)
    {
        $(".gv_phone").show();
		$(".toolsDescription").css('margin-top','20px');
        $(".gh_phone").hide();
        $(".gv_phone").height(nx+40);
    	$(".gv_phone img").width(ny).addClass("imageTv");
    	$(".gv_phone .phoneshowdiv").width(ny*1).height(nx*1);
    	$(".gv_phone").width(ny*1+20).height(nx*1+20);
    }else
    {
   	    $(".gv_phone").hide();
      	$(".gh_phone").show();
		$(".toolsDescription").css('margin-top','20px');
      	$(".gh_phone").height(1*ny+60);
    	$(".gh_phone img").width(nx).addClass("imageTh");
    	$(".gh_phone").width(nx*1+20).height(ny*1+20);
    	$(".gh_phone .phoneshowdiv").width(nx*1).height(ny*1);

    }
}
