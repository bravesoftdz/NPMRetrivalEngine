(function() {
// Check if cookies are enabled
function cookiesEnabled() {
return $.cookie('check', 'valid', { expires: 1 }) && $.cookie('check') == 'valid';
}
// Set the language of the domain and redirect to the domain
function setDomainLanguage(language) {
if($.cookie("lang") != null) {
$.cookie("lang", null, { path: '/', domain: 'ch.mathworks.com'});
}
$.cookie("lang", 'lang=' + language, {path: '/', expires: 365, domain: 'ch.mathworks.com'});
}
// do not display the domain selector if querystring is not present
// or if query string has nocookie value
// or if cookies are disabled
var displayDomainSelector = true;
function detectDomain() {
var queries = {};
$.each(document.location.search.substr(1).split('&'), function(c,q){
if(q != undefined && q != "") {
var i = q.split('=');
if(i[1] != undefined) {
queries[i[0].toString()] = i[1].toString();
}
}
});
var requested_domain = (queries['requestedDomain'] != undefined) ? queries['requestedDomain'] : "";
var nocookie = (queries['nocookie'] != undefined) ? queries['nocookie'] : "";
var client = (queries['client'] != undefined) ? queries['client'] : "";
if(requested_domain == "" || nocookie == "true" || !cookiesEnabled() || client == "addons" || window.overrideDomainSelector) {
displayDomainSelector = false;
}
}
detectDomain();
function extractDomain(url) {
var domain;
//find & remove protocol (http, ftp, etc.) and get domain
if (url.indexOf("://") > -1) {
domain = url.split('/')[2];
} else {
domain = url.split('/')[0];
}
//find & remove port number
domain = domain.split(':')[0];
return domain;
}
function getDomainContent(returnContent) {
// Get the location information from the marketing site
$.ajax({
url : "/includes_content/domainSelect/location.html",
type : "get",
async: false,
success : function(data) {
$("#location_content").html(data);
var button_link = $("#button_link").text();
var current_location = $("#current_location").text();
var location_key = $("#location_key").text();
//Set the location in the text and button
$(".text_location").text(current_location);
$("#domain_select_button").text("Select " + current_location);
// if the current location is different from the browser location, update the button functionality
var browser_domain = extractDomain(window.location.href);
if(button_link.indexOf(browser_domain) == -1) {
// Get the current page link from the links within the selector
new_button_link = $("." + location_key).attr("href");
$("#domain_select_button").attr("href",new_button_link);
$("#domain_select_button").removeAttr("data-dismiss");
}
}
});
if(returnContent) {
return $('#country-unselected').html();
}
}

}
if ( $( "li.mwa-nav_logout > a" ).length ) {
var login_url = $("li.mwa-nav_logout > a").attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$("li.mwa-nav_logout > a").attr("href", login_url);
}
if ( $( ".mwa-nav li a" ).length ) {

}

          <h2 class="title r2017b" itemprop="title content" id="example-TextDetectionExample">Automatically Detect and Recognize Text in Natural Images</h2>
          <div class="pull-right examples_short_list" style="width:250px; margin-left: 10px; margin-bottom: 10px;">
           <div data-pane="metadata" style="margin-bottom: 0px;" class="panel metadata_container">
            <div class="panel-body metadata_content" style="padding-top:0px; padding-bottom:0px;">
             <div class="row">
              <div class="col-xs-12">
               <a class="btn btn_secondary btn-block" href="matlab:openExample('vision/TextDetectionExample')">Open Script</a>
              </div>
             </div>
            </div>
           </div>
          </div>
          <div itemscope itemtype="http://www.mathworks.com/help/schema/MathWorksDocPage/Example" itemprop="example" class="em_example">
           <meta itemprop="exampleid" content="vision-ex54620063">
           <meta itemprop="exampletitle" content="Automatically Detect and Recognize Text in Natural Images">

          <span id="TextDetectionExample" class="anchor_target"></span> 
          <p class="shortdesc">This example shows how to detect regions in an image that contain text. This is a common task performed on unstructured scenes. Unstructured scenes are images that contain undetermined or random scenarios. For example, you can detect and recognize text automatically from captured video to alert a driver about a road sign. This is different than structured scenes, which contain known scenarios where the position of text is known beforehand.</p> 
          <p>Segmenting text from an unstructured scene greatly helps with additional tasks such as optical character recognition (OCR). The automated text detection algorithm in this example detects a large number of text region candidates and progressively removes those less likely to contain text.</p> 
          <div class="procedure">
           <h3 class="title" id="d119e513">Step 1: Detect Candidate Text Regions Using MSER<span id="TextDetectionExample-1" class="anchor_target"></span></h3>
           <p>The MSER feature detector works well for finding text regions [1]. It works well for text because the consistent color and high contrast of text leads to stable intensity profiles.</p>
           <p>Use the <code class="literal">detectMSERFeatures</code> function to find all the regions within the image and plot these results. Notice that there are many non-text regions detected alongside the text.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre>colorImage = imread(<span style="color:#A020F0">'handicapSign.jpg'</span>);
I = rgb2gray(colorImage);

<span style="color:#228B22">% Detect MSER regions.</span>
[mserRegions, mserConnComp] = detectMSERFeatures(I, <span style="color:#0000FF">...</span>
    <span style="color:#A020F0">'RegionAreaRange'</span>,[200 8000],<span style="color:#A020F0">'ThresholdDelta'</span>,4);

figure
imshow(I)
hold <span style="color:#A020F0">on</span>
plot(mserRegions, <span style="color:#A020F0">'showPixelList'</span>, true,<span style="color:#A020F0">'showEllipses'</span>,false)
title(<span style="color:#A020F0">'MSER regions'</span>)
hold <span style="color:#A020F0">off</span>
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e527" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_01.png" alt=""></p>
            </div>
           </div>
           <h3 class="title" id="d119e530">Step 2: Remove Non-Text Regions Based On Basic Geometric Properties<span id="TextDetectionExample-2" class="anchor_target"></span></h3>
           <p>Although the MSER algorithm picks out most of the text, it also detects many other stable regions in the image that are not text. You can use a rule-based approach to remove non-text regions. For example, geometric properties of text can be used to filter out non-text regions using simple thresholds. Alternatively, you can use a machine learning approach to train a text vs. non-text classifier. Typically, a combination of the two approaches produces better results [4]. This example uses a simple rule-based approach to filter non-text regions based on geometric properties.</p>
           <p>There are several geometric properties that are good for discriminating between text and non-text regions [2,3], including:</p>
           <div class="itemizedlist">
            <ul>
             <li><p>Aspect ratio</p></li>
             <li><p>Eccentricity</p></li>
             <li><p>Euler number</p></li>
             <li><p>Extent</p></li>
             <li><p>Solidity</p></li>
            </ul>
           </div>
           <p>Use <code class="literal">regionprops</code> to measure a few of these properties and then remove regions based on their property values.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Use regionprops to measure MSER properties</span>
mserStats = regionprops(mserConnComp, <span style="color:#A020F0">'BoundingBox'</span>, <span style="color:#A020F0">'Eccentricity'</span>, <span style="color:#0000FF">...</span>
    <span style="color:#A020F0">'Solidity'</span>, <span style="color:#A020F0">'Extent'</span>, <span style="color:#A020F0">'Euler'</span>, <span style="color:#A020F0">'Image'</span>);

<span style="color:#228B22">% Compute the aspect ratio using bounding box data.</span>
bbox = vertcat(mserStats.BoundingBox);
w = bbox(:,3);
h = bbox(:,4);
aspectRatio = w./h;

<span style="color:#228B22">% Threshold the data to determine which regions to remove. These thresholds</span>
<span style="color:#228B22">% may need to be tuned for other images.</span>
filterIdx = aspectRatio' &gt; 3;
filterIdx = filterIdx | [mserStats.Eccentricity] &gt; .995 ;
filterIdx = filterIdx | [mserStats.Solidity] &lt; .3;
filterIdx = filterIdx | [mserStats.Extent] &lt; 0.2 | [mserStats.Extent] &gt; 0.9;
filterIdx = filterIdx | [mserStats.EulerNumber] &lt; -4;

<span style="color:#228B22">% Remove regions</span>
mserStats(filterIdx) = [];
mserRegions(filterIdx) = [];

<span style="color:#228B22">% Show remaining regions</span>
figure
imshow(I)
hold <span style="color:#A020F0">on</span>
plot(mserRegions, <span style="color:#A020F0">'showPixelList'</span>, true,<span style="color:#A020F0">'showEllipses'</span>,false)
title(<span style="color:#A020F0">'After Removing Non-Text Regions Based On Geometric Properties'</span>)
hold <span style="color:#A020F0">off</span>
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e562" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_02.png" alt=""></p>
            </div>
           </div>
           <h3 class="title" id="d119e565">Step 3: Remove Non-Text Regions Based On Stroke Width Variation<span id="TextDetectionExample-3" class="anchor_target"></span></h3>
           <p>Another common metric used to discriminate between text and non-text is stroke width. <span class="emphasis"><em>Stroke width</em></span> is a measure of the width of the curves and lines that make up a character. Text regions tend to have little stroke width variation, whereas non-text regions tend to have larger variations.</p>
           <p>To help understand how the stroke width can be used to remove non-text regions, estimate the stroke width of one of the detected MSER regions. You can do this by using a distance transform and binary thinning operation [3].</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Get a binary image of the a region, and pad it to avoid boundary effects</span>
<span style="color:#228B22">% during the stroke width computation.</span>
regionImage = mserStats(6).Image;
regionImage = padarray(regionImage, [1 1]);

<span style="color:#228B22">% Compute the stroke width image.</span>
distanceImage = bwdist(~regionImage);
skeletonImage = bwmorph(regionImage, <span style="color:#A020F0">'thin'</span>, inf);

strokeWidthImage = distanceImage;
strokeWidthImage(~skeletonImage) = 0;

<span style="color:#228B22">% Show the region image alongside the stroke width image.</span>
figure
subplot(1,2,1)
imagesc(regionImage)
title(<span style="color:#A020F0">'Region Image'</span>)

subplot(1,2,2)
imagesc(strokeWidthImage)
title(<span style="color:#A020F0">'Stroke Width Image'</span>)
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e579" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_03.png" alt=""></p>
            </div>
           </div>
           <p>In the images shown above, notice how the stroke width image has very little variation over most of the region. This indicates that the region is more likely to be a text region because the lines and curves that make up the region all have similar widths, which is a common characteristic of human readable text.</p>
           <p>In order to use stroke width variation to remove non-text regions using a threshold value, the variation over the entire region must be quantified into a single metric as follows:</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Compute the stroke width variation metric</span>
strokeWidthValues = distanceImage(skeletonImage);
strokeWidthMetric = std(strokeWidthValues)/mean(strokeWidthValues);
</pre>
             </div>
            </div>
           </div>
           <p>Then, a threshold can be applied to remove the non-text regions. Note that this threshold value may require tuning for images with different font styles.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Threshold the stroke width variation metric</span>
strokeWidthThreshold = 0.4;
strokeWidthFilterIdx = strokeWidthMetric &gt; strokeWidthThreshold;
</pre>
             </div>
            </div>
           </div>
           <p>The procedure shown above must be applied separately to each detected MSER region. The following for-loop processes all the regions, and then shows the results of removing the non-text regions using stroke width variation.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Process the remaining regions</span>
<span style="color:#0000FF">for</span> j = 1:numel(mserStats)

    regionImage = mserStats(j).Image;
    regionImage = padarray(regionImage, [1 1], 0);

    distanceImage = bwdist(~regionImage);
    skeletonImage = bwmorph(regionImage, <span style="color:#A020F0">'thin'</span>, inf);

    strokeWidthValues = distanceImage(skeletonImage);

    strokeWidthMetric = std(strokeWidthValues)/mean(strokeWidthValues);

    strokeWidthFilterIdx(j) = strokeWidthMetric &gt; strokeWidthThreshold;

<span style="color:#0000FF">end</span>

<span style="color:#228B22">% Remove regions based on the stroke width variation</span>
mserRegions(strokeWidthFilterIdx) = [];
mserStats(strokeWidthFilterIdx) = [];

<span style="color:#228B22">% Show remaining regions</span>
figure
imshow(I)
hold <span style="color:#A020F0">on</span>
plot(mserRegions, <span style="color:#A020F0">'showPixelList'</span>, true,<span style="color:#A020F0">'showEllipses'</span>,false)
title(<span style="color:#A020F0">'After Removing Non-Text Regions Based On Stroke Width Variation'</span>)
hold <span style="color:#A020F0">off</span>
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e598" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_04.png" alt=""></p>
            </div>

           <h3 class="title" id="d119e601">Step 4: Merge Text Regions For Final Detection Result<span id="TextDetectionExample-7" class="anchor_target"></span></h3>
           <p>At this point, all the detection results are composed of individual text characters. To use these results for recognition tasks, such as OCR, the individual text characters must be merged into words or text lines. This enables recognition of the actual words in an image, which carry more meaningful information than just the individual characters. For example, recognizing the string 'EXIT' vs. the set of individual characters {'X','E','T','I'}, where the meaning of the word is lost without the correct ordering.</p>
           <p>One approach for merging individual text regions into words or text lines is to first find neighboring text regions and then form a bounding box around these regions. To find neighboring regions, expand the bounding boxes computed earlier with <code class="literal">regionprops</code>. This makes the bounding boxes of neighboring text regions overlap such that text regions that are part of the same word or text line form a chain of overlapping bounding boxes.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Get bounding boxes for all the regions</span>
bboxes = vertcat(mserStats.BoundingBox);

<span style="color:#228B22">% Convert from the [x y width height] bounding box format to the [xmin ymin</span>
<span style="color:#228B22">% xmax ymax] format for convenience.</span>
xmin = bboxes(:,1);
ymin = bboxes(:,2);
xmax = xmin + bboxes(:,3) - 1;
ymax = ymin + bboxes(:,4) - 1;

<span style="color:#228B22">% Expand the bounding boxes by a small amount.</span>
expansionAmount = 0.02;
xmin = (1-expansionAmount) * xmin;
ymin = (1-expansionAmount) * ymin;
xmax = (1+expansionAmount) * xmax;
ymax = (1+expansionAmount) * ymax;

<span style="color:#228B22">% Clip the bounding boxes to be within the image bounds</span>
xmin = max(xmin, 1);
ymin = max(ymin, 1);
xmax = min(xmax, size(I,2));
ymax = min(ymax, size(I,1));

<span style="color:#228B22">% Show the expanded bounding boxes</span>
expandedBBoxes = [xmin ymin xmax-xmin+1 ymax-ymin+1];
IExpandedBBoxes = insertShape(colorImage,<span style="color:#A020F0">'Rectangle'</span>,expandedBBoxes,<span style="color:#A020F0">'LineWidth'</span>,3);

figure
imshow(IExpandedBBoxes)
title(<span style="color:#A020F0">'Expanded Bounding Boxes Text'</span>)
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e615" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_05.png" alt=""></p>
            </div>
           </div>
           <p>Now, the overlapping bounding boxes can be merged together to form a single bounding box around individual words or text lines. To do this, compute the overlap ratio between all bounding box pairs. This quantifies the distance between all pairs of text regions so that it is possible to find groups of neighboring text regions by looking for non-zero overlap ratios. Once the pair-wise overlap ratios are computed, use a <code class="literal">graph</code> to find all the text regions "connected" by a non-zero overlap ratio.</p>
           <p>Use the <code class="literal">bboxOverlapRatio</code> function to compute the pair-wise overlap ratios for all the expanded bounding boxes, then use <code class="literal">graph</code> to find all the connected regions.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Compute the overlap ratio</span>
overlapRatio = bboxOverlapRatio(expandedBBoxes, expandedBBoxes);

<span style="color:#228B22">% Set the overlap ratio between a bounding box and itself to zero to</span>
<span style="color:#228B22">% simplify the graph representation.</span>
n = size(overlapRatio,1);
overlapRatio(1:n+1:n^2) = 0;

<span style="color:#228B22">% Create the graph</span>
g = graph(overlapRatio);

<span style="color:#228B22">% Find the connected text regions within the graph</span>
componentIndices = conncomp(g);
</pre>
             </div>
            </div>
           </div>
           <p>The output of <code class="literal">conncomp</code> are indices to the connected text regions to which each bounding box belongs. Use these indices to merge multiple neighboring bounding boxes into a single bounding box by computing the minimum and maximum of the individual bounding boxes that make up each connected component.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Merge the boxes based on the minimum and maximum dimensions.</span>
xmin = accumarray(componentIndices', xmin, [], @min);
ymin = accumarray(componentIndices', ymin, [], @min);
xmax = accumarray(componentIndices', xmax, [], @max);
ymax = accumarray(componentIndices', ymax, [], @max);

<span style="color:#228B22">% Compose the merged bounding boxes using the [x y width height] format.</span>
textBBoxes = [xmin ymin xmax-xmin+1 ymax-ymin+1];
</pre>
             </div>
            </div>
           </div>
           <p>Finally, before showing the final detection results, suppress false text detections by removing bounding boxes made up of just one text region. This removes isolated regions that are unlikely to be actual text given that text is usually found in groups (words and sentences).</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre><span style="color:#228B22">% Remove bounding boxes that only contain one text region</span>
numRegionsInGroup = histcounts(componentIndices);
textBBoxes(numRegionsInGroup == 1, :) = [];

<span style="color:#228B22">% Show the final text detection result.</span>
ITextRegion = insertShape(colorImage, <span style="color:#A020F0">'Rectangle'</span>, textBBoxes,<span style="color:#A020F0">'LineWidth'</span>,3);

figure
imshow(ITextRegion)
title(<span style="color:#A020F0">'Detected Text'</span>)
</pre>
             </div>
            </div>
           </div>
           <div class="informalfigure">
            <div id="d119e646" class="mediaobject">
             <p><img src="../../examples/vision/win64/TextDetectionExample_06.png" alt=""></p>
            </div>
           </div>
           <h3 class="title" id="d119e649">Step 5: Recognize Detected Text Using OCR<span id="TextDetectionExample-11" class="anchor_target"></span></h3>
           <p>After detecting the text regions, use the <code class="literal">ocr</code> function to recognize the text within each bounding box. Note that without first finding the text regions, the output of the <code class="literal">ocr</code> function would be considerably more noisy.</p>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeinput">
              <pre>ocrtxt = ocr(I, textBBoxes);
[ocrtxt.Text]
</pre>
             </div>
            </div>
           </div>
           <div class="code_responsive">
            <div class="programlisting">
             <div class="codeoutput">
              <pre>
ans =

    'HANDICIXPPED
     PARKING
     SPECIAL PLATE
     REQUIRED
     UNAUTHORIZED
     VEHICLES
     MAY BE TOWED
     AT OWNERS
     EXPENSE
     
     '

</pre>
             </div>
            </div>
           </div>
           <p>This example showed you how to detect text in an image using the MSER feature detector to first find candidate text regions, and then it described how to use geometric measurements to remove all the non-text regions. This example code is a good starting point for developing more robust text detection algorithms. Note that without further enhancements this example can produce reasonable results for a variety of other images, for example, posters.jpg or licensePlates.jpg.</p>
           <h3 class="title" id="d119e667">References<span id="TextDetectionExample-13" class="anchor_target"></span></h3>
           <p>[1] Chen, Huizhong, et al. "Robust Text Detection in Natural Images with Edge-Enhanced Maximally Stable Extremal Regions." Image Processing (ICIP), 2011 18th IEEE International Conference on. IEEE, 2011.</p>
           <p>[2] Gonzalez, Alvaro, et al. "Text location in complex images." Pattern Recognition (ICPR), 2012 21st International Conference on. IEEE, 2012.</p>
           <p>[3] Li, Yao, and Huchuan Lu. "Scene text detection via stroke width." Pattern Recognition (ICPR), 2012 21st International Conference on. IEEE, 2012.</p>
           <p>[4] Neumann, Lukas, and Jiri Matas. "Real-time scene text localization and recognition." Computer Vision and Pattern Recognition (CVPR), 2012 IEEE Conference on. IEEE, 2012.</p>
