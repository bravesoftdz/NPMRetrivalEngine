<span id="keyw">public class</span> ChilkatExample {

  <span id="keyw">static</span> {
    <span id="keyw">try</span> {
        System.loadLibrary(<span id="lit">"chilkat"</span>);
    } <span id="keyw">catch</span> (UnsatisfiedLinkError e) {
      System.err.println(<span id="lit">"Native code library failed to load.\n"</span> + e);

    }
  }

  <span id="keyw">public static void</span> main(String argv[])
  {
    <span id="dtyp"><a target="_top" href="http://www.chilkatsoft.com/refdoc/javaCkMhtRef.html">CkMht</a></span> mht = <span id="keyw">new</span> <a target="_top" href="http://www.chilkatsoft.com/refdoc/javaCkMhtRef.html">CkMht</a>();

    <span id="keyw">boolean</span> success = mht.<span id="dtyp">UnlockComponent</span>(<span id="lit">"Anything for 30-day trial"</span>);
    <span id="keyw">if</span> (success != <span id="lit">true</span>) {
        System.out.println(mht.<span id="dtyp">lastErrorText</span>());
        <span id="keyw">return</span>;
    }

    <span id="cmt">//  Download a URL into an in-memory MHT web archive contained</span>
    <span id="cmt">//  in a string variable:</span>
    <span id="keyw">String</span> mhtDoc;
    mhtDoc = mht.<span id="dtyp">getMHT</span>(<span id="lit">"http://www.gopackaging.com/"</span>);
    <span id="keyw">if</span> (mht.<span id="dtyp">get_LastMethodSuccess</span>() != <span id="lit">true</span>) {
        System.out.println(mht.<span id="dtyp">lastErrorText</span>());
        <span id="keyw">return</span>;
    }

    <span id="cmt">//  Now extract the HTML and embedded objects:</span>
    <span id="keyw">String</span> unpackDir = <span id="lit">"/Users/chilkat/temp/"</span>;
    <span id="keyw">String</span> htmlFilename = <span id="lit">"gopackaging.html"</span>;
    <span id="keyw">String</span> partsSubdir = <span id="lit">"objects"</span>;
    <span id="cmt">//  Extract to /Users/chilkat/temp/gopackaging.html.</span>
    <span id="cmt">//  images and other embedded objects are placed in</span>

    }


  }
