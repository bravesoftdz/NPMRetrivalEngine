<!-- Cookie -->
function setCookie(cname,cvalue,exdays){
	var d = new Date();
	d.setTime(d.getTime()+(exdays*24*60*60*1000));
	var expires = "expires="+d.toGMTString();
	document.cookie = cname + "=" + cvalue + "; " + expires+ ";path=/";
}
function getCookie(cname){
	var name = cname + "=";
	var ca = document.cookie.split(';');
	for(var i=0; i<ca.length; i++){
	  var c = ca[i].trim();
	  if (c.indexOf(name)==0) return c.substring(name.length,c.length);
	}
	return "";
}

function streamPublish(extFile){
	var cookieValue=getCookie("sharetodownload");
	if(cookieValue!='done'){
		if(cookieValue=="" || cookieValue=='null'){
			setCookie('sharetodownload', "", -1);
			setCookie('sharetodownload', 2, 7);
			if(!$( "#hiddenShare" ).hasClass('hiddenBox')) $( "#hiddenShare" ).addClass('hiddenBox');
		}else if(cookieValue>0){
			if(extFile=='wav' || extFile=='mp3' || extFile=='aif' || extFile=='zip'){
				if(cookieValue%2==0){
					if($( "#hiddenShare" ).hasClass('hiddenBox')) $( "#hiddenShare" ).removeClass('hiddenBox');
					$('html, body').animate({scrollTop: $("#hiddenShare").offset().top}, 500);
				}
				cookieValue++;
				setCookie('sharetodownload', "", -1);
				setCookie('sharetodownload', cookieValue, 7);
			}			
		}
	}

}

function checkLinkedinShares(){
	var newnumShareLinkeind=0;
	var oldnumShareLinkeind=getCookie("linkedinshares");	
	setCookie('linkedinshares', "", -1);
	setCookie('sharetodownload', "", -1);
	setCookie('sharetodownload', 'done', 7);
	if(!$( "#hiddenShare" ).hasClass('hiddenBox')) $( "#hiddenShare" ).addClass('hiddenBox');
	setTimeout(function(){ 
		$.ajax({
			dataType: "jsonp",
			url: "http://www.linkedin.com/countserv/count/share",
			data: {
				callback: "?",
				format: "jsonp",
				url: "http://www.gosoundtrack.com/"
			}
		}).done(function(data) {
			newnumShareLinkeind=data.count;
			if(newnumShareLinkeind>oldnumShareLinkeind) {
				setCookie('linkedinshares', "", -1);
				setCookie('sharetodownload', "", -1);
				setCookie('sharetodownload', 'done', 7);
				if(!$( "#hiddenShare" ).hasClass('hiddenBox')) $( "#hiddenShare" ).addClass('hiddenBox');
				ga('send', 'social', 'Linkedin', 'Share', window.location.href);
			}else {
				//dopo 40 secondi verifica se è aumentato il counter degli share di linkedin
				setCookie('sharetodownload', "", -1);
				setCookie('linkedinshares', newnumShareLinkeind, 7);
			}
			
		});
	}, 40000);
	return false;
}
